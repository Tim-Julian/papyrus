78  	// FIXME : Something should exist in API to do that
79  	private String print(Diagnostic diagnostic) {
80  		List<Diagnostic> children = diagnostic.getChildren();
81  		StringBuilder stringBuilder = new StringBuilder(diagnostic.getMessage());
82  		for (Diagnostic diagnosticChildren : children) {
83  			stringBuilder.append("\n"); //$NON-NLS-1$
84  			stringBuilder.append(diagnosticChildren.getMessage());
85  		}
86  		return stringBuilder.toString();
87  	}
