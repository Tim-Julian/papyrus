66  	/**
67  	 * Test method for {@link org.eclipse.papyrus.sysml14.service.types.command.SetNestedPathCommand#canExecute()}.
68  	 */
69  	@Test
70  	public void testCanExecute() {
71  		Model model = SysMLResource.createSysMLModel(new ResourceSetImpl());
72  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());
73  
74  		CreateElementRequest invalidRequest = new CreateElementRequest(ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_BLOCK));
75  		SetNestedPathCommand invalidCommand = new SetNestedPathCommand("Set a invalid nested path", invalidRequest.getNewElement(), invalidRequest, new ArrayList<Property>(), SetNestedPathCommand.NESTED_SOURCE);
76  		Assert.assertFalse("A nested path cannot be set in a Block", invalidCommand.canExecute());
77  		
78  		CreateRelationshipRequest validRequest = new CreateRelationshipRequest(ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_CONFORM));
79  		Connector connector = UMLFactory.eINSTANCE.createConnector();
80  		connector.createEnd();
81  		connector.createEnd();
82  		SetNestedPathCommand setNestedPathCommand = new SetNestedPathCommand("Set a correct nested path", connector, validRequest, new ArrayList<Property>(), SetNestedPathCommand.NESTED_SOURCE);
83  		Assert.assertTrue("An empty nested path can be set to any connector", setNestedPathCommand.canExecute());
84  	}

86  	/**
87  	 * Test method for {@link org.eclipse.papyrus.sysml14.service.types.command.SetNestedPathCommand#doExecuteWithResult(org.eclipse.core.runtime.IProgressMonitor, org.eclipse.core.runtime.IAdaptable)}.
88  	 * 
89  	 * @throws InterruptedException
90  	 * @throws RollbackException
91  	 */
92  	@Test
93  	public void testDoExecuteWithResultIProgressMonitorIAdaptable() throws InterruptedException, RollbackException {
94  		ResourceSetImpl resourceSet = new ResourceSetImpl();
95  		Model model = SysMLResource.createSysMLModel(resourceSet);
96  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());
97  		
98  		IElementType type = ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_CONFORM);
99  		Connector connector = UMLFactory.eINSTANCE.createConnector();
100 		connector.createEnd();
101 		connector.createEnd();	
102 		Class sourceClass = model.createOwnedClass("C1", false);
103 		Class targetClass = model.createOwnedClass("C2", false);
104 		targetClass.getOwnedConnectors().add(connector);
105 		Property createOwnedAttribute = targetClass.createOwnedAttribute("P1", null);
106 
107 		TransactionalEditingDomain.Factory.INSTANCE.createEditingDomain(resourceSet);
108 		InternalTransactionalEditingDomain editingDomain = (InternalTransactionalEditingDomain) TransactionUtil.getEditingDomain(model);
109 		
110 		CreateRelationshipRequest validRequest = new CreateRelationshipRequest(editingDomain, sourceClass, targetClass, type);
111 		List<Property> nestedPath = new ArrayList<>();
112 		nestedPath.add(createOwnedAttribute);
113 		SetNestedPathCommand setNestedPathCommand = new SetNestedPathCommand("setNestedPathCommand", connector, validRequest, nestedPath, SetNestedPathCommand.NESTED_SOURCE);
114 		Assert.assertTrue("The setNestedPathCommand should be executable", setNestedPathCommand.canExecute());
115 
116 		InternalTransaction startTransaction = editingDomain.startTransaction(false, null);
117 		editingDomain.getCommandStack().execute(GMFtoEMFCommandWrapper.wrap(setNestedPathCommand));
118 		startTransaction.commit();
119 		Object result = setNestedPathCommand.getCommandResult().getReturnValue();
120 		Assert.assertTrue("The result of a seNestedPathCommand should be a Connector", result instanceof Connector);
121 		NestedConnectorEnd nestedConnectorEnd = UMLUtil.getStereotypeApplication(utils.getSourceConnectorEnd((Connector) result), NestedConnectorEnd.class);	
122 		Assert.assertEquals("The new nested Path should be the one in parameter of the command",nestedPath, nestedConnectorEnd.getPropertyPath());
123 	}
