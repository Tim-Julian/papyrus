42  /**
43   * Test on DeriveReqt Supplier constraint
44   *
45   */
46  @SuppressWarnings("nls")
47  public class DeriveReqtSupplierModelConstraintTest {
48  
49  	public static final String CONSTRAINT_ID = "constraint.derivereqt.supplier"; //$NON-NLS-1$
50  
51  	/**
52  	 * Test method for the constraint
53  	 * 
54  	 * @throws InterruptedException
55  	 * @throws RollbackException
56  	 */
57  	@Test
58  	public void checkRuleValidation() throws InterruptedException, RollbackException {
59  		ResourceSetImpl resourceSet = new ResourceSetImpl();
60  		Model model = SysMLResource.createSysMLModel(resourceSet);
61  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());	
62  		
63  		Class supplierRequirement = model.createOwnedClass("SupplierRequirement", false);
64  		StereotypeApplicationHelper.getInstance(null).applyStereotype(supplierRequirement, RequirementsPackage.eINSTANCE.getRequirement());
65  		Abstraction deriveReqtValid = UMLFactory.eINSTANCE.createAbstraction();
66  		model.getPackagedElements().add(deriveReqtValid);
67  		deriveReqtValid.getSuppliers().add(supplierRequirement);
68  		deriveReqtValid.getClients().add(model.createOwnedClass("uselessClient1", false));
69  		StereotypeApplicationHelper.getInstance(null).applyStereotype(deriveReqtValid, RequirementsPackage.eINSTANCE.getDeriveReqt());
70  			
71  		Class supplierClass = model.createOwnedClass("SupplierClass", false);
72  		Abstraction deriveReqtInvalid = UMLFactory.eINSTANCE.createAbstraction();
73  		model.getPackagedElements().add(deriveReqtInvalid);		
74  		deriveReqtInvalid.getSuppliers().add(supplierClass);
75  		deriveReqtInvalid.getClients().add(model.createOwnedClass("uselessClient2", false));
76  		StereotypeApplicationHelper.getInstance(null).applyStereotype(deriveReqtInvalid, RequirementsPackage.eINSTANCE.getDeriveReqt());		
77  
78  		TransactionalEditingDomain.Factory.INSTANCE.createEditingDomain(resourceSet);
79  		InternalTransactionalEditingDomain editingDomain = (InternalTransactionalEditingDomain) TransactionUtil.getEditingDomain(model);
80  		
81  		final ValidateModelCommand validateModelCommand = new ValidateModelCommand(model, new UMLDiagnostician());
82  		
83  		InternalTransaction startTransaction = editingDomain.startTransaction(false, null);
84  		editingDomain.getCommandStack().execute(GMFtoEMFCommandWrapper.wrap(validateModelCommand));
85  		startTransaction.commit();
86  		// check that the constraint exist
87  		ConstraintRegistry instance = ConstraintRegistry.getInstance();
88  		IConstraintDescriptor descriptor = instance.getDescriptor(Activator.PLUGIN_ID, CONSTRAINT_ID);
89  		Assert.assertNotNull("Constraint is missing", descriptor);
90  
91  		List<Diagnostic> constraintDiagnosticList = DiagnosticUtil.findDiagnosticBySource(validateModelCommand.getDiagnostic(), Activator.PLUGIN_ID + "." + CONSTRAINT_ID);
92  				
93  		Assert.assertEquals("The rule "+CONSTRAINT_ID+" should not trigger an issue for this use case ", 0, DiagnosticUtil.filterDiagnosticsByElement(constraintDiagnosticList, deriveReqtValid).size());
94  		Assert.assertEquals("The rule "+CONSTRAINT_ID+" should trigger an issue for this use case ", 1, DiagnosticUtil.filterDiagnosticsByElement(constraintDiagnosticList, deriveReqtInvalid).size());		
95  	}
96  }
