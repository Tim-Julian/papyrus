42  /**
43   * Test on Refine One Client and One Supplier constraint
44   *
45   */
46  @SuppressWarnings("nls")
47  public class TestCaseParameterVerdictKindModelConstraintTest {
48  
49  	public static final String CONSTRAINT_ID = "constraint.testcase.parameterverdictkind"; //$NON-NLS-1$
50  
51  	/**
52  	 * Test method for the constraint
53  	 * 
54  	 * @throws InterruptedException
55  	 * @throws RollbackException
56  	 */
57  	@Ignore
58  	@Test
59  	public void checkRuleValidation() throws InterruptedException, RollbackException {
60  		ResourceSetImpl resourceSet = new ResourceSetImpl();
61  		Model model = SysMLResource.createSysMLModel(resourceSet);
62  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());	
63  		
64  		
65  		Class testCaseValid = model.createOwnedClass("Class", false);
66  		StereotypeApplicationHelper.getInstance(null).applyStereotype(testCaseValid, RequirementsPackage.eINSTANCE.getTestCase());
67  		Type returnType = model.createOwnedType("VerdictKind", testCaseValid.eClass());
68  		testCaseValid.createOwnedOperation("operation", null, null, returnType);
69  		
70  		Class testCaseInvalid = model.createOwnedClass("testCaseInvalid", false);
71  		StereotypeApplicationHelper.getInstance(null).applyStereotype(testCaseInvalid, RequirementsPackage.eINSTANCE.getTestCase());
72  		Type returnTypeInvalid = model.createOwnedType("NotVerdictKind", testCaseInvalid.eClass());
73  		testCaseInvalid.createOwnedOperation("operation", null, null, returnTypeInvalid);
74  			
75  
76  		TransactionalEditingDomain.Factory.INSTANCE.createEditingDomain(resourceSet);
77  		InternalTransactionalEditingDomain editingDomain = (InternalTransactionalEditingDomain) TransactionUtil.getEditingDomain(model);
78  		
79  		final ValidateModelCommand validateModelCommand = new ValidateModelCommand(model, new UMLDiagnostician());
80  		
81  		InternalTransaction startTransaction = editingDomain.startTransaction(false, null);
82  		editingDomain.getCommandStack().execute(GMFtoEMFCommandWrapper.wrap(validateModelCommand));
83  		startTransaction.commit();
84  		// check that the constraint exist
85  		ConstraintRegistry instance = ConstraintRegistry.getInstance();
86  		IConstraintDescriptor descriptor = instance.getDescriptor(Activator.PLUGIN_ID, CONSTRAINT_ID);
87  		Assert.assertNotNull("Constraint is missing", descriptor);
88  
89  		List<Diagnostic> constraintDiagnosticList = DiagnosticUtil.findDiagnosticBySource(validateModelCommand.getDiagnostic(), Activator.PLUGIN_ID + "." + CONSTRAINT_ID);
90  				
91  		Assert.assertEquals("The rule "+CONSTRAINT_ID+" should not trigger an issue for this use case ", 0, DiagnosticUtil.filterDiagnosticsByElement(constraintDiagnosticList, testCaseValid).size());
92  		Assert.assertEquals("The rule "+CONSTRAINT_ID+" should trigger an issue for this use case ", 1, DiagnosticUtil.filterDiagnosticsByElement(constraintDiagnosticList, testCaseInvalid).size());		
93  	}
