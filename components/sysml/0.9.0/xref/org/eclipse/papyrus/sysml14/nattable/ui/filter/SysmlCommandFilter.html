32  public class SysmlCommandFilter implements ICommandFilter {

39  	private SysmlCommandFilter() {
40  		// to prevent instantiation
41  	}

43  	private List<IElementType> visibleCommands;
44  
45  	public List<IElementType> getVisibleCommands() {

47  			List<IElementType> localVisibleCommands = new ArrayList<IElementType>();
48  
49  //			localVisibleCommands.add(SysmlServiceTypeUtil.ACTOR_PART_PROPERTY);

51  			localVisibleCommands.add(ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_BLOCK));

53  			localVisibleCommands.add(ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_CONSTRAINTBLOCK));

56  //			localVisibleCommands.add(ElementTypeRegistry.getInstance().getType(SysmlServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_DIMENSION));

59  //			localVisibleCommands.add(ElementTypeRegistry.getInstance().getType(SysmlServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_FLOW_PORT_IN));

68  //			localVisibleCommands.add(ElementTypeRegistry.getInstance().getType(SysmlServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_REFERENCE_PROPERTY));
69  			localVisibleCommands.add(ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_REQUIREMENT));
