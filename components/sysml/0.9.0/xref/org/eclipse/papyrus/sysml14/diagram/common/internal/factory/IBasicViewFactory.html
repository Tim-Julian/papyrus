13  package org.eclipse.papyrus.sysml14.diagram.common.internal.factory;
14  
15  import org.eclipse.emf.ecore.EObject;
16  import org.eclipse.gmf.runtime.diagram.core.preferences.PreferencesHint;
17  import org.eclipse.gmf.runtime.diagram.core.view.factories.ViewFactory;
18  import org.eclipse.gmf.runtime.notation.View;
19  
20  /**
21   * it define the the contract in order to be able to create a serialization of a figure in the diagram
22   * it is used to allowing reuse code from GMF generated diagram
23   *
24   */
25  public interface IBasicViewFactory extends ViewFactory {
26  
27  	/**
28  	 * Create the the view, a element that represents the serialization of a figure inside the GMF framework
29  	 * 
30  	 * @param domainElement
31  	 *            in our case this a SysML element
32  	 * @param containerView
33  	 *            the container of the created view
34  	 * @param semanticHint
35  	 *            the identifier that will associated to the created view
36  	 * @param index
37  	 *            the index where will be created the view ( if you don't know -1)
38  	 * @param persisted
39  	 *            true means that the node will be serialized in the file
40  	 * @param preferencesHint
41  	 *            a constant that is specific to a diagram
42  	 * @return the created view
43  	 */
44  	View createElementView(EObject domainElement, View containerView, String semanticHint, int index, boolean persisted, PreferencesHint preferencesHint);
45  }
