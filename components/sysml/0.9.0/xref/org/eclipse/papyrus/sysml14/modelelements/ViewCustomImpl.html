31  public class ViewCustomImpl extends ViewImpl implements View {
32  
33  
34  	/**
35  	 * 7.3.2.7 View : /viewpoint: Viewpoint The viewpoint for this View is derived from the conform relationship.
36  	 * 
37  	 * @see org.eclipse.papyrus.sysml14.modelelements.internal.impl.ViewImpl#basicGetViewPoint()
38  	 * <p> See the requirement <b>{@papyrus.req org.eclipse.papyrus.sysml14#Req015}</b>. 
39  	 * @return
40  	 */
41  	@Override
42  	public Viewpoint basicGetViewPoint() {
43  		// ViewPoint is related to View via a Conform dependency
44  		Viewpoint viewPoint = null;
45  		Conform conform = null;
46  
47  		Class viewBaseClass = getBase_Class();
48  		if (viewBaseClass != null) {
49  			// Find Conform link
50  			EList<Generalization> generalizations = viewBaseClass.getGeneralizations();
51  			if (generalizations != null) {
52  				Iterator<Generalization> itGen = generalizations.iterator();
53  				while (itGen.hasNext() && (conform == null)) {
54  					conform = UMLUtil.getStereotypeApplication(itGen.next(), Conform.class);
55  				}
56  			}
57  			// Find ViewPoint
58  			if (conform != null) {
59  				Generalization baseGeneralization = conform.getBase_Generalization();
60  				if (baseGeneralization != null){
61  					EList<Element> targets = baseGeneralization.getTargets();
62  					if (targets != null) {
63  						Iterator<Element> itElt = targets.iterator();
64  						while (itElt.hasNext()) {
65  							viewPoint = UMLUtil.getStereotypeApplication(itElt.next(), Viewpoint.class);
66  						}						
67  					}
68  				}
69  			}
70  		}
71  		return viewPoint;
72  	}
73  
74  	/**
75  	 * 
76  	 * 7.3.2.7 View : /stakeholder: Stakeholder [*] The list of stakeholders is derived from the viewpoint the view conforms to.
77  	 * @see org.eclipse.papyrus.sysml14.modelelements.internal.impl.ViewImpl#getStakeholder()
78  	 *
79  	 * @return
80  	 */
81  	@Override
82  	public EList<Stakeholder> getStakeholder() {
83  		EList<Stakeholder> stakeholderEList = new BasicEList<Stakeholder>();
84  		Viewpoint viewPoint = getViewPoint();
85  		if (viewPoint != null){
86  			stakeholderEList.addAll(viewPoint.getStakeholder());
87  		}
88  		return new UnmodifiableEList<Stakeholder>(this, ModelelementsPackage.eINSTANCE.getViewpoint_Stakeholder(), stakeholderEList.size(), stakeholderEList.toArray());
89  	}
