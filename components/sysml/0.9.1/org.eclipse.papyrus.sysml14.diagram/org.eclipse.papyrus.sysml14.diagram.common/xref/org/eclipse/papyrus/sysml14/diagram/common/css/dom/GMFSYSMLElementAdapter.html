94  	/**
95  	 * {@inheritDoc}
96  	 *
97  	 * Applied Stereotypes are manipulated as DOM Attributes
98  	 */
99  	@Override
100 	protected String doGetAttribute(String attr) {
101 		String parentValue = super.doGetAttribute(attr);
102 		if (parentValue != null) {
103 			return parentValue;
104 		}
105 
106 		// This is present only for performance issues 
107 		// isAbstract isn't a SysML feature but will be called many time for UML
108 		// and is very slow to answer (Check : L172)
109 		if (IS_ABSTRACT.equals(attr)) {
110 			return null;
111 		}			
112 		
113 		// get stereotype Label attribute
114 		if (stereotypeHelper.isStereotypeLabel(semanticElement)) {
115 			String value = getStereotypeLabelAttribute(attr);
116 			if (value != null && !value.isEmpty()) {
117 				return value;
118 			}
119 		}
120 
121 		// get stereotype Compartment attribute
122 		if (stereotypeHelper.isStereotypeCompartment(semanticElement) || stereotypeHelper.isStereotypeBrace(semanticElement)) {
123 			String value = getStereotypeCompartmentAttribute(attr);
124 			if (value != null && !value.isEmpty()) {
125 				return value;
126 			}
127 		}
128 
129 		// get stereotype Property attribute
130 		if (stereotypeHelper.isStereotypeProperty(semanticElement) || stereotypeHelper.isStereotypeBraceProperty(semanticElement)) {
131 			String value = getStereotypePropertyAttribute(attr);
132 			if (value != null && !value.isEmpty()) {
133 				return value;
134 			}
135 		}
136 
137 		if (semanticElement instanceof Element) {
138 			// Applied stereotypes
139 			Element currentElement = (Element) semanticElement;
140 			
141 			// IMPORTANT set the faster test first
142 			if(DEPRECATED.equals(attr)){
143 				return String.valueOf(DeprecatedElementUtil.isDeprecatedElement(semanticElement));
144 			}
145 			boolean isProperty =  semanticElement instanceof Property;
146 			// manage of isConstraint=true attribute for Property
147 			if (IS_CONSTRAINT.equals(attr) && isProperty) {
148 				IElementMatcher constraint = ((ISpecializationType) ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_CONSTRAINTPROPERTY)).getMatcher();
149 				return String.valueOf(constraint.matches((Property) semanticElement));
150 			}
151 			// manage of isPart=true attribute for Property
152 			if (IS_PART.equals(attr) && isProperty) {
153 				IElementMatcher part = ((ISpecializationType) ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_PART)).getMatcher();
154 				return String.valueOf(part.matches((Property) semanticElement));
155 			}
156 			// manage of isReference=true attribute for Property
157 			if (IS_REFERENCE.equals(attr) && isProperty) {
158 				IElementMatcher reference = ((ISpecializationType) ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_REFERENCE)).getMatcher();
159 				return String.valueOf(reference.matches((Property) semanticElement));
160 			}
161 			
162 
163 			// Get applied STereotypes Attributes
164 			if (APPLIED_STEREOTYPES_PROPERTY.equals(attr)) {
165 				List<String> appliedStereotypes = new LinkedList<String>();
166 				for (Stereotype stereotype : currentElement.getAppliedStereotypes()) {
167 					appliedStereotypes.add(stereotype.getName());
168 					appliedStereotypes.add(stereotype.getQualifiedName());
169 				}
170 				if (!appliedStereotypes.isEmpty()) {
171 					return ListHelper.deepToString(appliedStereotypes, CSS_VALUES_SEPARATOR);
172 				}
173 				return null;
174 			}
175 			for (EObject stereotypeApplication : currentElement.getStereotypeApplications()) {
176 				EStructuralFeature feature = stereotypeApplication.eClass().getEStructuralFeature(attr);
177 				if (feature != null) {
178 					if (feature.isMany()) {
179 						List<?> values = (List<?>) stereotypeApplication.eGet(feature);
180 						List<String> cssValues = new LinkedList<String>();
181 						for (Object value : values) {
182 							cssValues.add(getCSSValue(feature, value));
183 						}
184 						return ListHelper.deepToString(cssValues, CSS_VALUES_SEPARATOR);
185 					} else {
186 						Object value = stereotypeApplication.eGet(feature);
187 						return getCSSValue(feature, value);
188 					}
189 				}
190 				return null;
191 			}
192 
193 			if (attr.contains(QUALIFIER_SEPARATOR)) {
194 				List<String> qualifiers = ListHelper.asList(attr.split(QUALIFIER_SEPARATOR)); // Writable list
195 				String propertyName = qualifiers.remove(qualifiers.size() - 1); // Last element is the property name
196 				// Remaining strings can be used to build the Stereotype's qualified name
197 				String stereotypeName = ListHelper.deepToString(qualifiers, "::"); //$NON-NLS-1$
198 				Stereotype appliedStereotype = currentElement.getAppliedStereotype(stereotypeName);
199 				if (appliedStereotype != null) {
200 					EObject stereotypeApplication = currentElement.getStereotypeApplication(appliedStereotype);
201 					EStructuralFeature feature = stereotypeApplication.eClass().getEStructuralFeature(propertyName);
202 					if (feature != null) {
203 						Object value = stereotypeApplication.eGet(feature);
204 						return getCSSValue(feature, value);
205 					}
206 				}
207 			}
208 			
209 		}
210 		return null;
211 	}

213 	/**
214 	 * Retrieve the Matching String Value for the StereotypeCompartment Element
215 	 *
216 	 * @param attr
217 	 *            Attribute of the String to match with
218 	 * @return The matching value for this Attribute
219 	 */
220 	protected String getStereotypeCompartmentAttribute(String attr) {
221 		if (StereotypeDisplayConstant.STEREOTYPE_COMPARTMENT_NAME.equals(attr)) {
222 			BasicCompartment propertyCompartment = (BasicCompartment) semanticElement;
223 			return stereotypeHelper.getName(propertyCompartment);
224 		}
225 		return "";//$NON-NLS-1$
226 	}
227 
228 	/**
229 	 * Retrieve the Matching String Value for the StereotypeCompartment Element
230 	 *
231 	 * @param attr
232 	 *            Attribute of the String to match with
233 	 * @return The matching value for this Attribute
234 	 */
235 	protected String getStereotypePropertyAttribute(String attr) {
236 		// CSS can match property level
237 		if (StereotypeDisplayConstant.STEREOTYPE_PROPERTY_NAME.equals(attr)) {
238 			DecorationNode propertyLabel = (DecorationNode) semanticElement;
239 			if (propertyLabel.getElement() instanceof Property) {
240 				Property prop = (Property) propertyLabel.getElement();
241 				return prop.getName();
242 			}
243 			// CSS can match Container Name
244 		} else if (StereotypeDisplayConstant.STEREOTYPE_COMPARTMENT_NAME.equals(attr)) {
245 			EObject propertyCompartment = ((DecorationNode) semanticElement).eContainer();
246 			if (stereotypeHelper.isStereotypeCompartment(propertyCompartment)) {
247 
248 				return stereotypeHelper.getName((DecorationNode) propertyCompartment);
249 			}
250 
251 		}
252 		return "";//$NON-NLS-1$
253 	}
254 
255 	/**
256 	 * Get the matching Value of the Attribute
257 	 *
258 	 * @param attr
259 	 *            Attribute of the String to match with
260 	 * @return The matching value for this Attribute
261 	 */
262 	protected String getStereotypeLabelAttribute(String attr) {
263 		if (StereotypeDisplayConstant.STEREOTYPE_LABEL_NAME.equals(attr)) {
264 			DecorationNode label = (DecorationNode) semanticElement;
265 			return stereotypeHelper.getName(label);
266 		}
267 
268 		if (KIND.equals(attr) && stereotypeHelper.isStereotypeLabel(semanticElement)) {
269 			return StereotypeDisplayConstant.STEREOTYPE_LABEL_TYPE;
270 		}
271 		return null;
272 	}
273 
274 	@Override
275 	protected String getCSSValue(EStructuralFeature feature, Object value) {
276 		if (feature instanceof EReference && value instanceof NamedElement) {
277 			String name = ((NamedElement) value).getName();
278 			return name == null || name.isEmpty() ? EMPTY_VALUE : name; // Bug 467716: Never return null or empty string if the value is not null
279 		}
280 		return super.getCSSValue(feature, value);
281 	}

283 	/**
284 	 * @see org.eclipse.papyrus.infra.gmfdiag.css.dom.GMFElementAdapter#getLocalName()
285 	 *
286 	 * @return The Local name for the CSS matching.
287 	 */
288 	@Override
289 	public String getLocalName() {
290 		// In case of StereotypeComment type, the selector should match on the Stereotype Comment.
291 		if (localName == null && stereotypeHelper.isStereotypeComment(getNotationElement())) {
292 			return STEREOTYPE_COMMENT;
293 		}
294 		return super.getLocalName();
295 	}
