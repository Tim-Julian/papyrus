83  	@Before
84  	public void setUp() {
85  		// Prepare test
86  		Model model = SysMLResource.createSysMLModel(new ResourceSetImpl());
87  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());
88  		StereotypeApplicationHelper stereotypeApplicationHelper = StereotypeApplicationHelper.getInstance(null);
89  		classStereotypeViewpoint = model.createOwnedClass("classStereotypeViewpoint", false);
90  		stereotypeApplicationHelper.applyStereotype(classStereotypeViewpoint, ModelelementsPackage.eINSTANCE.getViewpoint());
91  		classStereotypeView = model.createOwnedClass("classStereotypeView", false);
92  		stereotypeApplicationHelper.applyStereotype(classStereotypeView, ModelelementsPackage.eINSTANCE.getView());
93  
94  		// class not sterotyped
95  		notStereotypeClass = model.createOwnedClass("notStereotypeClass", false);
96  
97  		// creation of the conform link
98  		conformLink = UMLFactory.eINSTANCE.createAbstraction();
99  		model.getPackagedElements().add(conformLink);
100 		conformLink.getClients().add(classStereotypeView);
101 		conformLink.getSuppliers().add(classStereotypeViewpoint);
102 		stereotypeApplicationHelper.applyStereotype(conformLink, ModelelementsPackage.eINSTANCE.getConform());
103 
104 
105 		classStereotypeView2 = model.createOwnedClass("classStereotypeView2", false);
106 		stereotypeApplicationHelper.applyStereotype(classStereotypeView2, ModelelementsPackage.eINSTANCE.getView());
107 
108 
109 		classStereotypeViewpoint2 = model.createOwnedClass("classStereotypeViewpoint2", false);
110 		stereotypeApplicationHelper.applyStereotype(classStereotypeViewpoint2, ModelelementsPackage.eINSTANCE.getViewpoint());
111 	}

114 	/**
115 	 * Test if the creation of conform following SysML requirements
116 	 */
117 	@Test
118 	public void testApproveRequest() {
119 		ConformReorientEditHelperAdvice conformReorientEditHelperAdvice = new ConformReorientEditHelperAdvice();
120 
121 		IEditCommandRequest invalidRequest = new ReorientRelationshipRequest(conformLink, notStereotypeClass, classStereotypeViewpoint, ReorientRequest.REORIENT_TARGET);
122 		Assert.assertFalse("Target reorient should forbiden for a conform not linked to any elements", conformReorientEditHelperAdvice.approveRequest(invalidRequest));
123 
124 		IEditCommandRequest validRequest = new ReorientRelationshipRequest(conformLink, classStereotypeViewpoint2, classStereotypeViewpoint, ReorientRequest.REORIENT_TARGET);
125 		Assert.assertTrue("Target reorient should allow on view point", conformReorientEditHelperAdvice.approveRequest(validRequest));
126 
127 		IEditCommandRequest invalidRequest2 = new ReorientRelationshipRequest(conformLink, notStereotypeClass, classStereotypeView, ReorientRequest.REORIENT_SOURCE);
128 		Assert.assertFalse("Source reotient should forbiden for a conform not linked to any elements", conformReorientEditHelperAdvice.approveRequest(invalidRequest2));
129 
130 		IEditCommandRequest validRequest2 = new ReorientRelationshipRequest(conformLink, classStereotypeView2, classStereotypeView, ReorientRequest.REORIENT_SOURCE);
131 		Assert.assertTrue("Source reorient should should allow on view ", conformReorientEditHelperAdvice.approveRequest(validRequest2));
132 
133 
134 	}
