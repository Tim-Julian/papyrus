14  package org.eclipse.papyrus.sysml14.nattable.allocation.tests.tester;
15  
16  import org.eclipse.core.runtime.IStatus;
17  import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
18  import org.eclipse.papyrus.sysml14.nattable.allocation.tester.TableAllocationCreationTester;
19  import org.eclipse.papyrus.sysml14.util.SysMLResource;
20  import org.eclipse.uml2.uml.Model;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Test for TableAllocationCreationTester
26   */
27  @SuppressWarnings("nls")
28  public class TableAllocationCreationTesterTest {
29  
30  	/**
31  	 * The creation isn't allowed on null elements
32  	 */
33  	@Test
34  	public void emptyContextTest() {
35  		TableAllocationCreationTester tableAllocationCreationTester = new TableAllocationCreationTester();
36  		IStatus allowed = tableAllocationCreationTester.isAllowed(null);
37  		Assert.assertEquals("The tester should not work on a null entry", IStatus.ERROR, allowed.getSeverity());
38  	}
39  	
40  	/**
41  	 * The creation of a Table Allocation is allowed on the root of a SyML 1.4 model
42  	 */
43  	@Test
44  	public void simpleModelTest() {
45  		TableAllocationCreationTester tableAllocationCreationTester = new TableAllocationCreationTester();
46  		Model model = SysMLResource.createSysMLModel(new ResourceSetImpl());
47  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());		
48  		IStatus allowed = tableAllocationCreationTester.isAllowed(model);
49  		Assert.assertEquals("The tester work on a model with SyML 1.4 profil entry", IStatus.OK, allowed.getSeverity());
50  	}
