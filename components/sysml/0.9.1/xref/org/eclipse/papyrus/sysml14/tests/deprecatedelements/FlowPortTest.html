36  /**
37   * A test case for the model object '
38   * <em><b>Flow Port</b></em>'.
39   * <p>
40   * The following features are tested:
41   * <ul>
42   * <li>{@link org.eclipse.papyrus.sysml14.deprecatedelements.FlowPort#isAtomic()
43   * <em>Is Atomic</em>}</li>
44   * </ul>
45   * </p>
46   * <p>
47   * The following operations are tested:
48   * <ul>
49   * <li>{@link org.eclipse.papyrus.sysml14.deprecatedelements.FlowPort#getIcon()
50   * <em>Get Icon</em>}</li>
51   * </ul>
52   * </p>
53   *
54   */
55  @SuppressWarnings("nls")
56  public class FlowPortTest {
57  
58  	private Port p0 = null;
59  
60  	private FlowPort fpDefault = null;
61  
62  	private FlowPort fpOut = null;
63  
64  	private FlowPort fpIn = null;
65  
66  	/**
67  	 * @see junit.framework.TestCase#setUp()
68  	 */
69  	@Before
70  	public void setUp() {
71  		// Prepare test
72  		Model model = SysMLResource.createSysMLModel(new ResourceSetImpl());		
73  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());
74  
75  		// Create classes & interface
76  		Class c0 = model.createOwnedClass("c0", false);
77  		Interface i2 = model.createOwnedInterface("i2");
78  
79  		// Create type
80  		PrimitiveType ptype = model.createOwnedPrimitiveType("type");
81  
82  		// Create ports
83  		p0 = c0.createOwnedPort("p0", null);
84  		Port p1 = c0.createOwnedPort("p1", ptype);
85  		Port p2 = c0.createOwnedPort("p2", i2); //$NON-NLS-1$
86  
87  		// Add FlowPort stereotypes
88  		StereotypeApplicationHelper stereotypeApplicationHelper = StereotypeApplicationHelper.getInstance(null);
89  		fpDefault = (FlowPort) stereotypeApplicationHelper.applyStereotype(p0,
90  				DeprecatedelementsPackage.eINSTANCE.getFlowPort());
91  
92  		fpOut = (FlowPort) stereotypeApplicationHelper.applyStereotype(p1,
93  				DeprecatedelementsPackage.eINSTANCE.getFlowPort());
94  		fpOut.setDirection(FlowDirection.OUT);
95  		fpIn = (FlowPort) stereotypeApplicationHelper.applyStereotype(p2,
96  				DeprecatedelementsPackage.eINSTANCE.getFlowPort());
97  		fpIn.setDirection(FlowDirection.IN);
98  
99  		p2.setIsConjugated(true);
100 
101 		// Add FlowSpecification
102 		stereotypeApplicationHelper.applyStereotype(i2, DeprecatedelementsPackage.eINSTANCE.getFlowSpecification());
103 	}
104 
105 	/**
106 	 * Tests the '
107 	 * {@link org.eclipse.papyrus.sysml14.deprecatedelements.FlowPort#isAtomic()
108 	 * <em>Is Atomic</em>}' feature getter.
109 	 *
110 	 * @see org.eclipse.papyrus.sysml14.deprecatedelements.FlowPort#isAtomic()
111 	 */
112 	@Test
113 	public void testIsAtomic() {		
114 		Assert.assertTrue("FlowPort without type is Atomic", fpDefault.isAtomic());
115 		Assert.assertTrue("FlowPort type is not a FlowSpecification",fpOut.isAtomic());
116 		Assert.assertFalse("FlowPort type is a FlowSpecification", fpIn.isAtomic());
117 	}
118 
119 	/**
120 	 * Tests the '
121 	 * {@link org.eclipse.papyrus.sysml14.deprecatedelements.FlowPort#getIcon() <em>Get Icon</em>}' operation. 
122 	 *
123 	 * @see org.eclipse.papyrus.sysml14.deprecatedelements.FlowPort#getIcon()
124 	 * 
125 	 */
126 	@Test
127 	public void testGetIcon() {
128 		Stereotype s = UMLUtil.getStereotype(fpDefault); // Retrieve Stereotype
129 		Image imageIn = s.getIcons().get(FlowDirection.IN.getValue()); // IN
130 		Image imageOut = s.getIcons().get(FlowDirection.OUT.getValue()); // OUT
131 		Image imageInOut = s.getIcons().get(FlowDirection.INOUT.getValue()); // INOUT
132 				
133 		Assert.assertEquals("A Flow property with an INOUT direction must have the INOUT image",imageInOut,fpDefault.getIcon());
134 		Assert.assertEquals("A Flow property with an OUT direction must have the OUT image",imageOut,fpOut.getIcon());
135 		Assert.assertEquals("A Flow property with an IN direction must have the IN image",imageIn,fpIn.getIcon());
136 
137 	}
138 	
139 	/**
140 	 * Tests the '
141 	 * {@link org.eclipse.papyrus.sysml14.util.DeprecatedElementUtil#isDeprecatedElement()
142 	 * <em>Is Deprecated</em>}' feature getter.
143 	 *
144 	 * @see org.eclipse.papyrus.sysml14.util.DeprecatedElementUtil#isDeprecatedElement()
145 	 */
146 	@Test
147 	public void testIsDeprecated() {
148 		Assert.assertTrue("FlowPort is deprecated", DeprecatedElementUtil.isDeprecatedElement(p0));
149 		Assert.assertTrue("FlowPort is deprecated", DeprecatedElementUtil.isDeprecatedElement(fpOut));
150 	}
151 }
