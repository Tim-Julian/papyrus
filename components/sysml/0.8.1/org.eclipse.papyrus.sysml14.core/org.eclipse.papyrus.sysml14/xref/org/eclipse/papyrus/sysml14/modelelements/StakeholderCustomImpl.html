17  import org.eclipse.emf.common.util.EList;

25  public class StakeholderCustomImpl extends StakeholderImpl {
26  	
27  	
28  	/**
29  	 * 7.3.2.6 Stakeholder : The interests of this stakeholder displayed as the body of the comments from concernList.
30  	 * @see org.eclipse.papyrus.sysml14.modelelements.internal.impl.StakeholderImpl#getConcernList()
31  	 *<p> See the requirement <b>{@papyrus.req org.eclipse.papyrus.sysml14#Req012}</b>. 
32  	 * @return
33  	 */
34  	@Override
35  	public EList<String> getConcern() {
36  		EList<String> concern = new BasicEList<String>();
37  		 EList<Comment> commentEList = getConcernList();
38  		 if (commentEList!=null && commentEList.size() > 0){ 
39  			 for (Comment comment : commentEList) {
40  				 String body = comment.getBody();
41  				 if (body != null){
42  					 concern.add(body); 
43  				 }		
44  			}
45  		 }
46  		return concern; 
47  	}
48  	
49  }
