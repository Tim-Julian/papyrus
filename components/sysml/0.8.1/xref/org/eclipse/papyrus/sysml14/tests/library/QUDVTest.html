27   *  - validate the model
28   * 
29   * @author Benoit Maggi

31  public class QUDVTest {
32  	
33  	public static final String QUDV_PATH = Activator.PLUGIN_ID +"/resources/library/QUDV.uml"; //$NON-NLS-1$
34  	
35  	/**
36  	 * Validate the model with the rules defined in the meta-model tooling
37  	 */
38  	@Test
39  	public void validatePropertiesModel() {
40  		URI createPlatformPluginURI = URI.createPlatformPluginURI(QUDV_PATH, true);
41  		Resource resource = new ResourceSetImpl().getResource(createPlatformPluginURI, true);
42  		Diagnostic diagnostic = Diagnostician.INSTANCE.validate(resource.getContents().get(0));
43  		// FIXME the model provided by OMG is raising warnings
44  		Assert.assertEquals("The QUDV.uml model is not valid ", Diagnostic.WARNING, diagnostic.getSeverity()); //$NON-NLS-1$
45  	}
46  }
