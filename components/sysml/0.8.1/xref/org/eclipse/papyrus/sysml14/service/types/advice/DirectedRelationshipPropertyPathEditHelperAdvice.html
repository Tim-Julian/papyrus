34  /**
35   * 8.3.2.7 DirectedRelationshipPropertyPath
36   * sourcePropertyPath: Property [0..*] {ordered, nonunique}
37   * A series of properties that identifies the source of the directed relationship in the context of the block specified by the
38   * sourceContext property. The ordering of properties is from a property of the sourceContext block, through a property
39   * of each intermediate block that types the preceding property, ending in a property with a type that owns or inherits the
40   * source of the directed relationship. The source is not included in the propertyPath list. The same property might
41   * appear more than once because a block can own a property with the same or specialized block as a type.
42   * • targetPropertyPath: Property [0..*] {ordered, nonunique}
43   * A series of properties that identifies the target of the directed relationship in the context of the block specified by the
44   * targetContext property. The ordering of properties is from a property of the targetContext block, through a property of
45   * each intermediate block that types the preceding property, ending in a property with a type that owns or inherits the
46   * target of the directed relationship. The target is not included in the propertyPath list. The same property might appear
47   * more than once because a block can own a property with the same or specialized block as a type.
48   * • sourceContext: Classifier [0.1]
49   * Gives the context for sourcePropertyPath to begin from.
50   * • targetContext: Classifier [0.1]
51   * Gives the context for targetPropertyPath to begin from.
52   *
53   */
54  public class DirectedRelationshipPropertyPathEditHelperAdvice extends AbstractEditHelperAdvice {
55  
56  
57  
58  
59  	@Override
60  	protected ICommand getAfterCreateRelationshipCommand(CreateRelationshipRequest request) {
61  
62  		ICommand defaultCommand = super.getAfterCreateRelationshipCommand(request);
63  
64  		View sourceView = RequestParameterUtils.getSourceView(request);
65  		View targetView = RequestParameterUtils.getTargetView(request);
66  
67  
68  		if (request.getElementType() instanceof ISpecializationType) {
69  			// get all specializations from elementTypes
70  			ArrayList<IElementType> elementTypesHierarchy = new ArrayList<IElementType>();
71  			elementTypesHierarchy.add(request.getElementType());
72  			ISpecializationType iSpecializationType = (ISpecializationType) request.getElementType();
73  			elementTypesHierarchy.addAll(Arrays.asList(iSpecializationType.getAllSuperTypes()));
74  
75  
76  			if ((elementTypesHierarchy.contains(ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_ALLOCATE_ABSTRACTION))) ||
77  					(elementTypesHierarchy.contains(ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_REFINE_ABSTRACTION))) ||
78  					(elementTypesHierarchy.contains(ElementTypeRegistry.getInstance().getType(SysMLServiceTypeUtil.ORG_ECLIPSE_PAPYRUS_SYSML14_TRACE_ABSTRACTION)))) {
79  
80  				if (sourceView.getElement() instanceof Property) {
81  					SetDirectedRelationshipPropertyPathCommand setPropertyPathCommand = new SetDirectedRelationshipPropertyPathCommand(request.getEditingDomain(), sourceView, true);
82  					EMFToIEditCommandWrapper editCommandWrapper = new EMFToCreateElementEditCommandWrapper(request.getContainer(), request, setPropertyPathCommand);
83  					defaultCommand = CompositeCommand.compose(defaultCommand, editCommandWrapper);
84  				}
85  
86  				if (targetView.getElement() instanceof Property) {
87  					SetDirectedRelationshipPropertyPathCommand setPropertyPathCommand = new SetDirectedRelationshipPropertyPathCommand(request.getEditingDomain(), targetView, false);
88  					EMFToIEditCommandWrapper editCommandWrapper = new EMFToCreateElementEditCommandWrapper(request.getContainer(), request, setPropertyPathCommand);
89  					defaultCommand = CompositeCommand.compose(defaultCommand, editCommandWrapper);
90  				}
91  			}
92  
93  		}
94  
95  		return defaultCommand;
96  	}
