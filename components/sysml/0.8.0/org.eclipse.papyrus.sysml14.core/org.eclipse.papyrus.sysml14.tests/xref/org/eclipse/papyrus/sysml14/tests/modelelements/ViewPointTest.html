33  /**
34   * A test case for the model object '<em><b>ViewPoint</b></em>'.
35   *  See the requirement <b>{@papyrus.req org.eclipse.papyrus.sysml14#Req013}</b>. 
36   * <p>
37   * The following features are tested:
38   * <ul>
39   * <li>{@link org.eclipse.papyrus.sysml14.modelelements.Viewpoint#getConcern <em>Concern</em>}</li>
40   * <li>{@link org.eclipse.papyrus.sysml14.modelelements.Viewpoint#getMethod <em>Method</em>}</li>
41   * </ul>
42   * </p>
43   */
44  public class ViewPointTest {
45  
46  	private static final String THIS_IS_THE_BODY_OF_COMMENT2 = "this is the body of comment2";
47  
48  	private static final String THIS_IS_THE_BODY_OF_COMMENT1 = "this is the body of comment1";
49  
50  	private Viewpoint defaultViewpoint = null;
51  
52  	private Behavior ownedBehavior = null;
53  
54  	/**
55  	 * @see junit.framework.TestCase#setUp()
56  	 */
57  	@Before
58  	public void setUp()  {
59  
60  		// Prepare test
61  		Model model = SysMLResource.createSysMLModel(new ResourceSetImpl());
62  		
63  		Assert.assertFalse("the SysML profil must be applied.", model.getAppliedProfiles().isEmpty());
64  		
65  		// Create classes & interface
66  		Class defaultClass = model.createOwnedClass("defaultClass", false);
67  
68  		// Add viewPoint stereotypes
69  		StereotypeApplicationHelper stereotypeApplicationHelper = StereotypeApplicationHelper.getInstance(null);
70  		defaultViewpoint = (Viewpoint)stereotypeApplicationHelper.applyStereotype(defaultClass, ModelelementsPackage.eINSTANCE.getViewpoint());
71  		Comment comment1=UMLFactory.eINSTANCE.createComment();
72  		comment1.setBody(THIS_IS_THE_BODY_OF_COMMENT1);
73  		model.getOwnedComments().add(comment1);
74  		Comment comment2=UMLFactory.eINSTANCE.createComment();
75  		comment2.setBody(THIS_IS_THE_BODY_OF_COMMENT2);
76  		model.getOwnedComments().add(comment2);
77  		defaultViewpoint.getConcernList().add(comment1);
78  		defaultViewpoint.getConcernList().add(comment2);
79  		Operation createOperation= UMLFactory.eINSTANCE.createOperation();
80  		createOperation.setName("View");
81  		defaultClass.getOwnedOperations().add(createOperation);
82  		ownedBehavior = defaultClass.createOwnedBehavior("aBehavior", UMLPackage.eINSTANCE.getActivity());
83  		createOperation.getMethods().add(ownedBehavior);
84  		stereotypeApplicationHelper.applyStereotype(createOperation, StandardPackage.eINSTANCE.getCreate());		
85  	}
86  
87  	/**
88  	 * Tests the '{@link org.eclipse.papyrus.sysml14.modelelements.viewpoint#getConcern()
89  	 * <em>concern</em>}' feature getter. 
90  	 * @see org.eclipse.papyrus.sysml14.modelelements.viewpoint#getConcern()
91  	 */
92  	@Test
93  	public void testGetConcern() {
94  		Assert.assertEquals("The derived property concern is badly calculated.The size of the list must 2.", 2, defaultViewpoint.getConcern().size());
95  		Assert.assertEquals("The derived property concern is badly calculated. Concern must contain at the first place the body of the first comment",THIS_IS_THE_BODY_OF_COMMENT1,defaultViewpoint.getConcern().get(0));
96  		Assert.assertEquals("The derived property concern is badly calculated. Concern must contain at the second place the body of the second comment",THIS_IS_THE_BODY_OF_COMMENT2,defaultViewpoint.getConcern().get(1));
97  	}
98  	/**
99  	 * Tests the '{@link org.eclipse.papyrus.sysml14.modelelements.viewpoint#getBehavior()
100 	 * <em>behavior</em>}' feature getter. 
101 	 * @see org.eclipse.papyrus.sysml14.modelelements.viewpoint#getBehavior()
102 	 */
103 	@Test
104 	public void testGetMethod() {
105 		Assert.assertEquals("The derived property method is badly calculated. The size of the list must 1",1,defaultViewpoint.getMethod().size());
106 		Assert.assertEquals("The derived property method is badly calculated. The view point must contained in the list",ownedBehavior,defaultViewpoint.getMethod().get(0));
107 	}
108 }
