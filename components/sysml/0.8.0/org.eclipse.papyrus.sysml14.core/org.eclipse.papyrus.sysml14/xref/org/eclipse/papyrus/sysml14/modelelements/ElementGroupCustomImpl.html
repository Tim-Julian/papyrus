14  package org.eclipse.papyrus.sysml14.modelelements;

16  import org.eclipse.emf.common.util.BasicEList;
17  import org.eclipse.emf.common.util.EList;
18  import org.eclipse.emf.ecore.util.EcoreEList.UnmodifiableEList;
19  import org.eclipse.papyrus.sysml14.modelelements.internal.impl.ElementGroupImpl;
20  import org.eclipse.uml2.uml.Comment;
21  import org.eclipse.uml2.uml.Element;
22  
23  /**
24   * 
25   *
26   */
27  public class ElementGroupCustomImpl extends ElementGroupImpl {
28  
29  	
30  
31  	/**
32  	 * 7.3.2.2 ElementGroup : /criterion[0..1]: String Specifies the rationale for being member of the group. 
33  	 * Adding an element to the group asserts that the criterion applies to this element.
34  	 * Derived from Comment::body.
35  	 * 
36  	 * @see org.eclipse.papyrus.sysml14.modelelements.internal.impl.ElementGroupImpl#getCriterion()
37  	 *<p> See the requirement <b>{@papyrus.req org.eclipse.papyrus.sysml14#Req014}</b>. 
38  	 * @return
39  	 */
40  	public String getCriterion() {
41  		String criterion = "";
42  		Comment comment = getBase_Comment();
43  		if (comment != null){
44  			criterion = comment.getBody();
45  		}
46  		return criterion;
47  	}
48  
49  
50  	/**
51  	 * 7.3.2.2 ElementGroup : /criterion[0..1]: String Specifies the rationale for being member of the group. 
52  	 * Adding an element to the group asserts that the criterion applies to this element.
53  	 * Derived from Comment::body.
54  	 * 
55  	 * @see org.eclipse.papyrus.sysml14.modelelements.internal.impl.ElementGroupImpl#setCriterion(java.lang.String)
56  	 *
57  	 * @param newCriterion
58  	 */
59  	public void setCriterion(String newCriterion) {
60  		Comment comment = getBase_Comment();
61  		if (comment != null){
62  			comment.setBody(newCriterion);
63  		}
64  	}
65  
66  	/**
67  	 * 
68  	 * 7.3.2.2 ElementGroup : /member: Element[0..*] Set specifying the members of the group. 
69  	 * Derived from Comment::annotatedElement.
70  	 * 
71  	 * @see org.eclipse.papyrus.sysml14.modelelements.internal.impl.ElementGroupImpl#getMember()
72  	 *
73  	 * @return
74  	 */
75  	public EList<Element> getMember() {
76  		BasicEList<Element> elementEList = new BasicEList<Element>();
77  		Comment comment = getBase_Comment();
78  		if (comment != null){
79  			elementEList.addAll( comment.getAnnotatedElements());
80  		}	
81  		return new UnmodifiableEList<Element>(this, ModelelementsPackage.eINSTANCE.getElementGroup_Member(), elementEList.size(), elementEList.toArray());
82  	}
83  	
84  	/**
85  	 * 7.3.2.2 ElementGroup : /size: Integer Number of members in the group. Derived.
86  	 * @see org.eclipse.papyrus.sysml14.modelelements.internal.impl.ElementGroupImpl#getSize()
87  	 *
88  	 * @return
89  	 */
90  	public int getSize() {
91  		return getMember().size();
92  	}	
93  }
