40  /**
41   * 8.3.2.3 Block
42   * the service edit must implement the following constraints
43   * isEncapsulated: Boolean [0..1]
44   * If true, then the block is treated as a black box; a part typed by this black box can only be connected via its ports or
45   * directly to its outer boundary. If false, or if a value is not present, then connections can be established to elements of
46   * its internal structure via deep-nested connector ends.
47   * this class satisfies the requirement{@papyrus.req org.eclipse.papyrus.sysml14.service.types#Req_007}
48   * 
49   * <pre>
50   * This advice is used to remove inconsistent {@link Connector} when a related {@link Block}
51   * is set as encapsulated ({@link Block#isEncapsulated()} property).
52   * </pre>
53   */
54  public class SetEncapsulatedEditHelperAdvice extends AbstractEditHelperAdvice {
55  
56  	/**
57  	 * <pre>
58  	 * {@inheritDoc}
59  	 * 
60  	 * Returns the command to destroy any {@link Connector} breaking encapsulation rules.
61  	 * </pre>
62  	 *
63  	 * @param request
64  	 *            the request to modify the model
65  	 * @return
66  	 * 		the command to destroy the views of the parts which are not owned by the new type
67  	 *
68  	 */
69  	@Override
70  	protected ICommand getBeforeSetCommand(SetRequest request) {
71  		ICommand setCommand = super.getBeforeSetCommand(request);
72  		Set<Connector> connectorToDelete = new HashSet<Connector>();
73  
74  		EObject elementToEdit = request.getElementToEdit();
75  		if ((elementToEdit instanceof Block) && (request.getFeature() == BlocksPackage.eINSTANCE.getBlock_IsEncapsulated())) {
76  
77  			Block blockApp = (Block) elementToEdit;
78  			Boolean setAsEncapsulated = (request.getValue() instanceof Boolean) ? (Boolean) request.getValue() : false;
79  			if (!blockApp.isEncapsulated() && setAsEncapsulated) {
80  
81  				for (Property property : getPropertyTypedByBlock(blockApp)) {
82  					for (NestedConnectorEnd nestedEnd : getRelatedNestedConnectorEnd(property)) {
83  						ConnectorEnd end = nestedEnd.getBase_ConnectorEnd();
84  						Connector connector = (end != null) ? (Connector) end.eContainer() : null;
85  						if (connector != null) {
86  							connectorToDelete.add(connector);
87  						}
88  					}
89  				}
90  			}
91  		}
92  
93  		// Add connector destroy command if needed
94  		if (!(connectorToDelete.isEmpty())) {
95  			DestroyDependentsRequest req = new DestroyDependentsRequest(request.getEditingDomain(), elementToEdit, false);
96  			req.setClientContext(request.getClientContext());
97  			req.addParameters(request.getParameters());
98  			ICommand connectorDestroyCommand = req.getDestroyDependentsCommand(connectorToDelete);
99  			if (connectorDestroyCommand != null) {
100 				setCommand = CompositeCommand.compose(connectorDestroyCommand, setCommand);
101 			}
102 		}
103 
104 		return setCommand;
105 	}
106 
107 	/**
108 	 * Get any {@link Property} typed by the given {@link Block} (its base class).
109 	 *
110 	 * @param block
111 	 *            the block.
112 	 * @return a collection of {@link Property}
113 	 */
114 	private Collection<Property> getPropertyTypedByBlock(Block block) {
115 		EReference[] eRefs = new EReference[] { UMLPackage.eINSTANCE.getTypedElement_Type() };
116 
117 		Collection<?> references = EMFCoreUtil.getReferencers(block.getBase_Class(), eRefs);
118 		List<Property> propertyReferences = new LinkedList<Property>();
119 		for (Object reference : references) {
120 			if (reference instanceof Property) {
121 				propertyReferences.add((Property) reference);
122 			}
123 		}
124 
125 		return propertyReferences;
126 	}
127 
128 	/**
129 	 * Get any {@link NestedConnectorEnd} stereotype application that holds the property in its path.
130 	 *
131 	 * @param property
132 	 *            the property that should be in the path.
133 	 * @return a collection of {@link NestedConnectorEnd}
134 	 */
135 	@SuppressWarnings("unchecked")
136 	private Collection<NestedConnectorEnd> getRelatedNestedConnectorEnd(Property property) {
137 		EReference[] eRefs = new EReference[] { BlocksPackage.eINSTANCE.getElementPropertyPath_PropertyPath() };
138 
139 		Collection<NestedConnectorEnd> references = EMFCoreUtil.getReferencers(property, eRefs);
140 		return (references != null) ? references : Collections.unmodifiableList(new ArrayList<NestedConnectorEnd>());
141 	}
