package org.eclipse.papyrus.sysml14.diagram.blockdefinition.tests.resources;

import java.util.List;

import org.eclipse.emf.common.util.Diagnostic;
import org.eclipse.emf.common.util.URI;
import org.eclipse.emf.ecore.resource.Resource;
import org.eclipse.emf.ecore.resource.impl.ResourceSetImpl;
import org.eclipse.emf.ecore.util.Diagnostician;
import org.eclipse.papyrus.sysml14.diagram.blockdefinition.Activator;
import org.junit.Assert;
import org.junit.Ignore;
import org.junit.Test;

/**
 * Test the doc model :
 *  - validate the model
 * 
 * @author Benoit Maggi
 */
public class DocModelTest {
	
	public static final String UML_MODEL_PATH = Activator.PLUGIN_ID+"/resources/doc/org.eclipse.papyrus.sysml.14.diagram.blockdefinition.uml"; //$NON-NLS-1$
	
	public static final String NOTATION_MODEL_PATH = Activator.PLUGIN_ID+"/resources/doc/org.eclipse.papyrus.sysml.14.diagram.blockdefinition.notation"; //$NON-NLS-1$
	
	public static final String DI_MODEL_PATH = Activator.PLUGIN_ID+"/resources/doc/org.eclipse.papyrus.sysml.14.diagram.blockdefinition.di"; //$NON-NLS-1$
	
	/**
	 * Validate the model with the rules defined in the meta-model tooling
	 */
	@Ignore //FIXME : Invalid model, problem in the import of the omg model
	@Test
	public void validateUmlModel() {
		URI createPlatformPluginURI = URI.createPlatformPluginURI(UML_MODEL_PATH, true);
		Resource resource = new ResourceSetImpl().getResource(createPlatformPluginURI, true);
		Diagnostic diagnostic = Diagnostician.INSTANCE.validate(resource.getContents().get(0));
		Assert.assertEquals("The doc org.eclipse.papyrus.sysml.14.diagram.blockdefinition.uml model is not valid "+print(diagnostic), Diagnostic.OK, diagnostic.getSeverity()); //$NON-NLS-1$
	}

	/**
	 * Validate the model with the rules defined in the meta-model tooling
	 */
	@Test
	public void validateNotationModel() {
		URI createPlatformPluginURI = URI.createPlatformPluginURI(NOTATION_MODEL_PATH, true);
		Resource resource = new ResourceSetImpl().getResource(createPlatformPluginURI, true);
		Diagnostic diagnostic = Diagnostician.INSTANCE.validate(resource.getContents().get(0));
		Assert.assertEquals("The doc org.eclipse.papyrus.sysml.14.diagram.blockdefinition.notation model is not valid ", Diagnostic.OK, diagnostic.getSeverity());//$NON-NLS-1$
	}	

	/**
	 * Validate the model with the rules defined in the meta-model tooling
	 */
	@Ignore //FIXME : org.eclipse.emf.common.util.BasicEList$BasicIndexOutOfBoundsException: index=0, size=0 (Probably because the model is empty)
	@Test
	public void validateDiModel() {
		URI createPlatformPluginURI = URI.createPlatformPluginURI(DI_MODEL_PATH, true);
		Resource resource = new ResourceSetImpl().getResource(createPlatformPluginURI, true);
		Diagnostic diagnostic = Diagnostician.INSTANCE.validate(resource.getContents().get(0));
		Assert.assertEquals("The doc org.eclipse.papyrus.sysml.14.diagram.blockdefinition.di model is not valid ", Diagnostic.OK, diagnostic.getSeverity()); //$NON-NLS-1$
	}	
	
	// FIXME : Something should exist in API to do that
	private String print(Diagnostic diagnostic) {
		List<Diagnostic> children = diagnostic.getChildren();
		StringBuilder stringBuilder = new StringBuilder(diagnostic.getMessage());
		for (Diagnostic diagnosticChildren : children) {
			stringBuilder.append("\n"); //$NON-NLS-1$
			stringBuilder.append(diagnosticChildren.getMessage());
		}
		return stringBuilder.toString();
	}
	
}
