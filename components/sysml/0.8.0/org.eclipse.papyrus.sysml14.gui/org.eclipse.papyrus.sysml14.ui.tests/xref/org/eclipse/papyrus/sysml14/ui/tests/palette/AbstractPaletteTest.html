41  /**
42   * Abstract class used to launch tests on a palette model :
43   *  - check quality id (see developer documentation)
44   *  - chack that the model is valid
45   *  
46   * @author Benoit Maggi
47   */
48  public abstract class AbstractPaletteTest {
49  
50  	public static final String NODE = "node"; //$NON-NLS-1$
51  
52  	public static final String EDGE = "edge"; //$NON-NLS-1$
53  
54  	private String rootPath = ""; //$NON-NLS-1$
55  
56  	private String palettePath = ""; //$NON-NLS-1$
57  
58  	public ElementTypeSetConfigurationRegistry instance = ElementTypeSetConfigurationRegistry.getInstance();
59  
60  	public boolean assertPaletteId(String id) {
61  		return id != null && (id.startsWith(rootPath + "." + NODE) || id.startsWith(rootPath + "." + EDGE));
62  	}
63  	
64  	/**
65  	 * @return root path of expected element ids
66  	 */
67  	protected abstract String getRootPath();
68  	
69  	/**
70  	 * @return path to the palette
71  	 */
72  	protected abstract String getPalettePath();
73  	
74  	/**
75  	 * Get Palette information befor launching a test
76  	 */
77  	@Before
78  	public void loadPalette() {
79  		palettePath = getPalettePath();
80  		rootPath =  getRootPath();
81  	}
82  	
83  	/**
84  	 * Check palette ids and element type reference
85  	 */
86  	@Test
87  	public void checkPaletteIds() {
88  		URI createPlatformPluginURI = URI.createPlatformPluginURI(palettePath, true);
89  		ResourceSetImpl resourceSetImpl = new ResourceSetImpl();
90  		Resource resource = resourceSetImpl.getResource(createPlatformPluginURI, true);
91  
92  
93  		TreeIterator<EObject> allContents = resource.getAllContents();
94  		while (allContents.hasNext()) {
95  			EObject eObject = (EObject) allContents.next();
96  
97  			if (eObject instanceof PaletteConfiguration) {
98  				PaletteConfiguration p = (PaletteConfiguration) eObject;
99  				EList<DrawerConfiguration> drawerConfigurations = p.getDrawerConfigurations();
100 				for (DrawerConfiguration drawerConfiguration : drawerConfigurations) {
101 					EList<ChildConfiguration> ownedConfigurations = drawerConfiguration.getOwnedConfigurations();
102 					for (ChildConfiguration childConfiguration : ownedConfigurations) {
103 
104 						// check id
105 						String id = childConfiguration.getId();
106 						Assert.assertTrue("Malformed palette id : " + id, assertPaletteId(id));
107 
108 						if (childConfiguration instanceof ToolConfiguration) {
109 							ToolConfiguration toolConfiguration = (ToolConfiguration) childConfiguration;
110 							
111 							// check that the referenced elementtype exist
112 							EList<ElementDescriptor> elementDescriptors = toolConfiguration.getElementDescriptors();
113 							for (ElementDescriptor elementDescriptor : elementDescriptors) {
114 								String elementTypeId = elementDescriptor.getElementTypeId();
115 								Assert.assertTrue("Unregistred element id : " + elementTypeId, ElementEditServiceUtils.getEditServiceProvider().isKnownElementType(elementTypeId));
116 							}
117 							
118 							// check icon file
119 							IconDescriptor icon = toolConfiguration.getIcon();
120 							String iconPath = "platform:/plugin/"+ icon.getPluginID()+"/"+  icon.getIconPath();
121 							 if (iconPath != null && !"".equals(iconPath)){
122 									try {
123 										URL url = new URL(iconPath);
124 										Assert.assertNotNull("The icon "+iconPath+"(for : "+toolConfiguration.getId()+") can't be found", FileLocator.find(url));
125 									} catch (MalformedURLException e) {
126 										Assert.fail("The new child menu is refering to a malformed url "+iconPath);
127 									}
128 							 }	
129 							 
130 							 // check that elements in edge drawer are edges (TODO : a better way would be to find a way to identify "edge" element type)
131 							 ToolKind kind = toolConfiguration.getKind();
132 							 if (ToolKind.CONNECTION_TOOL.equals(kind)){
133 								 Assert.assertTrue("An edge element "+toolConfiguration.getId()+" should always be in an edge Drawer "+drawerConfiguration.getId(),drawerConfiguration.getId().contains(EDGE));
134 							 } else { 
135 								 Assert.assertTrue("An node element "+toolConfiguration.getId()+" should always be in an node Drawer "+drawerConfiguration.getId(),drawerConfiguration.getId().contains(NODE));
136 							 }
137 						}
138 					}
139 				}
140 			}
141 		}
142 	}

144 	/**
145 	 * Validate the model with the rules defined in the meta-model tooling
146 	 */
147 	@Test
148 	public void useValidation() {
149 		URI createPlatformPluginURI = URI.createPlatformPluginURI(palettePath, true);
150 		Resource resource = new ResourceSetImpl().getResource(createPlatformPluginURI, true);
151 		Diagnostic diagnostic = Diagnostician.INSTANCE.validate(resource.getContents().get(0));
152 		Assert.assertEquals("The palette is not valid "+palettePath, Diagnostic.OK, diagnostic.getSeverity());
153 	}
