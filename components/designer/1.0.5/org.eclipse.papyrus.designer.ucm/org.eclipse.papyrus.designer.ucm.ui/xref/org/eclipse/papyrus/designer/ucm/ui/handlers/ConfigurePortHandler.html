15  package org.eclipse.papyrus.designer.ucm.ui.handlers;
16  
17  import org.eclipse.core.commands.ExecutionEvent;
18  import org.eclipse.core.commands.ExecutionException;
19  import org.eclipse.gmf.runtime.common.core.command.CommandResult;
20  import org.eclipse.jface.dialogs.IDialogConstants;
21  import org.eclipse.papyrus.designer.ucm.core.utils.CommandSupport;
22  import org.eclipse.papyrus.designer.ucm.core.utils.RunnableWithResult;
23  import org.eclipse.papyrus.designer.ucm.ui.dialogs.PortConfigurationDialog;
24  import org.eclipse.papyrus.uml.diagram.common.handlers.CmdHandler;
25  import org.eclipse.swt.widgets.Display;
26  import org.eclipse.uml2.uml.Port;
27  
28  public class ConfigurePortHandler extends CmdHandler {
29  
30  	// ------------------------------------------------------------------------
31  	// Execution
32  	// ------------------------------------------------------------------------
33  
34  	public static final String CONFIGURE_PORT = "Configure port"; //$NON-NLS-1$
35  
36  	@Override
37  	public boolean isEnabled() {
38  		updateSelectedEObject();
39  
40  		if (selectedEObject instanceof Port) {
41  			return true;
42  		}
43  
44  		return false;
45  	}
46  
47  	@Override
48  	public Object execute(ExecutionEvent event) throws ExecutionException {
49  
50  		if (selectedEObject instanceof Port) {
51  			final Port port = (Port) selectedEObject;
52  
53  			CommandSupport.exec(CONFIGURE_PORT, event, new RunnableWithResult() {
54  		
55  				@Override
56  				public CommandResult run() {
57  					PortConfigurationDialog portConfigurationDialog =
58  							new PortConfigurationDialog(Display.getDefault().getActiveShell(), port);
59  					portConfigurationDialog.open();
60  					if (portConfigurationDialog.getReturnCode() == IDialogConstants.OK_ID) {
61  						return CommandResult.newOKCommandResult();
62  					}
63  					else {
64  						return CommandResult.newCancelledCommandResult();
65  					}
66  				}
67  			});
68  		}
69  		return null;
70  	}
71  }
