48  /**
49   * Synchronize component implementation classes.
50   * An implementation depends on the component type in the following way:
51   * (1) It has to realize the interfaces that are provided by the components ports, unless
52   * the port is delegated
53   * (1b) For each realized interface, its operations need to be implemented (the tool should detect and report
54   * double entries which can be resolved via delegation)
55   * (2) It can use interface that are required by the component ports. These changes are reflected in the
56   * context object, but are currently not propagated on an implementation level.
57   *
58   * The first aspect may be triggered by
59   * (a) modification of the port (name, kind or type)
60   * (b) modification due to changes of the interface: rename the interface itself or add, remove or rename operations
61   * and add, remove or rename parameters.
62   *
63   * InterfaceRealizations are implemented by means of stereotyped generalizations. This has the advantage
64   * that the operation itself is inherited and does not have to be synchronized.
65   *
66   * @author ansgar
67   *
68   */
69  public class CompImplSync {
70  
71  	public static void syncViaInterface(Interface intf) {
72  		syncViaInterface(intf, null);
73  	}

75  	/**
76  	 * Delegate a synchronization operation to one or more implementations when an interface
77  	 * (which may have changed) is given. I.e. this operation is called when an operation
78  	 * of an implemented interface is added, removed or modified.
79  	 *
80  	 * @param an
81  	 *            interface which may be implemented by a set of component implementations.
82  	 * @param port
83  	 */
84  	public static void syncViaInterface(Interface intf, BehavioralFeature toBeRemoved) {
85  		// find all implementations of this interface
86  		for (DirectedRelationship relationship : intf.getTargetDirectedRelationships(UMLPackage.eINSTANCE.getInterfaceRealization())) {
87  			InterfaceRealization ir = (InterfaceRealization) relationship;
88  			Classifier cl = ir.getImplementingClassifier();
89  			if (cl instanceof Class) {
90  				Activator.log.info(String.format(Messages.CompImplSync_InfoSyncIntf, cl.getName()));
91  				Class implementation = (Class) cl;
92  				// syncRealizations (implementation);
93  				CompImplSync.interfaceModifications(implementation, toBeRemoved);
94  			}
95  		}
96  	}

98  	/**
99  	 * Delegate a synchronization operation to one or more implementations when a component
100 	 * type is given.
101 	 * The function will in particular handle realization operations due to provided
102 	 * ports and the behaviors associated with it (e.g. opaque-behaviors implementing the provided
103 	 * operations of a component)
104 	 *
105 	 * @param a
106 	 *            component type or a realized interface
107 	 * @param port
108 	 */
109 	public static boolean syncViaType(Class compType, boolean addOnly) {
110 		// find all implementations ...
111 		boolean foundGeneralization = false;
112 		for (DirectedRelationship relationship : compType.getTargetDirectedRelationships(UMLPackage.eINSTANCE.getGeneralization())) {
113 			if (relationship instanceof Generalization) {
114 				foundGeneralization = true;
115 				Classifier cl = ((Generalization) relationship).getSpecific();
116 				if (cl instanceof Class) {
117 					Class implementation = (Class) cl;
118 					updatePorts(implementation);
119 					if (!cl.isAbstract()) {
120 						// add realization relationship only for implementations
121 						if (addOnly) {
122 							addRealizations(implementation);
123 							// syncContextOps (implementation, false);
124 						} else {
125 							syncRealizations(implementation);
126 							// syncContextOps (implementation, true);
127 						}
128 					}
129 				}
130 			}
131 		}
132 		return foundGeneralization;
133 	}

152 	/**
153 	 * Update the ports of a class, i.e. recalculate its derived interfaces
154 	 *
155 	 * @param component
156 	 * @param port
157 	 */
158 	public static void updatePorts(Class implementation) {
159 		for (Port port : PortUtils.getAllPorts(implementation)) {
160 			org.eclipse.papyrus.designer.components.FCM.Port fcmPort = UMLUtil.getStereotypeApplication(port, org.eclipse.papyrus.designer.components.FCM.Port.class);
161 			// verify if in same eResource (= editable)
162 			if ((fcmPort != null) && (port.eResource() == implementation.eResource())) {
163 				fcmPort.update();
164 			}
165 		}
166 	}

175 		EList<Interface> providedIntfs = addRealizations(implementation);
176 		removeRealizations(providedIntfs, implementation);

179 	/**
180 	 * add derived realization relationships of a component implementation
181 	 * due to provided interfaces of its ports
182 	 *
183 	 * TODO: depends on OOTrafo, duplicate functionality
184 	 * 
185 	 * @param implementation a component implementation
186 	 */
187 	public static EList<Interface> addRealizations(Class implementation) {
188 		// create a list of all provided interfaces and check whether realization relationship
189 		// exists. If not, create
190 		EList<Interface> providedIntfs = new BasicEList<Interface>();
191 		for (PortInfo portInfo : PortUtils.flattenExtendedPorts(PortUtils.getAllPorts2(implementation))) {
192 			Interface providedIntf = portInfo.getProvided();
193 			// check, if there is a getter already. In this case, we assume that we should not synchronize
194 			// operations
195 			String opName = PrefixConstants.getP_Prefix + portInfo.getName();
196 			if (implementation.getOwnedOperation(opName, null, null) != null) {
197 				continue;
198 			}
199 			ConnectorEnd connEnd = ConnectorUtil.getDelegation(implementation, portInfo.getPort());
200 			// check that there is no delegation to a part which in turn has to implement the operations.
201 			if ((providedIntf != null) && (connEnd == null)) {
202 				if (providedIntfs.contains(providedIntf)) {
203 					// emit warning that more than one provider for same interface
204 					Activator.log.info("The interface " + providedIntf.getName() + //$NON-NLS-1$
205 							" is provided by more than one port of class <" + implementation.getQualifiedName() + //$NON-NLS-1$
206 							">, inputs would therefore be undistinguishable"); //$NON-NLS-1$
207 				}
208 				providedIntfs.add(providedIntf);
209 				InterfaceRealization ir = implementation.getInterfaceRealization(null,  providedIntf);
210 				if (ir == null) {
211 					RealizationUtils.addRealization(implementation, providedIntf);
212 				} else {
213 					String name = RealizationUtils.calcRealizationName(providedIntf);
214 					if (!name.equals(ir.getName())) {
215 						ir.setName(name);
216 					}
217 				}
218 			}
219 		}
220 		return providedIntfs;
221 	}

223 	/**
224 	 * remove derived realization relationships of a component implementation
225 	 * due to provided interfaces of its ports, i.e. remove those for which the
226 	 * port no longer provides the interface.
227 	 *
228 	 * @param implementation
229 	 */
230 	private static void removeRealizations(EList<Interface> providedIntfs, final Class implementation) {
231 		// now loop through all interface realization relations and remove those that are not
232 		// related to a provided port.
233 		final EList<InterfaceRealization> toBeRemoved = new BasicEList<InterfaceRealization>();
234 		for (InterfaceRealization ir : implementation.getInterfaceRealizations()) {
235 			String name = ir.getName();
236 			// automatically added interface realization is identified via its name (simpler
237 			// compared to use of stereotype)
238 			if ((name != null) && name.startsWith(RealizationUtils.REALIZATION_OF)) {
239 				Interface inheritedIntf = ir.getContract();
240 				if (!providedIntfs.contains(inheritedIntf)) {
241 					toBeRemoved.add(ir);
242 				}
243 			}
244 		}
245 		if (toBeRemoved.size() > 0) {
246 			// remove InterfaceRealization, since not part of provided interfaces
247 			CommandSupport.exec(TransactionUtil.getEditingDomain(implementation), Messages.CompImplSync_InfoSyncViaImpl, new Runnable() {
248 
249 				@Override
250 				public void run() {
251 					// implCopy = implementation;
252 					for (InterfaceRealization ir : toBeRemoved) {
253 						ir.destroy();
254 					}
255 				}
256 			});
257 		}
258 	}

260 	/**
261 	 * The interface that types a port has been modified. Subsequently, the names of
262 	 * methods have to be adapted accordingly, methods have to be removed or added
263 	 *
264 	 * @param implementation
265 	 * @param port
266 	 * @param newName
267 	 */
268 	public static void interfaceModifications(Class implementation, BehavioralFeature toBeRemoved) {
269 		syncDerivedOperations(implementation);
270 
271 		EList<Operation> ownedOperations = implementation.getOwnedOperations();
272 		// loop through all methods, adapt name according to specification name.
273 		Iterator<Behavior> behaviors = implementation.getOwnedBehaviors().iterator();
274 		while (behaviors.hasNext()) {
275 			Behavior behavior = behaviors.next();
276 			BehavioralFeature bf = behavior.getSpecification();
277 			if ((bf != null) && ownedOperations.contains(bf)) {
278 				BehavioralFeature sourceBf = (BehavioralFeature)
279 						UpdateUtils.getSource(bf);
280 
281 				// if an operation is scheduled for deletion, it still exists at this
282 				// point (This operation is called by an InterfaceListener which gets called
283 				// before the deletion takes place) and the specification of associated behaviors
284 				// still point to it. Therefore the check toBeRemoved is added.
285 				if (sourceBf != toBeRemoved) {
286 					String name = calcBehaviorName(sourceBf, implementation);
287 					if (!name.equals(behavior.getName())) {
288 						behavior.setName(name);
289 					}
290 				}
291 			}
292 		}
293 	}

295 	public static void syncDerivedOperations(Class implementation) {
296 		// check for unimplemented operations
297 		// EList<DirectedRelationship> relShips = implementation.getSourceDirectedRelationships (UMLPackage.eINSTANCE.getInterfaceRealization ());
298 		EList<Interface> realizedInterfaces = new BasicEList<Interface>();
299 		// implementation.getInterfaceRealizations returns wrong value??
300 
301 		EList<Operation> ownedOperations = implementation.getOwnedOperations();
302 		for (InterfaceRealization ir : implementation.getInterfaceRealizations()) {
303 			/*
304 			 * for (DirectedRelationship relShip : relShips)
305 			 * if (relShip instanceof InterfaceRealization) {
306 			 * InterfaceRealization ir = (InterfaceRealization) relShip;
307 			 */
308 			Interface intf = ir.getContract();
309 			if (intf == null) {
310 				// should not happen?
311 				continue;
312 			}
313 			realizedInterfaces.add(intf);
314 
315 			for (Operation operation : intf.getAllOperations()) {
316 				Operation copiedOperation = (Operation)
317 						UpdateUtils.getDerivedElement(ownedOperations, operation);
318 				if (copiedOperation == null) {
319 					// no owned operation derived from the interface operation found.
320 					// Would indicate normally that there is no suitable owned operation.
321 					// However: stereotype values may not be available during model load, check
322 					// whether there is an identical operation, before copying one.
323 					copiedOperation = OperationUtils.getSameOperation(operation, implementation);
324 					if (copiedOperation == null) {
325 						copiedOperation = implementation.createOwnedOperation(operation.getName(), null, null);
326 						OperationUtils.syncOperation(operation, copiedOperation);
327 						copiedOperation.setIsAbstract(false);
328 					}
329 					UpdateUtils.setSource(copiedOperation, operation);
330 				} else if (!OperationUtils.isSameOperation(operation, copiedOperation)) {
331 					OperationUtils.syncOperation(operation, copiedOperation);
332 					copiedOperation.setIsAbstract(false);
333 				}
334 			}
335 		}
336 		for (Behavior method : implementation.getOwnedBehaviors()) {
337 			SyncBehaviorParameters.syncParameters(method);
338 		}
339 
340 		// remove operations that are no longer provided via an interface of a port (and
341 		// that are derived elements, i.e. have a source attribute)
342 		Iterator<Operation> ownedOpsIter = ownedOperations.iterator();
343 		while (ownedOpsIter.hasNext()) {
344 			Operation ownedOp = ownedOpsIter.next();
345 
346 			DerivedElement de = UMLUtil.getStereotypeApplication(ownedOp, DerivedElement.class);
347 			if (de != null) {
348 				if (de.getSource() instanceof Operation) {
349 
350 					Operation sourceOp = (Operation) de.getSource();
351 					// check, whether sourceOp is offered by one of the realized interfaces
352 					boolean contains = false;
353 					for (Interface realizedIntf : realizedInterfaces) {
354 						if (realizedIntf.getAllOperations().contains(sourceOp)) {
355 							contains = true;
356 						}
357 					}
358 					if (!contains) {
359 						// operation belongs to an interface which is not implemented => remove
360 						ownedOpsIter.remove();
361 						ownedOp.destroy();
362 					}
363 				}
364 				else if (de.getSource() == null) {
365 					// source element does not exist => remove
366 					ownedOpsIter.remove();
367 					ownedOp.destroy();
368 				}
369 			}
370 		}
371 	}
