30  package org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.body;
31  
32  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.Node;
33  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.expr.AnnotationExpr;
34  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.internal.Utils;
35  
36  import java.util.List;
37  
38  /**
39   * @author Julio Vilmar Gesser
40   */
41  public abstract class BodyDeclaration extends Node implements AnnotableNode {
42  
43      private List<AnnotationExpr> annotations;
44  
45      public BodyDeclaration() {
46      }
47  
48      public BodyDeclaration(List<AnnotationExpr> annotations) {
49      	setAnnotations(annotations);
50      }
51  
52      public BodyDeclaration(int beginLine, int beginColumn, int endLine, int endColumn, List<AnnotationExpr> annotations) {
53          super(beginLine, beginColumn, endLine, endColumn);
54      	setAnnotations(annotations);
55      }
56  
57      @Override
58      public final List<AnnotationExpr> getAnnotations() {
59          annotations = Utils.ensureNotNull(annotations);
60          return annotations;
61      }
62  
63      /**
64       *
65       * @param annotations a null value is currently treated as an empty list. This behavior could change
66       *                    in the future, so please avoid passing null
67       */
68      public final void setAnnotations(List<AnnotationExpr> annotations) {
69          this.annotations = annotations;
70  		setAsParentNodeOf(this.annotations);
71      }
72  }
