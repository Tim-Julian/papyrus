68  	 * Retrieve an element from a list of named elements via its name
69  	 *
70  	 * @param namedElementList
71  	 * @param name
72  	 * @return
73  	 */
74  	public static NamedElement getNamedElementFromList(EList<? extends EObject> elementList, String name) {
75  		for (EObject element : elementList) {
76  			if (element instanceof NamedElement) {
77  				NamedElement namedElement = (NamedElement) element;
78  				if ((namedElement.getName() != null) && namedElement.getName().equals(name)) {
79  					return namedElement;
80  				}
81  			}
82  		}
83  		return null;
84  	}

105 	 * this method returns the component type of an implementation. It is based
106 	 * on the modeling convention that implementations inherit from types.
107 	 * TODO: It currently returns the first abstract (direct) superclass
108 	 * 
109 	 * @param clazz
110 	 *            A component implementation
111 	 * @return the first abstract superclass
112 	 */
113 	public static Class componentType(Class clazz) {
114 		if (clazz.isAbstract()) {
115 			return clazz;
116 		} else {
117 			for (Class superclass : clazz.getSuperClasses()) {
118 				if (superclass.isAbstract()) {
119 					return superclass;
120 				}
121 			}
122 			return null;
123 		}
124 	}

158 	/**
159 	 * Get an element via its qualified name. This function will find all elements in the
160 	 * resource set that have this qualified name (whether imported or not)
161 	 *
162 	 * @param rs
163 	 *            a resource set
164 	 * @param qualifiedName
165 	 *            the qualified name of an element
166 	 * @return the found element or null
167 	 */
168 	public static NamedElement getQualifiedElementFromRS(ResourceSet rs, String qualifiedName) {
169 		for (Resource resource : rs.getResources()) {
170 			if (resource instanceof UMLResource) {
171 				if (resource.getContents().size() > 0) {
172 					EObject topLevelElem = resource.getContents().get(0);
173 					if (topLevelElem instanceof Package) {
174 						NamedElement ne = getQualifiedElement((Package) topLevelElem, qualifiedName);
175 						if (ne != null) {
176 							return ne;
177 						}
178 					}
179 				}
180 			}
181 		}
182 		return null;
183 	}

185 	/**
186 	 * Get an element via its qualified name. It will ignore imports
187 	 *
188 	 * @param root
189 	 *            the root element (package) of a model
190 	 * @param qualifiedName
191 	 *            the qualified name of an element
192 	 * @return the found element or null
193 	 */
194 	public static NamedElement getQualifiedElement(Package root, String qualifiedName) {
195 		String[] path = qualifiedName.split(NamedElement.SEPARATOR);
196 		if (root.getName().equals(path[0])) {
197 			return getQualifiedElement(root, path, 1);
198 		}
199 		return null;
200 	}

202 	/**
203 	 * Retrieve an element via its qualified name within a root package.
204 	 *
205 	 * @return the found element, if it exists
206 	 */
207 	public static NamedElement getQualifiedElement(Package root, String[] path, int pos) {
208 		if (root == null) {
209 			return null;
210 		}
211 		if (pos == path.length) {
212 			return root;
213 		} else if (pos < path.length) {
214 			NamedElement segmentElem = root.getPackagedElement(path[pos++]);
215 			if (segmentElem != null) {
216 				if (pos == path.length) {
217 					return segmentElem;
218 				} else if (segmentElem instanceof Package) {
219 					// requires further recursion
220 					NamedElement foundElement = getQualifiedElement((Package) segmentElem, path, pos);
221 					// return, if not found
222 					if (foundElement != null) {
223 						return foundElement;
224 					}
225 				}
226 			}
227 		}
228 		return null;
229 	}

246 	@SuppressWarnings("unchecked")
247 	public static <T extends EObject> EList<T> getAllElementsOfType(Element examineElement, java.lang.Class<T> clazz, EList<Element> visitedPackages) {
248 		EList<T> list = new UniqueEList<T>();
249 		for (Element element : examineElement.allOwnedElements()) {
250 			if (element instanceof Package) {
251 				if (!visitedPackages.contains(element)) {
252 					visitedPackages.add(element);
253 					list.addAll(getAllElementsOfType(element, clazz, visitedPackages));
254 				}
255 			} else if (clazz.isInstance(element)) {
256 				list.add((T) element);
257 			}
258 		}
259 		return list;
260 	}

297 	 * Convenience function: Declare a dependency from source to destination. The function checks,
298 	 * if a dependency already exists to avoid double dependencies.
299 	 *
300 	 * @param source
301 	 *            source type of the dependency
302 	 * @param dest
303 	 *            destination of the dependency
304 	 */
305 	public static void declareDependency(Type source, Type dest) {
306 		// check, if a relationship already exists
307 		for (Relationship dependency : source.getRelationships(UMLPackage.eINSTANCE.getDependency())) {
308 			if (((Dependency) dependency).getSuppliers().contains(dest)) {
309 				return;
310 			}
311 		}
312 		source.createDependency(dest);
313 	}

335 	/**
336 	 * This method returns all types that are referenced by a classifier. This includes
337 	 * attribute types, types within operations as well as inherited types.
338 	 * This is useful to generate the #includes
339 	 * (which is used by the Acceleo code within the packaging plugin)
340 	 */
341 	public static EList<Classifier> getReferencedClassifiers(Classifier classifier) {
342 		EList<Classifier> list = new UniqueEList<Classifier>();
343 		list.addAll(classifier.parents());
344 
345 		if (classifier instanceof Class) {
346 			// get classifiers referenced by attributes
347 			for (Operation operation : ((Class) classifier).getOwnedOperations()) {
348 				for (Parameter parameter : operation.getOwnedParameters()) {
349 					Type type = parameter.getType();
350 					if (type instanceof Classifier) {
351 						list.add((Classifier) type);
352 					}
353 				}
354 			}
355 
356 			// get classifiers referenced by attributes
357 			for (Property attribute : ((Class) classifier).getOwnedAttributes()) {
358 				Type type = attribute.getType();
359 				if (type instanceof Classifier) {
360 					list.add((Classifier) type);
361 				}
362 			}
363 		}
364 		return list;
365 	}

367 	/**
368 	 * Return the interface which owns an operation that is implemented by a class.
369 	 * Context: a class might implement several interfaces by defining their operations.
370 	 * The operation is useful in the context of state-machines: when a transition is triggered by
371 	 * the call of an operation of the class, we'd like to know which interceptor (for which interface)
372 	 * belongs to it (since the operations are enumerated within each interface).
373 	 * TODO: move operation into state-chart java code
374 	 * TODO: would not work for ROOM ports typed with a collaboration
375 	 *
376 	 * @param operation
377 	 * @return the interface which the operation belongs
378 	 */
379 	public static Interface implementsInterface(Operation operation) {
380 		Element owner = operation.getOwner();
381 		if (owner instanceof BehavioredClassifier) {
382 			String name = operation.getName();
383 			EList<Type> types = new BasicEList<Type>();
384 			for (Parameter parameter : operation.getOwnedParameters()) {
385 				types.add(parameter.getType());
386 			}
387 			// loop over implemented realizations. Do not rely on FCM derivedElement information
388 			// as it might be missing on some models (it would point from an operation of the class
389 			// to the associated operation of the interface)
390 			for (InterfaceRealization ir : ((BehavioredClassifier) owner).getInterfaceRealizations()) {
391 				// check for types to allow for overloading
392 				Operation candidate = ir.getContract().getOwnedOperation(name, null, types);
393 				if (candidate != null) {
394 					return ir.getContract();
395 				}
396 			}
397 		} else if (owner instanceof Interface) {
398 			return (Interface) owner;
399 		}
400 		return null;
401 	}

403 	public static <T extends EObject> EList<T> reverse(EList<T> list) {
404 		EList<T> reverseList = new BasicEList<T>();
405 		for (int i = list.size() - 1; i >= 0; i--) {
406 			reverseList.add(list.get(i));
407 		}
408 		return reverseList;
409 	}

412 		EList<Namespace> list = new BasicEList<Namespace>(element.allNamespaces());
413 
414 		if (list.size() < 1) {
415 			return null;
416 		}
417 		list.remove(list.size() - 1); // remove last element (top-level)
418 		return list;

453 	 *

455 	 * @return

457 	public static String varName2(String umlName) {

462 		return umlName;
