58  	@Override
59  	public boolean approveRequest(IEditCommandRequest request) {
60  		if (request instanceof GetEditContextRequest) {
61  			GetEditContextRequest context = (GetEditContextRequest) request;
62  			if (context.getEditCommandRequest() instanceof CreateElementRequest) {
63  				return approveCreateElementRequest((CreateElementRequest) context.getEditCommandRequest());
64  			}
65  		}
66  		return super.approveRequest(request);
67  	}
68  
69  	/**
70  	 * Allow creation only, if container is a componentModule (could be indirectly)
71  	 */
72  	protected boolean approveCreateElementRequest(CreateElementRequest request) {
73  		IElementType type = request.getElementType();
74  		EObject container = request.getContainer();
75  		if (type != null && container instanceof Package) {
76  			if (CreateUtils.nearestPackageWithStereo((Element) container, ComponentModule.class) != null) {
77  				return true;
78  			}
79  		}
80  		return false;
81  	}

93  	/**
94  	 * @see org.eclipse.gmf.runtime.emf.type.core.edithelper.AbstractEditHelperAdvice#getAfterConfigureCommand(org.eclipse.gmf.runtime.emf.type.core.requests.CreateElementRequest)
95  	 */
96  	@Override
97  	protected ICommand getAfterConfigureCommand(ConfigureRequest request) {
98  		CompositeCommand compositeCommand = new CompositeCommand("Component implementation configuration command");
99  		EObject newElement = request.getElementToConfigure();
100 		if (newElement == null) {
101 			return super.getAfterConfigureCommand(request);
102 		}
103 		IElementEditService commandProvider = ElementEditServiceUtils.getCommandProvider(newElement);
104 
105 		IStaticContentProvider cp = new UCMContentProvider(PackageUtil.getRootPackage((Element) newElement), UMLPackage.eINSTANCE.getClass_(), ComponentType.class);
106 
107 		EnhancedPopupMenu popupMenu = MenuHelper.createPopupMenu(cp, "Choose component type", true);
108 
109 		if (popupMenu.show(Display.getDefault().getActiveShell())) {
110 			Object result = popupMenu.getSubResult();
111 			if (result.equals(MenuHelper.CREATE_NEW_TYPE) && newElement instanceof Element) {
112 				CreateElementRequest createComponentType = new CreateElementRequest(((Element) newElement).getNearestPackage(), /* UMLElementTypes.CLASS */ UCMElementTypesEnumerator.COMPONENT_TYPE,
113 						UMLPackage.eINSTANCE.getPackage_PackagedElement()/* Ucm_componentsPackage.eINSTANCE.getComponentType() */);
114 				ICommand createComponentTypeCmd = commandProvider.getEditCommand(createComponentType);
115 				try {
116 					createComponentTypeCmd.execute(null, null);
117 				} catch (ExecutionException e) {
118 					Activator.log.error(e);
119 				}
120 				result = GMFCommandUtils.getCommandEObjectResult(createComponentTypeCmd);
121 			}
122 			if (result instanceof Type) {
123 				Type componentType = (Type) result;
124 
125 				CreateElementRequest createGeneralizationReq = new CreateElementRequest(newElement, UMLElementTypes.GENERALIZATION, UMLPackage.eINSTANCE.getClassifier_Generalization());
126 				ICommand createGeneralizationCmd = commandProvider.getEditCommand(createGeneralizationReq);
127 				compositeCommand.add(createGeneralizationCmd);
128 				// TODO: command executed here + later? (since part of composition)
129 				try {
130 					createGeneralizationCmd.execute(null, null);
131 				} catch (ExecutionException e) {
132 					Activator.log.error(e);
133 				}
134 				newElement = GMFCommandUtils.getCommandEObjectResult(createGeneralizationCmd);
135 
136 				// addToDiagram(newElement);
137 				SetRequest setType = new SetRequest(createGeneralizationReq.getNewElement(), UMLPackage.eINSTANCE.getGeneralization_General(), componentType);
138 				compositeCommand.add(commandProvider.getEditCommand(setType));
139 			}
140 		} else {
141 			// User cancelled the menu
142 		}
143 
144 		return compositeCommand.isEmpty() ? super.getAfterConfigureCommand(request) : compositeCommand;
145 	}
