15  package org.eclipse.papyrus.designer.components.validation.constraints;
16  
17  import org.eclipse.core.runtime.IStatus;
18  import org.eclipse.emf.validation.AbstractModelConstraint;
19  import org.eclipse.emf.validation.IValidationContext;
20  import org.eclipse.papyrus.designer.transformation.profile.Transformation.DerivedElement;
21  import org.eclipse.papyrus.uml.tools.utils.StereotypeUtil;
22  import org.eclipse.uml2.uml.Operation;
23  import org.eclipse.uml2.uml.Parameter;
24  import org.eclipse.uml2.uml.ParameterDirectionKind;
25  
26  /**
27   * Verify that an operation has at most one return parameter
28   *
29   * Whereas this rule would make sense in general, it has been made specific to
30   * SW Designer by verifying whether FCM is applied (check if DerivedElement stereotype
31   * is applicable)
32   */
33  public class OperationsWithTwoReturnValues extends AbstractModelConstraint
34  {
35  	@Override
36  	public IStatus validate(IValidationContext ctx)
37  	{
38  		Operation operation = (Operation) ctx.getTarget();
39  		boolean firstReturn = true;
40  		for (Parameter parameter : operation.getOwnedParameters()) {
41  			if (parameter.getDirection() == ParameterDirectionKind.RETURN_LITERAL) {
42  				if (firstReturn) {
43  					firstReturn = false;
44  				}
45  				else {
46  					if (StereotypeUtil.isApplicable(operation, DerivedElement.class)) {
47  						return ctx.createFailureStatus("The operation '" + operation.getName() + "' has more than one return parameter. It must have at most one");
48  					}
49  					else {
50  						break;
51  					}
52  				}
53  			}
54  		}
55  		return ctx.createSuccessStatus();
56  	}
57  }
