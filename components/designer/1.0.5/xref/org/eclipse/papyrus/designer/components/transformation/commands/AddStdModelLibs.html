11  package org.eclipse.papyrus.designer.components.transformation.commands;
12  
13  import org.eclipse.emf.common.util.BasicEList;
14  import org.eclipse.emf.common.util.EList;
15  import org.eclipse.emf.common.util.URI;
16  import org.eclipse.emf.ecore.resource.ResourceSet;
17  import org.eclipse.emf.transaction.RecordingCommand;
18  import org.eclipse.emf.transaction.TransactionalEditingDomain;
19  import org.eclipse.papyrus.designer.transformation.base.utils.LibraryUtils;
20  import org.eclipse.uml2.uml.Package;
21  import org.eclipse.uml2.uml.PackageImport;
22  
23  /**
24   * This class adds "standard" model libraries for component-based development: the FCM profile, parts of the MARTE profile (allocation)
25   * and the associated package imports to your model.
26   */
27  public class AddStdModelLibs extends RecordingCommand {
28  
29  	public AddStdModelLibs(Package pkg, TransactionalEditingDomain domain) {
30  		super(domain, CMD_LABEL);
31  		selectedPkg = pkg;
32  		availableImportPackages = null;
33  		importList = null;
34  	}
35  
36  	public static final String CMD_LABEL = "Add Designer CORE libraries"; //$NON-NLS-1$
37  
38  	public static final String DESIGNER_BASIC_TRAFOS_URI = "pathmap://DML_TRAFOS/trafos.uml"; //$NON-NLS-1$
39  
40  	public static final String DESIGNER_MARTE_CALLS_URI = "pathmap://QML_MARTE/marte.uml"; //$NON-NLS-1$
41  
42  	protected EList<PackageImport> importList;
43  
44  	protected EList<PackageImport> availableImportPackages;
45  
46  	protected Package selectedPkg;
47  
48  	/**
49  	 * Check whether a package import is already done
50  	 *
51  	 * @param selectedPkg
52  	 * @param pi
53  	 * @return
54  	 */
55  	boolean isAlreadyImported(Package selectedPkg, PackageImport pi) {
56  		for (PackageImport existingPI : selectedPkg.getPackageImports()) {
57  			if ((existingPI.getImportedPackage() == null) ||
58  					(pi.getImportedPackage() == null)) {
59  				// import package are null (should not happen?!)
60  				continue;
61  			}
62  			if ((existingPI.getImportedPackage().getName() == null) ||
63  					(pi.getImportedPackage().getName() == null)) {
64  				// import package name not set (should not happen?!)
65  				continue;
66  			} else if (existingPI.getImportedPackage().getName().equals(pi.getImportedPackage().getName())) {
67  				return true;
68  			}
69  		}
70  		return false;
71  	}
72  
73  	/**
74  	 * @return a list of package-imports for standard SW Designer model libraries
75  	 */
76  	public EList<PackageImport> getAvailableImportPackages() {
77  		if (availableImportPackages != null) {
78  			return availableImportPackages;
79  		}
80  		final ResourceSet resourceSet = selectedPkg.eResource().getResourceSet();
81  		availableImportPackages = new BasicEList<PackageImport>();
82  
83  		PackageImport pi = LibraryUtils.getModelLibraryImportFromURI(URI.createURI(DESIGNER_BASIC_TRAFOS_URI), resourceSet);
84  		if (pi != null) {
85  			availableImportPackages.add(pi);
86  		}
87  		pi = LibraryUtils.getModelLibraryImportFromURI(URI.createURI(DESIGNER_MARTE_CALLS_URI), resourceSet);
88  		if (pi != null) {
89  			availableImportPackages.add(pi);
90  		}
91  		return availableImportPackages;
92  	}
93  
94  	/**
95  	 * @return a list of package-imports that are already imported (within the selected pkg).
96  	 */
97  	public EList<PackageImport> getAlreadyImportedPackages() {
98  		EList<PackageImport> availablePackages = getAvailableImportPackages();
99  		EList<PackageImport> selection = new BasicEList<PackageImport>();
100 		for (PackageImport pi : availablePackages) {
101 			if (isAlreadyImported(selectedPkg, pi)) {
102 				selection.add(pi);
103 			}
104 		}
105 		return selection;
106 	}
107 
108 	/**
109 	 * Initialize import list. Must be called, before the command gets executed.
110 	 *
111 	 * @param importList
112 	 *            a list of package-imports that will be added to the selectedPkg when
113 	 *            the command gets executed.
114 	 */
115 	public void setImportList(EList<PackageImport> importList) {
116 		this.importList = importList;
117 	}
118 
119 
120 	/**
121 	 * Initialize import list. Must be called, before the command gets executed.
122 	 * Convenience method for dialogs that return an array.
123 	 *
124 	 * @param importArray
125 	 *            an array of objects (which must contain package-imports) that will
126 	 *            be added to the selectedPkg when the command gets executed.
127 	 */
128 	public void setImportList(Object importArray[]) {
129 		importList = new BasicEList<PackageImport>();
130 		for (Object importPkg : importArray) {
131 			if (importPkg instanceof PackageImport) {
132 				importList.add((PackageImport) importPkg);
133 			}
134 		}
135 	}
136 
137 	@Override
138 	protected void doExecute() {
139 
140 		for (PackageImport pi : importList) {
141 			if (!isAlreadyImported(selectedPkg, pi)) {
142 				selectedPkg.getPackageImports().add(pi);
143 			}
144 		}
145 	}
146 }
