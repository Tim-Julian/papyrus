34  public class LocateCppProject {
35  
36  	private static final boolean Headless = TestInfo.runsHeadless();
37  
38  	/**
39  	 * Locate and return the target project for the given packageable element. Return null if
40  	 * no target project can be found.
41  	 *
42  	 * Ensures that the target project is correctly setup to contain generated C/C++ code. Does
43  	 * not create a new project, but may modify existing ones.
44  	 *
45  	 * @param pe
46  	 *            a packageable element within a model
47  	 * @param createIfMissing
48  	 *            if true, ask the user to apply the C++ nature if required.
49  	 * @return the associated project, if the C++ nature is applied.
50  	 */
51  	public static IProject getTargetProject(PackageableElement pe, boolean createIfMissing) {
52  		Package rootPkg = PackageUtil.getRootPackage(pe);
53  
54  		IWorkspaceRoot root = ResourcesPlugin.getWorkspace().getRoot();
55  
56  		IPreferenceStore store = Activator.getDefault().getPreferenceStore();
57  		String prefix = store != null ? store.getString(CppCodeGenConstants.P_PROJECT_PREFIX) : "bad preferences."; //$NON-NLS-1$
58  		
59  		org.eclipse.papyrus.designer.languages.common.profile.Codegen.Project projectStereo =
60  				UMLUtil.getStereotypeApplication(rootPkg, org.eclipse.papyrus.designer.languages.common.profile.Codegen.Project.class);
61  		
62  		String projectName;
63  		if (projectStereo != null) {
64  			projectName = projectStereo.getProjectName();
65  		}
66  		else {
67  			projectName = prefix + rootPkg.getName();
68  					
69  		}
70  		IProject modelProject = root.getProject(projectName);
71  		
72  		if (!modelProject.exists()) {
73  			if (Headless)
74  			{
75  				try {
76  					modelProject.create(null);
77  				} catch (CoreException e) {
78  					return null;
79  				}
80  			}
81  			else
82  			{
83  				boolean create = createIfMissing && openQuestion(
84  						Messages.LocateCppProject_CreateTargetProjectTitle,
85  						String.format(Messages.LocateCppProject_CreateTargetProjectDesc, projectName));
86  				if (create) {
87  					ILangProjectSupport langSupport = LanguageProjectSupport.getProjectSupport(CppModelElementsCreator.LANGUAGE_NAME);
88  					if (langSupport != null) {
89  						modelProject = langSupport.createProject(projectName);
90  						langSupport.setSettings(modelProject, langSupport.initialConfigurationData());
91  						if (modelProject == null) {
92  							return null;
93  						}
94  					}
95  					else {
96  						return null;
97  					}
98  				}
99  				else {
100 					return null;
101 				}
102 			}
103 		}
104 
105 		// Make sure the target project is open. If it was just created then it is likely open,
106 		// however if it is an old project then it could have been closed.
107 		if (!modelProject.isOpen()) {
108 			try {
109 				modelProject.open(null);
110 			} catch (CoreException e) {
111 				return null;
112 			}
113 		}
114 
115 		// Make sure the target project has the C and C++ build natures.
116 		try {
117 			if (!modelProject.hasNature(CCProjectNature.CC_NATURE_ID)) {
118 				boolean apply = createIfMissing && (Headless || openQuestion(
119 						Messages.LocateCppProject_ApplyCNatureTitle,
120 						Messages.LocateCppProject_ApplyCNatureDesc));
121 				if (!apply) {
122 					return null;
123 				}
124 				CProjectNature.addCNature(modelProject, null);
125 				CCProjectNature.addCCNature(modelProject, null);
126 			}
127 		} catch (CoreException e) {
128 			Activator.log.error(e);
129 		}
130 		return modelProject;
131 	}
132 	
133 	private static boolean openQuestion(final String title, final String message)
134 	{
135 		final boolean[] ret = new boolean[] { false };
136 		Display.getDefault().syncExec(new Runnable() {
137 			@Override
138 			public void run() {
139 				ret[0] = MessageDialog.openQuestion(Display.getCurrent().getActiveShell(), title, message);
140 			}
141 		});
142 		return ret[0];
143 	}
144 }
