12  package org.eclipse.papyrus.designer.languages.common.extensionpoints;
13  
14  import org.eclipse.core.resources.IProject;
15  import org.eclipse.uml2.uml.Classifier;
16  import org.eclipse.uml2.uml.Package;
17  
18  /**
19   * Common interface to create projects for a specific programming language.
20   */
21  public interface ILangProjectSupport {
22  
23  	/**
24  	 * Create a project for a specific language and configure it (according to
25  	 * gathered configuration data before). In case of C/C++ for instance, a CDT
26  	 * project should be created, in case of Java, a JDT project. An
27  	 * implementation should call setProject and setSettings before returning
28  	 * the project
29  	 *
30  	 * @param projectName
31  	 *            the named of the project
32  	 * @return the created project
33  	 */
34  	public IProject createProject(String projectName);
35  
36  	/**
37  	 * Create a project for a specific language and configure it (according to
38  	 * gathered configuration data before). In case of C/C++ for instance, a CDT
39  	 * project should be created, in case of Java, a JDT project. An
40  	 * implementation should call setProject and setSettings before returning
41  	 * the project. It also adds the Maven nature to the project when the user
42  	 * wants a Maven project to be generated.
43  	 *
44  	 * @param projectName
45  	 *            the named of the project
46  	 * @param modelRoot
47  	 *            the root package of the model used to generate the project
48  	 * @return the created project
49  	 */
50  	public IProject createProject(String projectName, Package modelRoot);
51  
52  	/**
53  	 * Write the project settings from the passed project specific settings.
54  	 * (which may be obtained via gatherConfigData).
55  	 *
56  	 * @param the
57  	 *            operating system for which code should be produced
58  	 */
59  	public void setSettings(IProject project, AbstractSettings settings);
60  
61  	/**
62  	 * Create an initial (empty) configuration.
63  	 * 
64  	 * @return the project specific configuration.
65  	 */
66  	public AbstractSettings initialConfigurationData();
67  
68  	/**
69  	 * Gather configuration data from implementations, e.g. required include
70  	 * paths or libraries. It will enrich the project specific settings.
71  	 *
72  	 * @param implementation
73  	 *            a class copied to the target
74  	 */
75  	public void gatherConfigData(Classifier implementation, AbstractSettings settings);
76  }
