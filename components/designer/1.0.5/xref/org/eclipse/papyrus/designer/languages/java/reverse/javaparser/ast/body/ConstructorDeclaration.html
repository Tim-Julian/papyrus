176     /**
177      * The declaration returned has this schema:
178      *
179      * [accessSpecifier] className ([paramType [paramName]])
180      * [throws exceptionsList]
181      */
182     @Override
183     public String getDeclarationAsString(boolean includingModifiers, boolean includingThrows,
184                                          boolean includingParameterName) {
185         StringBuffer sb = new StringBuffer();
186         if (includingModifiers) {
187             AccessSpecifier accessSpecifier = ModifierSet.getAccessSpecifier(getModifiers());
188             sb.append(accessSpecifier.getCodeRepresenation());
189             sb.append(accessSpecifier == AccessSpecifier.DEFAULT ? "" : " ");
190         }
191         sb.append(getName());
192         sb.append("(");
193         boolean firstParam = true;
194         for (Parameter param : getParameters())
195         {
196             if (firstParam) {
197                 firstParam = false;
198             } else {
199                 sb.append(", ");
200             }
201             if (includingParameterName) {
202                 sb.append(param.toStringWithoutComments());
203             } else {
204                 sb.append(param.getType().toStringWithoutComments());
205             }
206         }
207         sb.append(")");
208         if (includingThrows) {
209             boolean firstThrow = true;
210             for (NameExpr thr : getThrows()) {
211                 if (firstThrow) {
212                     firstThrow = false;
213                     sb.append(" throws ");
214                 } else {
215                     sb.append(", ");
216                 }
217                 sb.append(thr.toStringWithoutComments());
218             }
219         }
220         return sb.toString();
221     }
