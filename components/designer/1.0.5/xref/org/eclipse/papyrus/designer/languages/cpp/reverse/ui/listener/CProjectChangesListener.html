13  package org.eclipse.papyrus.designer.languages.cpp.reverse.ui.listener;
14  
15  import java.util.ArrayList;
16  import java.util.Collections;
17  import java.util.LinkedList;
18  import java.util.List;
19  import java.util.Map;
20  
21  import org.eclipse.cdt.core.model.ElementChangedEvent;
22  import org.eclipse.cdt.core.model.ICElementDelta;
23  import org.eclipse.cdt.core.model.IElementChangedListener;
24  import org.eclipse.papyrus.designer.languages.cpp.reverse.reverse.change.CElementChange;
25  
26  /**
27   * A listener that detects changes on a CDT project.
28   * The events and processed and the listener feeds a map
29   * that will be used for reverse in merge mode.
30   *
31   */
32  public class CProjectChangesListener implements IElementChangedListener {
33  	private Map<String, List<CElementChange>> changesMap;
34  	
35  	public CProjectChangesListener(Map<String, List<CElementChange>> changesMap) {
36  		this.changesMap = changesMap;
37  	}
38  
39  	@Override
40  	public void elementChanged(ElementChangedEvent event) {
41  		String eventName = "";
42  		if (event.getType() == 1) {
43  			eventName = "POST_CHANGE";
44  		} else if (event.getType() == 4) {
45  			eventName = "POST_RECONCILE";
46  		} else {
47  			eventName = event.getType() + "";
48  		}
49  		
50  		System.out.println("Process event of type " + eventName);
51  		
52  		if (event.getType() == ElementChangedEvent.POST_RECONCILE || event.getType() == ElementChangedEvent.POST_CHANGE) {
53  			List<CElementChange> affectedLeafElements = new ArrayList<CElementChange>();
54  			fillAffectedLeafElements(event.getDelta(), affectedLeafElements);
55  			
56  			List<CElementChange> changes = null;
57  			for (CElementChange cElementChange : affectedLeafElements) {
58  				if (cElementChange.getChangeKind() != 0) {
59  					if (cElementChange.getElement().getCProject() != null) {
60  						String projectName = cElementChange.getElement().getCProject().getElementName();
61  						if (changesMap != null) {
62  							if (changesMap.get(projectName) != null) {
63  								changes = changesMap.get(projectName);
64  							} else {
65  								changes = Collections.synchronizedList(new LinkedList<CElementChange>());
66  								changesMap.put(cElementChange.getElement().getCProject().getElementName(), changes);
67  							}
68  						}
69  						
70  						if (changes != null) {
71  							changes.add(cElementChange);
72  							
73  							String changeName = "";
74  							if (cElementChange.getChangeKind() == 1) {
75  								changeName = "ADDED";
76  							} else if (cElementChange.getChangeKind() == 2) {
77  								changeName = "REMOVED";
78  							} else if (cElementChange.getChangeKind() == 4) {
79  								changeName = "CHANGED";
80  							} else {
81  								changeName = cElementChange.getChangeKind() + "";
82  							}
83  							System.out.println("---> " + cElementChange.getElement().getElementName() + " of type " + cElementChange.getElement().getElementType() + " has been " + changeName + " ==> PUT IN MAP");
84  						}
85  					}
86  				} else {
87  					System.out.println("---> " + cElementChange.getElement().getElementName() + " of type " + cElementChange.getElement().getElementType() + " has been " + cElementChange.getChangeKind() + " ==> NOTHING");
88  				}
89  			}
90  		}
91  	}
92  	
93  	private void fillAffectedLeafElements(ICElementDelta delta, List<CElementChange> affectedLeafElements) {
94  		if (affectedLeafElements == null) {
95  			return;
96  		}
97  		
98  		if (delta != null) {
99  			if (delta.getAffectedChildren().length == 0 && delta.getElement() != null) {
100 				affectedLeafElements.add(new CElementChange(delta));
101 			} else {
102 				for (ICElementDelta childDelta : delta.getAffectedChildren()) {
103 					fillAffectedLeafElements(childDelta, affectedLeafElements);
104 				}
105 			}
106 		}
107 	}
108 
109 }
