122 /**
123  * This visitor adapter can be used to save time when some specific nodes needs
124  * to be changed. To do that just extend this class and override the methods
125  * from the nodes who needs to be changed, returning the changed node.
126  * 
127  * @author Julio Vilmar Gesser
128  */
129 public abstract class ModifierVisitorAdapter<A> implements GenericVisitor<Node, A> {
130 
131 	private void removeNulls(final List<?> list) {
132 		for (int i = list.size() - 1; i >= 0; i--) {
133 			if (list.get(i) == null) {
134 				list.remove(i);
135 			}
136 		}
137 	}

139 	@Override public Node visit(final AnnotationDeclaration n, final A arg) {
140 		if (n.getJavaDoc() != null) {
141 			n.setJavaDoc((JavadocComment) n.getJavaDoc().accept(this, arg));
142 		}
143 		final List<AnnotationExpr> annotations = n.getAnnotations();
144 		if (annotations != null) {
145 			for (int i = 0; i < annotations.size(); i++) {
146 				annotations.set(i, (AnnotationExpr) annotations.get(i).accept(this, arg));
147 			}
148 			removeNulls(annotations);
149 		}
150 		final List<BodyDeclaration> members = n.getMembers();
151 		if (members != null) {
152 			for (int i = 0; i < members.size(); i++) {
153 				members.set(i, (BodyDeclaration) members.get(i).accept(this, arg));
154 			}
155 			removeNulls(members);
156 		}
157 		return n;
158 	}
