46  /**
47   * Create an automatic template binding for a connector, i.e. a binding that assures
48   * that the ports of an interaction component match those of an application component
49   * (if possible)
50   */
51  public class ConnectorBinding {

53  	/**
54  	 * Helper function for getActual
55  	 *
56  	 * @param port
57  	 *            a port of the the connector template (not evaluated except for debugging output)
58  	 * @param connector
59  	 *            the connector that is checked for reification
60  	 * @param composite
61  	 *            a composite class
62  	 * @param actual
63  	 *            the current candidate for an actual, i.e. result of previous attempts to bind (at different connector ends)
64  	 * @param isProvided
65  	 *            is an interface provided or required.
66  	 * @return the type at the "other" (makes sense for property connector only) end of a connector
67  	 * @throws TransformationException
68  	 */
69  	private static Type matchOtherEnd(Port port, Feature connector, Type actual, boolean isProvided)
70  			throws TransformationException {
71  		if (connector instanceof Connector) {
72  			return matchOtherEnd(port, (Connector) connector, actual, isProvided);
73  		}
74  		else if (connector instanceof Property) {
75  			return matchOtherEnd(port, (Property) connector, actual, isProvided);
76  		}
77  		return null;
78  	}

80  	private static Type matchOtherEnd(Port port, Connector connector, Type actual, boolean isProvided)
81  			throws TransformationException {
82  		// the template provides the formal parameter at one of its ports. Now examine
83  		// with whom this port is connected, i.e. examine all connectorEnds
84  		// The type of the connected port determines the binding.
85  		Activator.log.info(String.format("", port.getName())); //$NON-NLS-1$
86  		boolean found = false;
87  		for (ConnectorEnd connEnd : connector.getEnds()) {
88  			// the connector end targets a port of a part or the composite (in case of delegation)
89  			ConnectableElement connElem = connEnd.getRole();
90  			if (connElem instanceof Port) {
91  				Port otherPort = (Port) connElem;
92  				Interface otherInterface;
93  
94  				// distinction between assembly or delegation (other end is null) connectors
95  				boolean isAssembly = (connEnd.getPartWithPort() != null);
96  
97  				if (isProvided == isAssembly) {
98  					// need required for other, if first port isProvided and assembly, or if first port isRequired and delegation
99  					otherInterface = PortUtils.getRequired(otherPort);
100 				} else {
101 					otherInterface = PortUtils.getProvided(otherPort);
102 				}
103 				Activator.log.info(String.format(Messages.ConnectorBinding_ConnectorsPort, otherInterface));
104 				if (otherInterface != null) {
105 					if (actual == null || actual == otherInterface) {
106 						actual = otherInterface;
107 						Activator.log.info(String.format(
108 								Messages.ConnectorBinding_InfoActualReturnIntfIs, actual.getQualifiedName()));
109 						found = true;
110 					} else if (actual != otherInterface) {
111 						continue;
112 					}
113 				}
114 			}
115 		}
116 		if (!found) {
117 			throw new TransformationException(String.format(
118 					Messages.ConnectorBinding_CannotFindConsistentBinding,
119 					port.getName(), connector.getName(), connector.getNamespace().getName(), (actual != null) ? actual.getName() : "null"));
120 		}
121 		return actual;
122 	}

125 	private static Type matchOtherEnd(Port port, Property partConnector, Type actual, boolean isProvided)
126 			throws TransformationException {
127 		// the template provides the formal parameter at one of its ports. Now examine
128 		// with whom this port is connected, i.e. examine all connectors of the
129 		// composite. The type of the connected port determines the binding.
130 		Activator.log.info(String.format(
131 				Messages.ConnectorBinding_InfoMatchOtherEnd, port.getName()));
132 
133 		for (Connector connector : partConnector.getClass_().getOwnedConnectors()) {
134 			if (ConnectorUtil.connectsPart(connector, partConnector)) {
135 				// the connector end targets a port of a part or the composite (in case of delegation)
136 				ConnectorEnd connEnd = ConnectorUtil.connEndNotPart(connector, partConnector);
137 				ConnectableElement connElem = connEnd.getRole();
138 				if (connElem instanceof Port) {
139 					Port otherPort = (Port) connElem;
140 					Interface otherInterface;
141 
142 					// distinction between assembly or delegation (other end is null) connectors
143 					boolean isAssembly = (connEnd.getPartWithPort() != null);
144 
145 					if (isProvided == isAssembly) {
146 						// need required for other, if first port isProvided and assembly, or if first port isRequired and delegation
147 						otherInterface = PortUtils.getRequired(otherPort);
148 					} else {
149 						otherInterface = PortUtils.getProvided(otherPort);
150 					}
151 					Activator.log.info(String.format(Messages.ConnectorBinding_InfoConnectsPort, otherInterface));
152 					if (otherInterface != null) {
153 						if (actual == null) {
154 							actual = otherInterface;
155 							Activator.log.info(String.format(
156 									Messages.ConnectorBinding_InfoActualReturnIntfIs, actual.getQualifiedName()));
157 						} else if (actual != otherInterface) {
158 							throw new TransformationException(String.format(
159 									Messages.ConnectorBinding_CannotFindConsistentBinding,
160 									port.getName(), connector.getName(), connector.getNamespace().getName(), actual.getName()));
161 						}
162 					}
163 				}
164 			}
165 		}
166 		return actual;
167 	}

170 	/**
171 	 * This function tries to bind a formal template parameter to an "actual". The criterion
172 	 * for a successful binding is that the (primitive) connections of the reified connector
173 	 * part have a matching port kind.
174 	 * The function makes a matching based on (1) either provided and required interfaces (2)
175 	 * or identical portKinds and different conjugations
176 	 *
177 	 * @param composite
178 	 *            The composite in which a binding should be determined.
179 	 * @param part
180 	 *            the part within the composite for which we try to find a binding,
181 	 *            The information about the part is used to determine which connection
182 	 *            end belongs to the template part and which to another part/port (which
183 	 *            determines the binding).
184 	 * @param template
185 	 *            A template class
186 	 * @param formal
187 	 *            the formal template parameter
188 	 * @throws TransformationException
189 	 *             if no consistent binding can be found
190 	 * @return the actual
191 	 */
192 	public static Type getActual(Feature partOrConnector, Class template, ParameterableElement formal)
193 			throws TransformationException {
194 		Type actual = null;
195 
196 		// loop through all ports of the template, use getAllPorts to include inherited ports as well.
197 		for (Port port : PortUtils.getAllPorts(template)) {
198 			Interface providedIntf = PortUtils.getProvided(port);
199 
200 			// need to check, if the port at the opposite connector is matching, provided that we use a specific
201 			// value for the the formal. Typically, ports of the template are typed with the formal parameter. Thus,
202 			// we would have to infer the formal from a given provided or required interface. This is not possible
203 			// in general (no inverse mapping operation), but the simple case that either the provided or required
204 			// interface is identical to the formal parameter are checked.
205 			//
206 			// 1. provided interface is formal
207 			if (providedIntf == formal) {
208 				actual = matchOtherEnd(port, partOrConnector, actual, true);
209 			}
210 
211 			// 2. required interface is formal
212 			Interface requiredIntf = PortUtils.getRequired(port);
213 			if (requiredIntf == formal) {
214 				actual = matchOtherEnd(port, partOrConnector, actual, false);
215 			}
216 
217 			// 3. port type is formal (but none of the two cases above)
218 			// this check is stronger than the first two, since it requires port kind identity.
219 			// If none of the first two variants holds, the port is translating interfaces
220 			// for which this stronger condition holds.
221 			Type type = PortUtils.getFCMType(port);
222 			if ((providedIntf != formal) && (requiredIntf != formal) && (type == formal)) {
223 				// the template provides the formal parameter at one of its ports. Now examine
224 				// with whom this port is connected, i.e. examine all ends of the
225 				// connector. The type of the connected port determines the binding.
226 				Activator.log.info(String.format(
227 						Messages.ConnectorBinding_InfoProvidedPortTypeMatches, port.getName()));
228 				boolean found = false;
229 
230 				if (partOrConnector instanceof Property) {
231 					Property partConnector = (Property) partOrConnector;
232 					// iterate over the connectors of the composite containing the part
233 					for (Connector connector : partConnector.getClass_().getOwnedConnectors()) {
234 						if (ConnectorUtil.connectsPart(connector, partConnector)) {
235 							// the connector end targets a port of a part or the composite (in case of delegation)
236 							ConnectorEnd connEnd = ConnectorUtil.connEndNotPart(connector, partConnector);
237 							actual = matchViaEnd(connEnd, port, partOrConnector, actual);
238 							if (actual != null) {
239 								found = true;
240 								break;
241 							}
242 						}
243 					}
244 				}
245 				else {
246 					for (ConnectorEnd connEnd : ((Connector) partOrConnector).getEnds()) {
247 						actual = matchViaEnd(connEnd, port, partOrConnector, actual);
248 						if (actual != null) {
249 							found = true;
250 							break;
251 						}
252 					}
253 				}
254 				if (!found) {
255 					String errorMsg = createErrorMsg(port, partOrConnector);
256 					throw new TransformationException(errorMsg);
257 				}
258 			}
259 		}
260 		return actual;
261 	}

263 	protected static String createErrorMsg(Port port, Feature partOrConnector) {
264 		String errorMsg = String.format(
265 				Messages.ConnectorBinding_CannotFindBindingForPort,
266 				port.getName(), partOrConnector.getName());
267 		for (Classifier cl : partOrConnector.getFeaturingClassifiers()) {
268 			errorMsg += " of class '" + cl.getName() + "'"; //$NON-NLS-1$ //$NON-NLS-2$
269 		}
270 		errorMsg += ". "; //$NON-NLS-1$
271 		return errorMsg;
272 	}

274 	private static Type matchViaEnd(ConnectorEnd connEnd, Port port, Feature partOrConnector, Type actual) throws
275 			TransformationException {
276 		String errorMsg = createErrorMsg(port, partOrConnector);
277 
278 		ConnectableElement connElem = connEnd.getRole();
279 		if (connElem instanceof Port) {
280 			Port otherPort = (Port) connElem;
281 
282 			Activator.log.info(String.format(
283 					Messages.ConnectorBinding_InfoConnPortConnectedVia,
284 					port.getName(), partOrConnector.getName()));
285 			Type otherType = PortUtils.getFCMType(otherPort);
286 
287 
288 			// type must be the same, kind as well. But depending on delegation or not, one of the ports
289 			// must be conjugated
290 			// TODO: isAssembly check will wrongly return false, if no port is connected [well, in this case, the role is not a port but a part]
291 			boolean isAssembly = (connEnd.getPartWithPort() != null);
292 			boolean sameConjugation = (otherPort.isConjugated() == port.isConjugated());
293 			// if assembly connection, conjugations must be different.
294 			if (isAssembly == sameConjugation) {
295 				return actual;
296 			}
297 
298 			// now compare kinds
299 			if (!PortUtils.sameKinds(port, otherPort)) {
300 				// no match, try next end
301 				return actual; // was: continue
302 			}
303 
304 			if (otherType != null) {
305 				if (actual == null) {
306 					actual = otherType;
307 					Activator.log.info(String.format(
308 							Messages.ConnectorBinding_InfoActualReturnIntfIs, actual.getQualifiedName()));
309 				} else if (actual != otherType) {
310 					throw new TransformationException(errorMsg + " " + String.format( //$NON-NLS-1$
311 							Messages.ConnectorBinding_FormalAlreadyBound, actual.getName()));
312 				}
313 			}
314 		}
315 		return actual;
316 	}

318 	/**
319 	 * The objective of this function is to find a suitable binding for a part
320 	 * typed with a template class (connector) based compatible port types. The binding
321 	 * is obtained by examining the connections between the composite and the contained
322 	 * part.
323 	 *
324 	 * @param composite
325 	 *            the composite which owns the part
326 	 * @param containedPart
327 	 *            the part which represents the connector
328 	 * @param template
329 	 *            an element within a package template (might be deeply nested) which
330 	 *            has ports that depend on a template parameter. This method tries to find a binding
331 	 *            for the template parameters that renders the ports of this element compatible with
332 	 *            the ports of the composition context.
333 	 * @return the bound package
334 	 *         ? binding information, i.e. information about the bound package, whether already bound and the binding.
335 	 */
336 	public static TemplateBinding obtainBinding(Class composite,
337 			Feature connector, Class template, boolean createBinding) throws TransformationException {
338 		// obtain the signature of an element within a package template.
339 
340 		TemplateSignature signature = TemplateUtils.getSignature(template);
341 		if (signature == null) {
342 			// not a template. This is not an exception, caller need to handle "null" a return
343 			// value indicating that no template instantiation needs to be done.
344 			Activator.log.info(String.format(
345 					Messages.ConnectorBinding_NoTemplateSignature, template.getName()));
346 			return null;
347 		}
348 
349 		// list of actuals (must be ParameterableElements, in our case always a named element as well.
350 		EList<ParameterableElement> actuals = new BasicEList<ParameterableElement>();
351 
352 		Package pkgTemplate = (Package) signature.getOwner();
353 		String name = pkgTemplate.getName();
354 
355 		/*
356 		 * Signature extension - useful?? (since binding has to be made as well)
357 		 * Stereotype sigExtSt = signature.getAppliedStereotype ("FCM::SignatureExt");
358 		 * SignatureExt sigExt = null;
359 		 * if (sigExtSt != null) {
360 		 * sigExt = (SignatureExt) signature.getStereotypeApplication (sigExtSt);
361 		 * }
362 		 *
363 		 * for (TemplateSignature sig : sigExt.getExtension ()) {
364 		 * bindFormal ();
365 		 * }
366 		 */
367 
368 		// loop on template parameters
369 		boolean firstTP = true;
370 		for (TemplateParameter parameter : signature.getParameters()) {
371 			ParameterableElement formal = parameter.getParameteredElement();
372 
373 			// now obtain suitable binding for this parameter - look for ports that are typed with
374 			// the formal template parameter
375 			ParameterableElement actual = null;
376 			if (firstTP) {
377 				// only try to match the first template parameter based on ports.
378 				actual = getActual(connector, template, formal);
379 				firstTP = false;
380 			} else {
381 				// additional template parameters that are bound automatically based on their name
382 				// (this might be considered as a hack, it is merely a convention for automatic template
383 				// binding within the SW Designer context.
384 				if (formal instanceof NamedElement) {
385 					NamedElement formalNE = (NamedElement) formal;
386 					if ((formal instanceof Class) && (formalNE.getName().equals("T"))) { //$NON-NLS-1$
387 						actual = composite;
388 					}
389 					if ((formal instanceof Port) && (formalNE.getName().equals("P"))) { //$NON-NLS-1$
390 						// port (but not available?)
391 						// actual = TransformationContext.current.port;
392 						// TODO
393 						actual = null;
394 					}
395 				}
396 			}
397 			if (actual instanceof NamedElement) {
398 				actuals.add(actual);
399 				name = name + "_" + ((NamedElement) actual).getName(); //$NON-NLS-1$
400 			} else {
401 				String reason = String.format(
402 						Messages.ConnectorBinding_CannotFindBinding,
403 						((NamedElement) formal).getName(), composite.getName(), connector.getName(), template.getName());
404 				throw new TransformationException(reason);
405 			}
406 		}
407 
408 		// the bound package is instantiated in the same model, in which the
409 		// composite can be found (avoid modifying an imported model).
410 
411 		if (!createBinding) {
412 			return null;
413 		}
414 
415 		Namespace owner = (Namespace) pkgTemplate.getOwner();
416 		owner = CreationUtils.getAndCreate(composite.getModel(), owner.allNamespaces());
417 
418 		Package boundPackage = (Package) owner.getMember(name);
419 		if (boundPackage == null) {
420 			// class does not exist yet, needs to be created.
421 			boundPackage = ((Package) owner).createNestedPackage(name);
422 
423 			Activator.log.info(String.format(
424 					Messages.ConnectorBinding_InfoCreateBoundPackage, name, owner.getName()));
425 		}
426 
427 		TemplateBinding binding = boundPackage.getTemplateBinding(signature);
428 		if (binding == null) {
429 			// binding is not existing yet (should normally only happen, if the class has
430 			// just been created - but it's better to re-check, even if the bound package
431 			// was already there)
432 			binding = boundPackage.createTemplateBinding(signature);
433 
434 			Iterator<ParameterableElement> actualsIter = actuals.iterator();
435 
436 			// loop on template parameters;
437 			for (TemplateParameter parameter : signature.getParameters()) {
438 				TemplateParameterSubstitution substitution = binding.createParameterSubstitution();
439 				substitution.setFormal(parameter);
440 
441 				// now obtain suitable binding for this parameter - look for ports that are typed with
442 				// the formal template parameter
443 				ParameterableElement actual = actualsIter.next();
444 				// for UML <= 2.1
445 				// substitution.getActuals ().add ((ParameterableElement) actual);
446 				// for UML > 2.2
447 				substitution.setActual(actual);
448 			}
449 		}
450 		return binding;
451 	}
