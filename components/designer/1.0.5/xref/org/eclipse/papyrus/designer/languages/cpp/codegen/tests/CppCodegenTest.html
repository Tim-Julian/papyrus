218 	private void assertGenerate(String fragment) throws Exception {
219 		selectSemanticElement(fragment);
220 		handlerService.executeCommand(GENERATE_COMMAND_ID, null);
221 	}
222 
223 	private void selectSemanticElement(String uriFragment) throws Exception {
224 		URI elementUri = modelUri.appendFragment(uriFragment);
225 		EObject semantic = model.eResource().getResourceSet().getEObject(elementUri, true);
226 
227 		// #openSemanticElement returns the multiEditor if successful and null otherwise
228 
229 		// the open often fails if pages are passed in, so we first try to open without specifying
230 		// pages
231 		IMultiDiagramEditor editor = elementActivator.openSemanticElement(semantic);
232 		assertNotNull(editor);
233 
234 		try {
235 			// wait (max 1 second) until editor becomes available
236 			int i = 0;
237 			while (editor.getActiveEditor() == null) {
238 				editor = elementActivator.openSemanticElement(semantic);
239 				Thread.sleep(10);
240 				if (i++ > 100) {
241 					fail("Timeout during wait for editor to become active");
242 				}
243 			}
244 		} catch (InterruptedException e) {
245 		}
246 
247 		// make sure there is an active multiEditor so that the selection will be available
248 		assertNotNull(editor.getActiveEditor());
249 	}

251 	/**
252 	 * Compare the files in folder with what we expect to see. When comparing
253 	 * file content, filter out whitespace which will replace all whitespace
254 	 * with a single space in the actual file content and the expected file
255 	 * content in order to avoid problems with differences caused by code
256 	 * formatting options where the test suite is run. Then a simple string
257 	 * comparison is done.
258 	 */
259 	private void assertGeneratedMatchesExpected(String fileName, String... depthSegments) throws Exception {
260 		IFolder srcGenFolder = FileComparison.getGeneratedProject(GenProjectName).getFolder(TestConstants.SRC_GEN);
261 		assertTrue("src-gen folder \"" + srcGenFolder + "\" does not exist", srcGenFolder.exists());
262 		IFolder generatedFolder = srcGenFolder.getFolder(GenFolderName);
263 		FileComparison.assertGeneratedMatchesExpected(generatedFolder, modelProject, fileName, depthSegments);
264 	}
