65  	/**
66  	 * Returns the required types of a specialization class of a template class
67  	 */
68  	public static List<Type> getTemplateRequiredTypes(Type type) {
69  		List<Type> types = new ArrayList<Type>();
70  		
71  		if (type instanceof Classifier && !((Classifier) type).getTemplateBindings().isEmpty()) { // TODO template stereotype
72  			TemplateBinding templateBinding = ((Classifier) type).getTemplateBindings().get(0);
73  			TemplateSignature signature = templateBinding.getSignature();
74  			
75  			if (signature != null && signature.getTemplate() instanceof Classifier) {
76  				types.add((Classifier) signature.getTemplate());
77  				
78  				List<TemplateParameter> templateParameters = signature.getOwnedParameters();
79  				Map<TemplateParameter, NamedElement> parameterToClassMap = new HashMap<TemplateParameter, NamedElement>();
80  				
81  				List<TemplateParameterSubstitution> substitutions = templateBinding.getParameterSubstitutions();
82  				for (TemplateParameterSubstitution substitution : substitutions) {
83  					if (substitution.getFormal() != null
84  							&& templateParameters.contains(substitution.getFormal())
85  							&& substitution.getActual() instanceof NamedElement) {
86  						parameterToClassMap.put(substitution.getFormal(), (NamedElement) substitution.getActual());
87  					}
88  				}
89  				
90  				if (parameterToClassMap.size() == templateParameters.size()) {
91  					for (TemplateParameter templateParameter : templateParameters) {
92  						NamedElement substitutionType = parameterToClassMap.get(templateParameter);
93  						if (substitutionType instanceof Type) {
94  							types.add((Type) substitutionType);
95  						}
96  					}
97  				}
98  			}
99  			
100 			return types;
101 		}
102 		
103 		types.add(type);
104 		return types;
105 	}

133 		if (tbs.size() > 0) {
134 			for (TemplateBinding tb : tbs) {

243 	/**
244 	 * Retrieve a list of types of attributes that belong to the current classifier. Filter by stereotypes.
245 	 *
246 	 * @param current
247 	 *            Class on which the attributes are searched
248 	 * @param excludedStereotypes
249 	 *            List of ALL stereotypes that must no be applied
250 	 * @param includedStereotypes
251 	 *            List of ANY stereotype that must no be applied (at least one)
252 	 * @param bypassForInnerClassifiers
253 	 *            Always include inner classifier types or not
254 	 * @param noSharedAggregation
255 	 *            Always exclude attributes with a shared aggregation kind
256 	 * @return Collection of classifiers which are the types of the attributes
257 	 */
258 	public static EList<Classifier> getTypesViaAttributes(Classifier current, EList<Class<? extends EObject>> excludedStereotypes, EList<Class<? extends EObject>> includedStereotypes, boolean bypassForInnerClassifiers, boolean noSharedAggregation) {
259 		EList<Classifier> result = new UniqueEList<Classifier>();
260 
261 		Iterator<Property> attributes;
262 		attributes = current.getAttributes().iterator();
263 		while (attributes.hasNext()) {
264 			Property currentAttribute = attributes.next();
265 			Type type = currentAttribute.getType();
266 			
267 			if  (type != null) {
268 				// Check type is inner classifier
269 				if (bypassForInnerClassifiers && !(type.getOwner() instanceof Package)) { // if we force inner class types to be processed
270 					addFarthestOwnerType(type, result);
271 				} else {
272 					// Check other conditions
273 					boolean ok = true;
274 					
275 					// Check attribute is shared aggregation
276 					if (noSharedAggregation && currentAttribute.getAggregation() == AggregationKind.SHARED_LITERAL) {
277 						ok = false;
278 					}
279 					// Check attribute does not have an excluded stereotype
280 					if (ok) {
281 						if (excludedStereotypes != null && GenUtils.hasAnyStereotype(currentAttribute, excludedStereotypes)) {
282 							ok = false;
283 						}
284 					}
285 					// Check attribute has an included stereotype
286 					if (ok) {
287 						if (includedStereotypes != null && !GenUtils.hasAnyStereotype(currentAttribute, includedStereotypes)) {
288 							ok = false;
289 						}
290 					}
291 					
292 					// All ok
293 					if (ok) {
294 						addFarthestOwnerType(type, result);
295 					}
296 				}
297 			}
298 		}
299 		
300 		return result;
301 	}

390 	 * Retrieve the operations in the current classifier. For each
391 	 * operation, collect types of its parameters.
392 	 * This method thus finds types, on
393 	 * which the signature depends.
394 	 * We check if operations and parameters must have or not have some stereotypes.
395 	 *
396 	 * @param current
397 	 *            Classifier on which the operations are searched for
398 	 * @param excludedOperationStereotypes
399 	 *            List of ALL stereotypes that must not be applied to an operation
400 	 * @param includedOperationStereotypes
401 	 *            List of ANY stereotype that must be applied to an operation (at least one)
402 	 * @param excludedParameterStereotypes
403 	 *            List of ALL stereotypes that must not be applied to a parameter
404 	 * @param includedParameterStereotypes
405 	 *            List of ANY stereotype that must be applied to a parameter (at least one)
406 	 * @param bypassForInnerClassifiers
407 	 *            Always include types that are inner classifiers
408 	 * @return Collection of classifiers which are the types of the operation parameters
409 	 */
410 	public static EList<Classifier> getTypesViaOperations(Classifier current,
411 			EList<Class<? extends EObject>> excludedOperationStereotypes,
412 			EList<Class<? extends EObject>> includedOperationStereotypes,
413 			EList<Class<? extends EObject>> excludedParameterStereotypes,
414 			EList<Class<? extends EObject>> includedParameterStereotypes,
415 			boolean bypassForInnerClassifiers) {
416 		EList<Classifier> result = new UniqueEList<Classifier>();
417 		
418 		for (Operation operation : current.getOperations()) {
419 			boolean ok = true;
420 			
421 			if (excludedOperationStereotypes != null && GenUtils.hasAnyStereotype(operation, excludedOperationStereotypes)) {
422 				ok = false;
423 			}
424 			
425 			if (includedOperationStereotypes != null && !GenUtils.hasAnyStereotype(operation, includedOperationStereotypes)) {
426 				ok = false;
427 			}
428 			
429 			boolean operationOk = ok;
430 			if (operationOk || bypassForInnerClassifiers) {
431 				for (Parameter param : operation.getOwnedParameters()) {
432 					Type type = param.getType();
433 					if  (type != null) {
434 						if (bypassForInnerClassifiers && !(type.getOwner() instanceof Package)) { // if we force inner class types to be processed
435 							addFarthestOwnerType(type, result);
436 						} else if (operationOk) {
437 							ok = true;
438 							
439 							if (excludedParameterStereotypes != null && GenUtils.hasAnyStereotype(param, excludedParameterStereotypes)) {
440 								ok = false;
441 							}
442 							
443 							if (includedParameterStereotypes != null && !GenUtils.hasAnyStereotype(param, includedParameterStereotypes)) {
444 								ok = false;
445 							}
446 							
447 							if (ok) {
448 								addFarthestOwnerType(type, result);
449 							}
450 						}
451 					}
452 				}
453 			}
454 		}
455 		
456 		return result;
457 	}

1418 	 *         one of the operation's methods
1419 	 */
1420 	public static String getBodyFromOB(OpaqueBehavior ob, Pattern selectedLanguages) {
1421 		Iterator<String> bodies = ob.getBodies().iterator();
1422 		for (String language : ob.getLanguages()) {
1423 			// additional sanity check: number of languages and number of bodies should be synchronized,
1424 			// but there is no guarantee that this is the case
1425 			if (bodies.hasNext()) {
1426 				String body = bodies.next();
1427 				Matcher matcher = selectedLanguages.matcher(language);
1428 				if (matcher.matches()) {
1429 					// additional "\r" confuses Acceleo
1430 					return cleanCR(body);
1431 				}
1432 			}
1433 		}
1434 		return ""; //$NON-NLS-1$
1435 	}
