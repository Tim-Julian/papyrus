16  
17  import org.eclipse.emf.ecore.EClass;
18  import org.eclipse.papyrus.designer.languages.java.reverse.exception.NotFoundException;
19  import org.eclipse.uml2.uml.Classifier;
20  import org.eclipse.uml2.uml.NamedElement;
21  import org.eclipse.uml2.uml.Namespace;
22  import org.eclipse.uml2.uml.Package;
23  
24  /**
25   * A context local to the currently processed {@link NamedElement}. This NamedElement is also a {@link Namespace}.
26   * 
27   * 
28   * @author cedric dumoulin
29   *
30   */
31  public class LocalContext {
32  
33  	protected Namespace namespace;
34  	protected LocalContext parentContext;
35  	
36  	/**
37  	 * Constructor.
38  	 * @param contextNamespace The namespace associated to this current context.
39  	 * 
40  	 *
41  	 */
42  	public LocalContext(Namespace contextNamespace) {
43  		this.namespace = contextNamespace;
44  		
45  	}
46  
47  	/**
48  	 * Constructor.
49  	 * @param contextNamespace The namespace associated to this current context.
50  	 * 
51  	 *
52  	 */
53  	public LocalContext(Namespace contextNamespace, LocalContext parentContext) {
54  		this.namespace = contextNamespace;
55  		this.parentContext = parentContext;
56  	}
57  
58  	/**
59  	 * @return the namespace
60  	 */
61  	public Namespace getNamespace() {
62  		return namespace;
63  	}
64  
65  	/**
66  	 * Lookup for a Classifier with the specified shortname in the contexts.
67  	 * First, check this context, and if not found, recursively check parent contexts.
68  	 * 
69  	 * @param shortname
70  	 * @param classType
71  	 * @return the classifier, or null if not found.
72  	 */
73  	public Classifier lookupClassifier(String shortname, EClass classType) {
74  		
75  		Classifier result;
76  		
77  		result = (Classifier)namespace.getOwnedMember(shortname, false, classType);
78  		if( result != null) {
79  			return result;
80  		}
81  		// Check in parent
82  		if( parentContext != null ) {
83  			return parentContext.lookupClassifier(shortname, classType);
84  		}
85  		// Not found
86  		return null;
87  	}
88  
89  	/**
90  	 * Get the Current package of the contexts.
91  	 * The current package is the first {@link Package} encountered in contexts' namespaces. 
92  	 * 
93  	 * @return The current package, or null if not found (should not happen).
94  	 */
95  	public Package getCurrentPackage()  {
96  		
97  		if( parentContext != null ) {
98  			return parentContext.getCurrentPackage();
99  		}
100 		// There is no parent, so we are the outest most context. The namespace should be a Package
101 		if( namespace instanceof Package ) {
102 			return (Package)namespace;
103 		}
104 		
105 		// Not found !! Should not happen.
106 //		throw new NotFoundException("No package found in Contexts.");
107 		return null;
108 	}
109 
110 }
