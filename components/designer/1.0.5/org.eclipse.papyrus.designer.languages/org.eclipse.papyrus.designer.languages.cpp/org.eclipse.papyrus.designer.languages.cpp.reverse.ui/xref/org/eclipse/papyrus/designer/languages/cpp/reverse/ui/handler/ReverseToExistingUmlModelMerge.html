44  /**
45   * Handler for the "Reverse to existing model (merge)" command
46   *
47   */
48  public class ReverseToExistingUmlModelMerge extends AbstractHandler {
49  	
50  	@Override
51  	public Object execute(ExecutionEvent event) throws ExecutionException {
52  		try {
53  			ISelection selection = HandlerUtil.getCurrentSelection(event);
54  
55  			if (selection instanceof ITextSelection) {
56  				IEditorPart editorPart = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getActivePage().getActiveEditor();
57  
58  				if (editorPart instanceof AbstractTextEditor) {
59  					ITextEditor textEditor = (ITextEditor) editorPart;
60  
61  					if (textEditor != null) {
62  						IFile resource = (IFile) textEditor.getEditorInput().getAdapter(IResource.class);
63  
64  						if (resource != null) {
65  							scheduleReverse(resource.getProject());
66  							return Status.OK_STATUS;
67  						}
68  					}
69  				}
70  			}
71  
72  			if (selection instanceof IStructuredSelection) {
73  				Object selectedObject = ((IStructuredSelection) selection).getFirstElement();
74  				if (selectedObject instanceof IResource) {
75  					scheduleReverse(((IResource) selectedObject).getProject());
76  					return Status.OK_STATUS;
77  				}
78  			}
79  
80  
81  		} catch (Exception e) {
82  			Activator.log.error(e);
83  		}
84  
85  		return null;
86  	}
87  	
88  	private void scheduleReverse(final IProject project) {
89  		if (project != null) {
90  			final ICProject cProject = CoreModel.getDefault().create(project);
91  			Job job = new Job("Reversing project " + cProject.getElementName() + " to existing UML model (merge)") {
92  
93  				@Override
94  				protected IStatus run(final IProgressMonitor monitor) {
95  					try {
96  						Display.getDefault().asyncExec(new Runnable() {
97  							@Override
98  							public void run() {
99  								try {
100 									String[] types = {"uml"};
101 									ResourceFileSelectionDialog fileSelector = new ResourceFileSelectionDialog("UML file selector", "Select a UML file", types); // $NON-NLS-1$
102 									fileSelector.open();
103 									Object file = fileSelector.getFirstResult();
104 									
105 									if (file instanceof IFile && ((IFile) file).exists()) {
106 										String path = ((IFile) file).getFullPath().toString();
107 										
108 										ReverseCpp2Uml reverser = new ReverseCpp2Uml(cProject, monitor, ReverseCpp2Uml.Cpp_LangID, path);
109 										reverser.syncIncrementalProject();
110 
111 										MessageBox messageBox = new MessageBox(Display.getDefault().getActiveShell(), SWT.ICON_INFORMATION | SWT.OK);
112 										messageBox.setMessage("Merged changes to existing UML model " + ((IFile) file).getLocationURI().getPath()); // $NON-NLS-1$
113 										messageBox.open();
114 									}
115 								} catch (Exception e) {
116 									Activator.log.error(e);
117 								}
118 								
119 							}
120 						});
121 					} catch (Exception e) {
122 						Activator.log.error(e);
123 					}
124 
125 					return Status.OK_STATUS;
126 				}
127 
128 			};
129 
130 			job.setUser(true);
131 			job.schedule();
132 		}
133 	}
134 }
