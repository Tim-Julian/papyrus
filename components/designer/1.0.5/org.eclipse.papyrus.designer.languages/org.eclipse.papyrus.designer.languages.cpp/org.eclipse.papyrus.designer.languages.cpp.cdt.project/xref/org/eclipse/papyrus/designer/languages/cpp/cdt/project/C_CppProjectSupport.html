59  	/**
60  	 * Create a C++ project. Caller should test before calling, whether the
61  	 * project exists already
62  	 *
63  	 * @param projectName
64  	 * @return the created project
65  	 */
66  	@Override
67  	public IProject createProject(String projectName) {
68  		IWorkspaceRoot root = ResourcesPlugin.getWorkspace().getRoot();
69  
70  		IProject project = root.getProject(projectName);
71  		dialogStatus = 0;
72  		try {
73  			IWorkbench wb = PlatformUI.getWorkbench();
74  
75  			// create CDT wizard for C++ or C
76  			final CDTCommonProjectWizard wiz = this instanceof CppProjectSupport ? new CCNamedProjectWizard(projectName)
77  					: new CNamedProjectWizard(projectName);
78  
79  			wiz.setWindowTitle("create project " + projectName); //$NON-NLS-1$
80  			wiz.init(wb, null);
81  
82  			Display.getDefault().syncExec(new Runnable() {
83  
84  				@Override
85  				public void run() {
86  					WizardDialog wizDiag = new WizardDialog(Display.getCurrent().getActiveShell(), wiz);
87  
88  					wizDiag.create();
89  					dialogStatus = wizDiag.open();
90  				}
91  			});
92  			if (dialogStatus == Window.OK) {
93  				// update project (name might have changed by user)
94  				project = wiz.getLastProject();
95  			} else if (dialogStatus == Window.CANCEL) {
96  				return null;
97  			}
98  		} catch (Exception e) {
99  			e.printStackTrace();
100 			project = null;
101 		}
102 
103 		if ((project == null) || !project.exists()) {
104 			throw new RuntimeException(
105 					"Could not create CDT project. This might indicate that there is a problem with your CDT installation."); //$NON-NLS-1$
106 		}
107 		return project;
108 	}

110 	@Override
111 	public void setSettings(IProject project, AbstractSettings abstractSettings) {
112 		CDTSettings settings = (CDTSettings) abstractSettings;
113 		try {
114 			// ((CProject) project).
115 			// IProjectDescription desc = m_project.getDescription();
116 
117 			ICProjectDescriptionManager mngr = CoreModel.getDefault().getProjectDescriptionManager();
118 			ICProjectDescription cdesc = mngr.getProjectDescription(project, true);
119 
120 			// loop over all configurations
121 			for (ICConfigurationDescription configDescr : cdesc.getConfigurations()) {
122 
123 				ICFolderDescription folderDescription = configDescr.getRootFolderDescription();
124 
125 				ICLanguageSetting[] languageSettings = folderDescription.getLanguageSettings();
126 
127 				// copy string array into ICLanguageSetting array
128 				ICLanguageSettingEntry[] icIncludePaths = new ICLanguageSettingEntry[settings.includePaths.size()];
129 				for (int i = 0; i < settings.includePaths.size(); i++) {
130 					String path = settings.includePaths.get(i);
131 					if (isWsRelPath(path)) {
132 						icIncludePaths[i] = new CIncludePathEntry(removeWsPrefix(path),
133 								ICSettingEntry.VALUE_WORKSPACE_PATH);
134 					} else {
135 						icIncludePaths[i] = new CIncludePathEntry(path, 0);
136 					}
137 				}
138 
139 				// define name of used operating system from model (attribute of
140 				// "Target" stereotype)
141 				// and add it to list of macros
142 				if (settings.targetOS != null) {
143 					settings.macros.add("OS_" + settings.targetOS); //$NON-NLS-1$
144 				}
145 
146 				// define macros
147 				EList<ICLanguageSettingEntry> icMacros = new BasicEList<ICLanguageSettingEntry>();
148 				for (int i = 0; i < settings.macros.size(); i++) {
149 					// TODO: need to define values for macros as well?
150 					icMacros.add(new CMacroEntry(settings.macros.get(i), "", 0)); //$NON-NLS-1$
151 				}
152 
153 				// now set include path and preprocessor code
154 				for (ICLanguageSetting lang : languageSettings) {
155 					for (String ext : lang.getSourceExtensions()) {
156 						if (ext.equals(CPP) || ext.equals(C)) {
157 							lang.setSettingEntries(ICSettingEntry.INCLUDE_PATH, icIncludePaths);
158 							ICLanguageSettingEntry icOldMacros[] = lang.getSettingEntries(ICSettingEntry.MACRO);
159 							for (ICLanguageSettingEntry entry : icOldMacros) {
160 								icMacros.add(entry);
161 							}
162 							lang.setSettingEntries(ICSettingEntry.MACRO, icMacros);
163 							break;
164 						}
165 					}
166 				}
167 				IConfiguration main = ManagedBuildManager.getConfigurationForDescription(configDescr);
168 
169 				// add to -l (libraries)
170 				ITool cfTool = main.calculateTargetTool();
171 
172 				if (cfTool != null) {
173 					for (IOption opt : cfTool.getOptions()) {
174 						if (opt.getValueType() == IOption.LIBRARIES) {
175 							main.setOption(cfTool, opt, settings.libs.toArray(new String[0]));
176 						} else if (opt.getValueType() == IOption.LIBRARY_PATHS) {
177 							main.setOption(cfTool, opt, settings.libPaths.toArray(new String[0]));
178 						}
179 					}
180 					mngr.setProjectDescription(project, cdesc, true, null);
181 				}
182 				else {
183 					Activator.log.warn("could not write project settings. Please check whether a C++ toolchain (compiler) is installed");
184 				}
185 			}
186 			ManagedBuildManager.saveBuildInfo(project, true);
187 		} catch (BuildException be) {
188 			throw new RuntimeException(be.getMessage());
189 		} catch (CoreException ce) {
190 			throw new RuntimeException(ce.getMessage());
191 		}
192 	}

194 	@Override
195 	public AbstractSettings initialConfigurationData() {
196 		CDTSettings settings = new CDTSettings();
197 		settings.includePaths = new UniqueEList<String>();
198 		// include project directory (all paths are relative to it)
199 		// TODO: choose source folder depending on model (which is currently not passed as a parameter)
200 		settings.includePaths.add(WS_PREFIX + "src-gen"); //$NON-NLS-1$
201 
202 		settings.libs = new UniqueEList<String>();
203 		settings.libPaths = new UniqueEList<String>();
204 		settings.macros = new UniqueEList<String>();
205 		return settings;
206 	}

208 	@Override
209 	public void gatherConfigData(Classifier implementation, AbstractSettings abstractSettings) {
210 		CDTSettings settings = (CDTSettings) abstractSettings;
211 		Element owner = implementation.getNearestPackage();
212 		while (owner instanceof Package) {
213 			ExternLibrary cppLibrary = UMLUtil.getStereotypeApplication(owner, ExternLibrary.class);
214 			if ((cppLibrary != null) && (settings != null)) {
215 				settings.includePaths.addAll(cppLibrary.getIncludePaths());
216 				for (String libPath : cppLibrary.getLibPaths()) {
217 					if (isWsRelPath(libPath)) {
218 						// libPaths starting with a slash (separator char) are
219 						// relative to workspace location
220 						// TODO: need to support absolute paths (host file
221 						// system?) as well?
222 						// (additional prefix. Eclipse standards?) Problem:
223 						// workspace_loc is added
224 						// automatically for absolute includePaths
225 						settings.libPaths.add("${workspace_loc:" + removeWsPrefix(libPath) + "}");
226 					} else {
227 						// relative to project root, otherwise
228 						settings.libPaths.add(libPath);
229 					}
230 				}
231 				settings.libs.addAll(cppLibrary.getLibs());
232 				settings.macros.addAll(cppLibrary.getMacros());
233 			}
234 			owner = owner.getOwner();
235 		}
236 	}

238 	public static String removeWsPrefix(String path) {
239 		return path.substring(WS_PREFIX.length());
240 	}
241 
242 	/**
243 	 * @return true, if the path is a workspace relative path
244 	 */
245 	public static boolean isWsRelPath(String path) {
246 		return path.startsWith(WS_PREFIX);
247 
248 	}

251 	public IProject createProject(String projectName, Package modelRoot) {
252 		return createProject(projectName);
