24  @SuppressWarnings("all")
25  public class CppPrimitiveTypes {
26    /**
27     * Support two different kinds of primitive types
28     * (1) those that are native types of the programming language such as long
29     *     For these, no additional definition has to be done and they should be referenced
30     *     with their name only
31     * (2) those that correspond to a typedef (e.g. typedef long ErrorType). These require
32     *     a typedef definition within the package and need to be referenced with their
33     *     fully qualified name (e.g. MyPackage::ErrorType)
34     * (3) Sometimes the name of the type defined by a typedef does not appear at the end, e.g.
35     *     when defining functionPointers such as "typedef int (*HandlerFct) (void*);"
36     * 	   (a function with a void* argument returning int)
37     *     Support this case via a specific 'typeName' constant that gets replaced by the name of the
38     *     type
39     */
40    public static CharSequence CppPrimitiveTypeDefinition(final PrimitiveType primitiveType) {
41      StringConcatenation _builder = new StringConcatenation();
42      CharSequence _CppElementDoc = CppDocumentation.CppElementDoc(primitiveType);
43      _builder.append(_CppElementDoc, "");
44      _builder.newLineIfNotEmpty();
45      {
46        boolean _hasStereotype = GenUtils.hasStereotype(primitiveType, Typedef.class);
47        if (_hasStereotype) {
48          _builder.append("typedef ");
49          Typedef _stereotypeApplication = UMLUtil.<Typedef>getStereotypeApplication(primitiveType, Typedef.class);
50          String _definition = _stereotypeApplication.getDefinition();
51          String _name = primitiveType.getName();
52          String _replaceAll = _definition.replaceAll("typeName", _name);
53          _builder.append(_replaceAll, "");
54          _builder.newLineIfNotEmpty();
55          _builder.append("\t");
56          {
57            Typedef _stereotypeApplication_1 = UMLUtil.<Typedef>getStereotypeApplication(primitiveType, Typedef.class);
58            String _definition_1 = _stereotypeApplication_1.getDefinition();
59            boolean _contains = _definition_1.contains("typeName");
60            boolean _not = (!_contains);
61            if (_not) {
62              String _name_1 = primitiveType.getName();
63              _builder.append(_name_1, "\t");
64            }
65          }
66          _builder.append(";");
67          _builder.newLineIfNotEmpty();
68          _builder.append("\t");
69          _builder.newLine();
70        } else {
71          String _stdtypes = CppGenUtils.getStdtypes(primitiveType);
72          _builder.append(_stdtypes, "");
73          _builder.newLineIfNotEmpty();
74          _builder.append(" ");
75          _builder.newLine();
76        }
77      }
78      return _builder;
79    }
