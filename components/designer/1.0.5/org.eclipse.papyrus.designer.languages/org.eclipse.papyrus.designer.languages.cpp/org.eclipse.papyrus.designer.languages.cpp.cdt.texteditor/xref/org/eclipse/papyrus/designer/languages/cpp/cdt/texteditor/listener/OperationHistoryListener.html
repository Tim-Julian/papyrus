32  
33  public class OperationHistoryListener implements IOperationHistoryListener {
34  
35  	public static void init() {
36  		if (ohl == null) {
37  			ohl = new OperationHistoryListener();
38  			OperationHistoryFactory.getOperationHistory().addOperationHistoryListener(ohl);
39  		}
40  	}
41  
42  	private static OperationHistoryListener ohl = null;
43  
44  	@Override
45  	public void historyNotification(OperationHistoryEvent event) {
46  		int eventType = event.getEventType();
47  		if (eventType == OperationHistoryEvent.OPERATION_ADDED || eventType == OperationHistoryEvent.REDONE
48  				|| eventType == OperationHistoryEvent.UNDONE) {
49  
50  			EList<Classifier> regenListCopy = new BasicEList<Classifier>(ModelListener.regenList);
51  			ModelListener.regenList.clear();
52  
53  			// re-generate files from list - if they are still in a resource
54  			for (Classifier cl : regenListCopy) {
55  				if (cl.eResource() != null) {
56  					try {
57  						ServicesRegistry serviceRegistry = ServiceUtilsForEObject.getInstance().getServiceRegistry(cl);
58  						TextEditorModelSharedResource model = (TextEditorModelSharedResource) ServiceUtils.getInstance()
59  								.getModelSet(serviceRegistry).getModelChecked(TextEditorModelSharedResource.MODEL_ID);
60  
61  						if (model != null) {
62  							TextEditorModel tem = model.getTextEditorModel(cl);
63  							if (tem != null) {
64  								SyncModelToCDT.syncModelToCDT(cl, tem.getGeneratorID());
65  							}
66  						}
67  					} catch (ServiceException e) {
68  						Activator.log.error(e);
69  					} catch (NotFoundException e) {
70  						Activator.log.error(e);
71  					}
72  				}
73  			}
74  		}
75  	}
76  }
