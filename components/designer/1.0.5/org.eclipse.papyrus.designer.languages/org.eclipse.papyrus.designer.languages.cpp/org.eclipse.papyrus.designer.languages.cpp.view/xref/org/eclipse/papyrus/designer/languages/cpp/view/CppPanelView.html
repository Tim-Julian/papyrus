24  /**
25   * This class extends <code>ViewPart</code>. This class draws the Accord/Cpp
26   * panel in the Modeling Perspective. It also implements <code>ISelectionListener</code> in order to know which element is selected
27   * in the workbench. In fact, the content of the view changes depending on the
28   * type of the element selected
29   * @deprecated This view will be replaced by an XWT view for the C++ profile
30   */
31  @Deprecated
32  public class CppPanelView extends AbstractCppPanelView {
33  
34  	/**
35  	 * Creates the new panel, when switching UI.
36  	 * <p>
37  	 * This class should be overloaded when working with RSA or other tools, using a new PanelFactory
38  	 *
39  	 * @param parent
40  	 *            the parent of the new panel
41  	 * @param style
42  	 *            the SWT style of this panel
43  	 * @param element
44  	 *            the element for which the panel is created
45  	 * @return the newly created panel
46  	 */
47  	@Override
48  	protected CppAbstractPanel createPanel(Composite parent, int style,
49  			Element element) {
50  		return org.eclipse.papyrus.designer.languages.cpp.view.panels.PanelFactory.eINSTANCE
51  				.createPanel(parent, 0, currentTarget);
52  	}
53  
54  	/*
55  	 * (non-Javadoc)
56  	 *
57  	 * @see org.eclipse.ui.ISelectionListener#selectionChanged(org.eclipse.ui.IWorkbenchPart,
58  	 * org.eclipse.jface.viewers.ISelection)
59  	 */
60  	@Override
61  	public void selectionChanged(IWorkbenchPart part, ISelection selection) {
62  
63  		IStructuredSelection sSelection = null;
64  		if (selection instanceof IStructuredSelection) {
65  			sSelection = (IStructuredSelection) selection;
66  		}
67  
68  		// exclude case of an empty selection which is not a Tree selection, since changing views provokes an
69  		// empty selection (selection gets lost, although same element remains selected)
70  		if ((selection != null) && (sSelection != null) && sSelection.isEmpty()) {
71  			return;
72  		}
73  
74  		// No available selection: switch to default panel
75  		if ((sSelection == null) || (sSelection.size() != 1)) {
76  			currentTarget = null;
77  			switchUI();
78  			return;
79  		}
80  
81  		// Retrieve selected object
82  		EObject currentObject = EMFHelper.getEObject(sSelection.getFirstElement());
83  
84  		if (currentObject instanceof Element) {
85  			if (currentTarget != currentObject) {
86  				currentTarget = currentObject;
87  				switchUI();
88  			}
89  		}
90  	}
91  }
