71  	public AllocationDialog(Shell parent, Package cdp) {
72  		super(parent);
73  		m_cdp = cdp;
74  		nodeOrThreadList = new BasicEList<InstanceSpecification>();
75  		nodeOrThreadList.add(null); // dummy entry for no allocation
76  		DepUtils.getAllInstances(cdp.getModel(), nodeOrThreadList, new ElementFilter() {
77  			
78  			@Override
79  			public boolean acceptElement(Element element) {
80  				if (element instanceof InstanceSpecification) {
81  					InstanceSpecification instance = (InstanceSpecification) element;
82  					if (instance.getName() == null) {
83  						// don't allocate to root element (detectable via the empty name)
84  						return false;
85  					}
86  					Classifier cl = DepUtils.getClassifier(instance);
87  					if (cl instanceof Class) {
88  						if (StereotypeUtil.isApplied(cl, SwSchedulableResource.class)) {
89  							// threads are valid allocation targets. Therefore, threads are always added to
90  							// list, even if within a deployment plan.
91  							return true;
92  						}
93  	                   if (StereotypeUtil.isApplied(instance.getNearestPackage(), DeploymentPlan.class)) {
94                              // instance is part of a deployment plan => don't add to list.
95                              return false;
96                          }
97                          return true;
98  					}
99  				}
100 				return false;
101 			}
102 		});
103 	}

113 	@Override

119 		fTree.setHeaderVisible(true);
120 		GridData data = new GridData(GridData.FILL_BOTH);
121 		fTree.setLayoutData(data);
122 		data.heightHint = 150;

124 
125 		// Turn off drawing to avoid flicker
126 		fTree.setRedraw(false);

128 		TreeColumn instances = new TreeColumn(fTree, SWT.LEFT);
129 		instances.setText("Instance");
130 		instances.setWidth(200);
131 		TreeColumn explicitAlloc = new TreeColumn(fTree, SWT.LEFT);
132 		explicitAlloc.setText("explicit allocation");
133 		explicitAlloc.setWidth(150);
134 		TreeColumn implicitAlloc = new TreeColumn(fTree, SWT.LEFT);
135 		implicitAlloc.setText("implicit allocation");
136 		implicitAlloc.setWidth(150);
137 
138 		for (InstanceSpecification is : DepUtils.getTopLevelInstances(m_cdp)) {
139 			fillTree(fTree, null, is);
140 		}
141 
142 		// Turn drawing back on!
143 		fTree.setRedraw(true);
144 
145 		fTree.addSelectionListener(new SelectionAdapter() {
146 
147 			@Override
148 			public void widgetSelected(SelectionEvent e) {
149 				if (fTree.getSelection().length > 0) {
150 					selectInstance(fTree.getSelection()[0]);
151 				}
152 			}
153 		});
154 
155 		Label label = new Label();
156 		label.setText("seletected instance:");
157 		fLabel = new Label();
158 
159 		createAllocInfo(contents);
160 		return contents;
161 	}

164 	 * Helper method to fill a tree with data
165 	 *
166 	 * @param tree
167 	 *            the tree to fill
168 	 */
169 	private void fillTree(Tree tree, TreeItem treeItem, InstanceSpecification is) {
170 		TreeItem item;
171 		if (treeItem != null) {
172 			item = new TreeItem(treeItem, SWT.NONE);
173 		} else {
174 			item = new TreeItem(tree, SWT.NONE);
175 		}
176 		item.setData(is);
177 		setTextFromData(item);
178 
179 		// create children
180 		for (InstanceSpecification subIS : DepUtils.getContainedNonSharedInstances(is)) {
181 			fillTree(tree, item, subIS);
182 		}
183 	}

185 	// obtain name and explicit/implicit node Allocation
186 	protected void setTextFromData(TreeItem ti) {
187 		Object data = ti.getData();
188 		if (data instanceof InstanceSpecification) {
189 			InstanceSpecification is = (InstanceSpecification) data;
190 
191 			String name = is.getName();
192 			int index = name.lastIndexOf("."); //$NON-NLS-1$
193 			if (index != -1) {
194 				name = name.substring(index + 1);
195 			}
196 			InstanceSpecification explicitNodeOrThread = AllocUtils.getNodeOrThread(is);
197 			Classifier cl = DepUtils.getClassifier(is);
198 			String nodeName;
199 			if (explicitNodeOrThread == null) {
200 				nodeName = "-"; //$NON-NLS-1$
201 			} else {
202 				nodeName = getAllocName(explicitNodeOrThread);
203 			}
204 			if (cl instanceof Class) {
205 				// if (BootLoaderGen.hasUnconnectedStartRoutine(null, (Class) cl, null)) {
206 				// TODO
207 					nodeName += " (main)"; //$NON-NLS-1$
208 				// }
209 			}
210 			EList<InstanceSpecification> implicitNodes = AllocUtils.getAllNodesOrThreadsParent(is);
211 			implicitNodes.addAll(AllocUtils.getAllNodesOrThreadsParent(is));
212 			String list = ""; //$NON-NLS-1$
213 			for (InstanceSpecification node : implicitNodes) {
214 				if (list.equals("")) { //$NON-NLS-1$
215 					list = getAllocName(node);
216 				} else {
217 					list += ", " + getAllocName(node); //$NON-NLS-1$
218 				}
219 			}
220 			ti.setText(new String[] { name, nodeName, "[" + list + "]" }); //$NON-NLS-1$//$NON-NLS-2$
221 		}
222 	}
223 
224 	protected void refreshTree(TreeItem ti) {
225 		setTextFromData(ti);
226 		for (TreeItem subItem : ti.getItems()) {
227 			refreshTree(subItem);
228 		}
229 	}

231 	private String getAllocName(InstanceSpecification nodeOrThread) {
232 		if (nodeOrThread == null) {
233 			return "no explicit allocation";
234 		} else {
235 			String name = nodeOrThread.getName();
236 			InstanceSpecification threadNode = AllocUtils.getNode(nodeOrThread);
237 			return name + (threadNode != null ? " on " + threadNode.getName() : "");
238 		}
239 	}

241 	protected void createAllocInfo(Composite parent) {
242 		// create extension kind combo
243 		fAlloc = DialogUtils.createComboWithText(parent, "Alloc to node:",
244 				SWT.DROP_DOWN | SWT.READ_ONLY, SWT.NONE);
245 
246 		String items[] = new String[nodeOrThreadList.size()];
247 		int i = 0;
248 		for (InstanceSpecification nodeOrThread : nodeOrThreadList) {
249 			items[i++] = getAllocName(nodeOrThread);
250 		}
251 		fAlloc.setItems(items);
252 		fAlloc.addSelectionListener(new SelectionAdapter() {
253 
254 			@Override
255 			public void widgetSelected(SelectionEvent event) {
256 				if (currentIS != null) {
257 					updateAllocation(currentIS, fAlloc.getSelectionIndex());
258 					// selectionCount should always be 1
259 					for (TreeItem ti : fTree.getSelection()) {
260 						refreshTree(ti);
261 					}
262 				}
263 			}
264 		});
265 
266 		fAlloc.setEnabled(false);
267 	}

269 	/**
270 	 * Select a rule, i.e. update the visual representation from the rule
271 	 *
272 	 * @param rule
273 	 */
274 	protected void selectInstance(TreeItem item) {
275 		// for (Port port : rule.getPortSet ()) {
276 		// fPorts.setSelection()
277 		// }
278 		currentIS = (InstanceSpecification) item.getData();
279 		fLabel.setText(currentIS.getName());
280 
281 		fAlloc.setEnabled(true);
282 		InstanceSpecification nodeOrThread = AllocUtils.getNodeOrThread(currentIS);
283 
284 		for (int i = 0; i < nodeOrThreadList.size(); i++) {
285 			if (nodeOrThreadList.get(i) == nodeOrThread) {
286 				fAlloc.select(i);
287 			}
288 		}
289 	}
290 
291 	private void updateAllocation(InstanceSpecification is, int index) {
292 		InstanceSpecification oldNode = AllocUtils.getNodeOrThread(is);
293 		InstanceSpecification newNode = nodeOrThreadList.get(index);
294 
295 		if (oldNode == newNode) {
296 			return;
297 		}
298 
299 		// add or update
300 		if (oldNode == null) {
301 			if (!AllocUtils.allocate(is, newNode)) {
302 				if (MessageDialog.openQuestion(Display.getDefault().getActiveShell(), "Error",
303 						"Stereotype application failed. The profile MARTE::Allocation is probably not applied. Try to apply it?")) {
304 					TransactionalEditingDomain domain = TransactionUtil.getEditingDomain(is);
305 					Command applyProfile = new AddMarteAndFcmProfile(PackageUtil.getRootPackage(is), AddMarteAndFcmProfile.APPLY_ALLOC, domain);
306 					CommandSupport.exec(domain, applyProfile);
307 					AllocUtils.allocate(is, newNode);
308 				}
309 			}
310 		} else {
311 			AllocUtils.updateAllocation(is, oldNode, newNode);
312 		}
313 	}
