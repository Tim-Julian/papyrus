23  import org.eclipse.swt.widgets.Display;
24  import org.eclipse.swt.widgets.Shell;
25  import org.eclipse.uml2.uml.Package;
26  
27  /**
28   * Implementation class for ClassAction action
29   */
30  public class AllocateHandler extends CmdHandler {
31  
32  	/**
33  	 * {@inheritDoc}
34  	 */
35  	@Override
36  	public boolean isEnabled() {
37  		updateSelectedEObject();
38  		if (selectedEObject instanceof Package) {
39  			return true;
40  		}
41  		return false;
42  	}
43  
44  	/**
45  	 * {@inheritDoc}
46  	 */
47  	@Override
48  	public Object execute(ExecutionEvent event)
49  			throws ExecutionException {
50  		if (!(selectedEObject instanceof Package)) {
51  			return null;
52  		}
53  
54  		final Package cdp = (Package) selectedEObject;
55  		final Shell shell = Display.getDefault().getActiveShell();
56  
57  		// org.eclipse.papyrus.designer.components.vsl.ParseVSL.test();
58  		if (DepUtils.getTopLevelInstances(cdp).isEmpty()) {
59  			MessageDialog.openInformation(shell, "Error",
60  					"Deployment plan has no main instance (check stereotype attribute mainInstance)");
61  			return null;
62  		}
63  
64  		// 1. select possible connectors according to port types
65  		// (only show compatible connectors check-box?)
66  		// 2. select implementation group according to connector type
67  
68  		// container dialog: either extension, rule or interceptor
69  		// howto select? which? (and howto add/remove?) - Std - dialog is good?
70  		CommandSupport.exec("Instance allocation", event, new RunnableWithResult() {
71  			
72  			@Override
73  			public CommandResult run() {
74  				AllocationDialog allocDialog =
75  						new AllocationDialog(shell, cdp);
76  				allocDialog.setTitle("Allocate instances");
77  				allocDialog.open();
78  				if (allocDialog.getReturnCode() == IDialogConstants.OK_ID) {
79  					return CommandResult.newOKCommandResult();
80  				}
81  				else {
82  					return CommandResult.newCancelledCommandResult();
83  				}
84  			}
85  		});
86  
87  		return null;
88  	}
89  }
