31  /**
32   * Instantiate (bind via xtend template) the text within a C++Include stereotype.
33   *
34   * The actual is the first actual within the template binding. This function does not check
35   * whether the classifier has the template stereotype.
36   *
37   * Note: this function is C++ specific, but many parts of the model library are C++ specific as well
38   *
39   */
40  public class BindCppIncludeToFirstActual implements PostCopyListener, IM2MTrafo {
41  
42  	@Override
43  	public void postCopyEObject(LazyCopier copier, EObject targetEObj) {
44  		// if (copy.get(sourceEObj) isWithinTemplate)
45  		if (targetEObj instanceof Classifier) {
46  
47  			Classifier targetCl = (Classifier) targetEObj;
48  			try {
49  				TemplateBinding binding = TemplateInstantiation.context.getBinding();
50  				Classifier actual = TemplateUtils.getFirstActualFromBinding(binding);
51  				Include cppInclude = UMLUtil.getStereotypeApplication(targetCl, Include.class);
52  				if ((actual != null) && (cppInclude != null)) {
53  					TransformationContext.current.classifier = targetCl;
54  					String newBody = TextTemplateBinding.bind(cppInclude.getBody(), actual, null);
55  					String newPreBody = TextTemplateBinding.bind(cppInclude.getPreBody(), actual, null);
56  					String newHeader = TextTemplateBinding.bind(cppInclude.getHeader(), actual, null);
57  					cppInclude.setBody(newBody);
58  					cppInclude.setPreBody(newPreBody);
59  					cppInclude.setHeader(newHeader);
60  				}
61  			} catch (TransformationException e) {
62  				// create nested exception
63  				throw new RuntimeException(e);
64  			}
65  		}
66  	}
67  }
