49  	/**
50  	 * Retrieve a list of nodes to which the instance is allocated to *or*
51  	 * to which one of the contained instances is allocated to (recursively)
52  	 *
53  	 * @param instance
54  	 * @return a list of nodes
55  	 */
56  	public static EList<InstanceSpecification> getAllNodes(InstanceSpecification instance) {
57  		EList<InstanceSpecification> nodeList = getNodes(instance);
58  		for (Slot slot : instance.getSlots()) {
59  			if (!DepUtils.isShared(slot)) {
60  				InstanceSpecification containedInstance = DepUtils.getInstance(slot);
61  				if (containedInstance != null) {
62  					nodeList.addAll(getAllNodes(containedInstance));
63  				}
64  			}
65  		}
66  		return nodeList;
67  	}

69  	/**
70  	 * Retrieve a list of nodes to which the set of passed instance is allocated to *or*
71  	 * to which one of the contained instances is allocated to (recursively)
72  	 *
73  	 * @param instanceList a list of instances
74  	 * @return a list of nodes
75  	 */
76  	public static EList<InstanceSpecification> getAllNodes(Collection<InstanceSpecification> instances) {
77  		EList<InstanceSpecification> nodeList = new UniqueEList<InstanceSpecification>();
78  		for (InstanceSpecification instance : instances) {
79  			nodeList.addAll(getAllNodes(instance));
80  		}
81  		return nodeList;
82  	}
83  	/**
84  	 * Retrieve a list of nodes to which the instance is allocated to *or*
85  	 * to which one of the contained instances is allocated to (recursively)
86  	 *
87  	 * @param instanceAttribute
88  	 *            an attribute within a composite that represents a component instance
89  	 *            or a set thereof, if the composite itself is instantiated multiple times)
90  	 * @return a list of nodes
91  	 */
92  	public static Property getThreadAlloc(Property instanceAttribute) {
93  		for (DirectedRelationship relation : instanceAttribute.getSourceDirectedRelationships()) {
94  			if (StereotypeUtil.isApplied(relation, Allocate.class)) {
95  				if (relation.getTargets().size() != 1) {
96  					continue;
97  				}
98  				Element targetElem = relation.getTargets().get(0);
99  				if (!(targetElem instanceof Property)) {
100 					continue;
101 				}
102 				Property target = (Property) targetElem;
103 				if (StereotypeUtil.isApplied(target.getType(), SwSchedulableResource.class)) {
104 					return target;
105 				}
106 			}
107 		}
108 		return null;
109 	}

113 	 * Retrieve a list of nodes to which the instance is allocated to *or*
114 	 * to which one of the containing instances is explicitly allocated
115 	 *
116 	 * @param instance
117 	 * @return a list of nodes
118 	 */
119 	public static EList<InstanceSpecification> getAllNodesOrThreadsParent(InstanceSpecification instance) {
120 		EList<InstanceSpecification> nodeList = getNodesOrThreads(instance);
121 		Package cdp = instance.getNearestPackage();
122 		// TODO: not very efficient: loop each time over all instance specifications
123 		for (NamedElement parentNE : cdp.getMembers()) {
124 			if (parentNE instanceof InstanceSpecification) {
125 				// possible parent instance specification
126 				InstanceSpecification parentIS = (InstanceSpecification) parentNE;
127 				if (DepUtils.getContainedInstances(parentIS).contains(instance)) {
128 					nodeList.addAll(getAllNodesOrThreadsParent(parentIS));
129 				}
130 			}
131 		}
132 		return nodeList;
133 	}

151 	/**
152 	 * This method returns a list of nodes (or threads) to which the passed instance is allocated.
153 	 * It is based on MARTE Allocation (a stereotyped abstraction) which is a generic
154 	 * mechanism to deploy UML elements to nodes.
155 	 *
156 	 * @param instanceOrThread
157 	 *            The instance for which we like to know the allocation information
158 	 * @return
159 	 */
160 	public static EList<InstanceSpecification> getNodesOrThreads(InstanceSpecification instanceOrThread) {
161 		EList<InstanceSpecification> nodeList = new UniqueEList<InstanceSpecification>();
162 
163 		for (DirectedRelationship relationship : instanceOrThread.getSourceDirectedRelationships(UMLPackage.eINSTANCE.getAbstraction())) {
164 			Abstraction abstraction = (Abstraction) relationship;
165 			if (StereotypeUtil.isApplied(abstraction, Allocate.class)) {
166 				for (Element target : abstraction.getTargets()) {
167 					if (target instanceof InstanceSpecification) {
168 						nodeList.add((InstanceSpecification) target);
169 					}
170 				}
171 			}
172 		}
173 		return nodeList;
174 	}
175 
176 	/**
177 	 * This method returns a list of nodes to which the passed instance is allocated. If
178 	 * the instance is allocated directly to a node, this node is returned. If the instance
179 	 * is allocated to a thread, the allocation of the thread to a node is returned.
180 	 *
181 	 * @param instanceOrThread
182 	 *            The instance that should be deployed
183 	 * @return
184 	 */
185 	public static EList<InstanceSpecification> getNodes(InstanceSpecification instanceOrThread) {
186 		EList<InstanceSpecification> nodeList = new UniqueEList<InstanceSpecification>();
187 		EList<InstanceSpecification> nodeOrThreads = getNodesOrThreads(instanceOrThread);
188 		for (InstanceSpecification nodeOrThread : nodeOrThreads)
189 		{
190 			Classifier nodeOrThreadC = DepUtils.getClassifier(nodeOrThread);
191 			if (StereotypeUtil.isApplied(nodeOrThreadC, SwSchedulableResource.class)) {
192 				// treat case that instance is allocated to a thread
193 				// follow allocation of Thread
194 				nodeList.add(getNode(nodeOrThread));
195 			}
196 			else {
197 				nodeList.add(nodeOrThread);
198 			}
199 		}
200 		return nodeList;
201 	}

203 	/**
204 	 * This method updates (and optionally removes) the allocations of an instance
205 	 *
206 	 * @param instance
207 	 *            The instance that should be deployed
208 	 * @param oldNode
209 	 *            the old node allocation
210 	 * @param newNode
211 	 *            the new node allocation, may be null to indicate removal
212 	 * @return
213 	 */
214 	public static EList<Node> updateAllocation(InstanceSpecification instance, InstanceSpecification oldNode, InstanceSpecification newNode) {
215 		EList<Node> nodeList = new UniqueEList<Node>();
216 
217 		Iterator<DirectedRelationship> relShipIt =
218 				instance.getSourceDirectedRelationships(UMLPackage.eINSTANCE.getAbstraction()).iterator();
219 
220 		while (relShipIt.hasNext()) {
221 			Abstraction abstraction = (Abstraction) relShipIt.next();
222 			if (StereotypeUtil.isApplied(abstraction, Allocate.class)) {
223 				EList<NamedElement> suppliers = abstraction.getSuppliers(); // use suppliers instead of targets (derived)
224 				for (int index = 0; index < suppliers.size(); index++) {
225 					if (suppliers.get(index) == oldNode) {
226 						if (newNode == null) {
227 							// remove relationship completely, since we assume a single target
228 							// (cannot call remove on relShipIt, since list is derived)
229 							abstraction.destroy();
230 						} else {
231 							suppliers.set(index, newNode);
232 						}
233 						break;
234 					}
235 				}
236 			}
237 		}
238 		return nodeList;
239 	}

241 	/**
242 	 * Retrieve a list of nodes to which the instance is allocated to *or*
243 	 * to which one of the contained instances is allocated to (recursively)
244 	 *
245 	 * @param instance
246 	 * @param A
247 	 *            port of the composite. Contained instances are examined only if
248 	 *            they have a connection with this port.
249 	 * @return a list of nodes
250 	 */
251 	public static EList<InstanceSpecification> getAllNodesForPort(InstanceSpecification instance, Port port) {
252 		EList<InstanceSpecification> nodeList = getNodes(instance);
253 		Classifier composite = DepUtils.getClassifier(instance);
254 		if (composite instanceof Class) {
255 			Class compositeCL = (Class) composite;
256 			for (Slot slot : instance.getSlots()) {
257 				Property containedProperty = (Property) slot.getDefiningFeature();
258 
259 				for (Connector connection : compositeCL.getOwnedConnectors()) {
260 					// is one connector end targeted at the containedProperty ?
261 					ConnectorEnd end = ConnectorUtil.connEndForPart(connection, containedProperty);
262 					if (end == null) {
263 						continue;
264 					}
265 
266 					// does the connector at the same connect the composite's port?
267 					if (ConnectorUtil.connectsPort(connection, port)) {
268 						Port containedPort = (Port) end.getRole();
269 						nodeList.addAll(getAllNodesForPort(DepUtils.getInstance(slot), containedPort));
270 					}
271 				}
272 			}
273 		}
274 		return nodeList;
275 	}

277 	public static boolean allocate(InstanceSpecification instance, InstanceSpecification node) {
278 		Package cdp = (Package) instance.getOwner();
279 		Abstraction allocation = (Abstraction)
280 				cdp.createPackagedElement("allocate " + instance.getName() + //$NON-NLS-1$
281 						" to " + node.getName(), UMLPackage.eINSTANCE.getAbstraction()); //$NON-NLS-1$
282 		if (StereotypeUtil.apply(allocation, Allocate.class) == null) {
283 			// stereotype application failed
284 			allocation.destroy();
285 			return false;
286 		}
287 		allocation.getClients().add(instance);
288 		allocation.getSuppliers().add(node);
289 		return true;
290 	}

292 	/**
293 	 * Assign node name to leafs
294 	 *
295 	 * @param instance
296 	 * @param A
297 	 *            port of the composite. Contained instances are examined only if the
298 	 *            have a connection with this port. If null, no sub-instances are examined
299 	 * @param nodes
300 	 *            A list of nodes to which leafs are allocated
301 	 */
302 	public static void propagateNodesViaPort(InstanceSpecification instance, Port port, EList<InstanceSpecification> nodes) {
303 		if (instance == null) {
304 			return;
305 		}
306 		EList<Slot> slots = instance.getSlots();
307 		Class composite = DepUtils.getImplementation(instance);
308 
309 		// Only set allocation on leafs (TODO: assure that components with configuration attributes are not considered as leafs)
310 		// if(composite.getParts().size() == 0) {
311 		for (InstanceSpecification node : nodes) {
312 			AllocUtils.allocate(instance, node);
313 		}
314 
315 		if ((composite != null) && (port != null)) {
316 			Class compositeCL = composite;
317 			for (Slot slot : slots) {
318 				Property containedProperty = (Property) slot.getDefiningFeature();
319 
320 				for (Connector connection : compositeCL.getOwnedConnectors()) {
321 					if (ConnectorUtil.connectsPort(connection, port)) {
322 						ConnectorEnd end = ConnectorUtil.connEndForPart(connection, containedProperty);
323 						// other connector end targeted at containedProperty?
324 						if (end != null) {
325 							Port containedPort = (Port) end.getRole();
326 							propagateNodesViaPort(DepUtils.getInstance(slot), containedPort, nodes);
327 						}
328 					}
329 				}
330 			}
331 		}
332 	}

334 	/**
335 	 * Return the start Port of a component, i.e. a port that corresponds to the "magic" port name
336 	 * start and is typed with the interface IStart
337 	 *
338 	 * @param component
339 	 *            a component implementation
340 	 * @return The start port or null
341 	 */
342 	public static Port getStartPort(Class component) {
343 		Element startPortElem = ElementUtils.getNamedElementFromList(component.getAllAttributes(), startPortName);
344 		if (startPortElem instanceof Port) {
345 			Port startPort = (Port) startPortElem;
346 			if (startPort.getType().getName().equals(startPortType)) {
347 				return startPort;
348 			}
349 		}
350 		return null;
351 	}
