103 	/**
104 	 * Instantiate a deployment plan, i.e. generate an intermediate model via a
105 	 * sequence of transformations
106 	 *
107 	 * @param umlElement
108 	 *            a deployment plan (UML package) or a configuration (UML class)
109 	 * @param monitor
110 	 *            a progress monitor.
111 	 *
112 	 * @param project
113 	 *            the current project. This information is used to store the
114 	 *            intermediate model in a subfolder (tmpModel) of the current
115 	 *            project
116 	 * @param genOptions
117 	 *            select whether to produce an intermediate model only, also
118 	 *            code, ... @see GenerationOptions
119 	 */
120 	private void instantiate(IProgressMonitor monitor, int genOptions) {
121 		try {
122 			initialize(monitor, genOptions);
123 			executeTransformation();
124 		} catch (final TransformationException e) {
125 			printAndDisplayErrorMessage(e, Messages.InstantiateDepPlan_TransformationException, false);
126 		} catch (final Exception e) {
127 			printAndDisplayErrorMessage(e, Messages.InstantiateDepPlan_ErrorsDuringTransformation, true);
128 		}
129 	}

131 	private void initialize(IProgressMonitor monitor, int genOptions) throws TransformationException {
132 		this.monitor = monitor;
133 		this.generationOptions = genOptions;
134 		this.generateCode = (genOptions & GenerationOptions.MODEL_ONLY) == 0;
135 		this.generateCACOnly = (genOptions & GenerationOptions.CAC_ONLY) != 0;
136 	}

138 	protected void executeTransformation() throws Exception {
139 		ModelManagement intermediateModelManagement = null;
140 		UIContext.monitor = monitor;
141 		UIContext.configureProject = (generationOptions & GenerationOptions.REWRITE_SETTINGS) != 0;
142 		
143 		// 1a: create a new model (and applies same profiles / imports)
144 		Model existingModel = srcModelComponentDeploymentPlan.getModel();
145 		TransformationContext tc = new TransformationContext();
146 		TransformationContext.current = tc;
147 		TransformationContext.initialSourceRoot = existingModel;
148 		TransformationContext.initialDeploymentPlan = srcModelComponentDeploymentPlan;
149 		
150 		intermediateModelManagement = ModelManagement.createNewModel(existingModel.getName());
151 
152 		// get the temporary model
153 		Package intermediateModel = intermediateModelManagement.getModel();
154 
155 		// create a package for global enumerations that are used by xtend code
156 		EnumService.createEnumPackage(intermediateModel);
157 
158 		// create a lazy copier towards the intermediate model
159 		LazyCopier intermediateModelCopier = new LazyCopier(existingModel, intermediateModel, CopyExtResources.SELECTED, true);
160 
161 		// obtain the component deployment plan in target model, only make a shallow
162 		// copy right now, otherwise all instances would already be copied.
163 		intermediateModelCopier.shallowCopy(srcModelComponentDeploymentPlan);
164 
165 		InstanceConfigurator.onNodeModel = false;
166 
167 		tc.copier = intermediateModelCopier;
168 		Map<InstanceSpecification, InstanceSpecification> instanceMap = new HashMap<InstanceSpecification, InstanceSpecification>();
169 		
170 		// calculate which resources to copy: based on the classifiers that are referenced from the instances
171 		for (InstanceSpecification instance : DepUtils.getInstances(srcModelComponentDeploymentPlan)) {
172 			Classifier cl = DepUtils.getClassifier(instance);
173 			intermediateModelCopier.addResource(cl.eResource());
174 		}
175 
176 		for (InstanceSpecification instance : DepUtils.getTopLevelInstances(srcModelComponentDeploymentPlan)) {
177 			// InstanceSpecification newInstance = mainModelTrafo.transformInstance(instance, null);
178 			InstanceSpecification newInstance = intermediateModelCopier.getCopy(instance);
179 
180 			// --------------------------------------------------------------------
181 			checkProgressStatus();
182 			// --------------------------------------------------------------------
183 
184 			TransformationUtil.applyInstanceConfigurators(newInstance);
185 
186 			TransformationUtil.propagateAllocation(newInstance);
187 			instanceMap.put(instance, newInstance);
188 		}
189 		tc.deploymentPlan = intermediateModelCopier.getCopy(srcModelComponentDeploymentPlan);
190 		tc.modelRoot = PackageUtil.getRootPackage(tc.deploymentPlan);
191 
192 		M2MTrafoChain chain = DepUtils.getTransformationChain(srcModelComponentDeploymentPlan);
193 		
194 		ExecuteTransformationChain.apply(chain);
195 
196 		/*
197 		 * if (!generateCACOnly) {
198 		 * deployOnNodes(instanceMap, existingModel, intermediateModel);
199 		 * }
200 		 */
201 
202 		// setURIs and save (first update all URIs to ensure that model references use the
203 		// correct URI before saving)
204 		intermediateModelManagement.setURI(project, TEMP_MODEL_FOLDER, TEMP_MODEL_POSTFIX);
205 		// also save additional root elements
206 		for (ModelManagement mm : tc.copier.getAdditionalRootPkgs()) {
207 			mm.setURI(project, TEMP_MODEL_FOLDER, TEMP_MODEL_POSTFIX);
208 		}
209 		intermediateModelManagement.save();
210 		for (ModelManagement mm : tc.copier.getAdditionalRootPkgs()) {
211 			mm.save();
212 		}
213 
214 		// --------------------------------------------------------------------
215 		checkProgressStatus();
216 		// --------------------------------------------------------------------
217 
218 		intermediateModelManagement.dispose();
219 		// also dispose additional models
220 		for (ModelManagement mm : tc.copier.getAdditionalRootPkgs()) {
221 			mm.dispose();
222 		}
223 	}

226 		PackageableElement deploymentPlanFolder = generatedModel.getPackagedElement(DeployConstants.depPlanFolderHw);
227 		if (deploymentPlanFolder != null) {
228 			deploymentPlanFolder.destroy();
229 		}

232 	private void removeDerivedInterfacesInRoot(Model generatedModel) {
233 		// 2b: remove derived interfaces in root: derived interfaces
234 		// that can not be placed in the same package as the port type (e.g.
235 		// since read-only type from system library), are put in a top-level
236 		// package called "derivedInterfaces". Due to the copying of imports,
237 		// the top-level package has changed which implies that new derived
238 		// interfaces are put into a different package and the derivedInterfaces
239 		// package in the original root becomes obsolete. Delete this obsolete
240 		// package, if existing.
241 		for (PackageableElement packagedElement : generatedModel.getPackagedElements()) {
242 			if (packagedElement instanceof Package) {
243 				NamedElement derivedInterfaces = ((Package) packagedElement).getPackagedElement("derivedInterfaces"); //$NON-NLS-1$
244 				if (derivedInterfaces instanceof Package) {
245 					derivedInterfaces.destroy();
246 				}
247 			}
248 		}
249 	}

252 	 * 
253 	 * @param canonicalProjectName
254 	 *            the automatically calculated project name
255 	 * @param userProjectName
256 	 *            the project name choosen by the user
257 	 */
258 	public void updateProjectMapping(final String canonicalProjectName, final String userProjectName) {
259 		CommandSupport.exec(srcModelComponentDeploymentPlan, "Update project mapping", new Runnable() {
260 
261 			@Override
262 			public void run() {
263 				DeploymentPlan depPlan = UMLUtil.getStereotypeApplication(srcModelComponentDeploymentPlan, DeploymentPlan.class);
264 				String mapName = canonicalProjectName + "=" + userProjectName;
265 				for (String mapping : depPlan.getProjectMappings()) {
266 					if (mapping.startsWith(canonicalProjectName)) {
267 						mapping = mapName;
268 						return;
269 					}
270 				}
271 				depPlan.getProjectMappings().add(mapName);
272 			}
273 		});
274 
275 	}

277 	private void checkProgressStatus() throws InterruptedException {
278 		if (monitor.isCanceled()) {
279 			throw new InterruptedException();
280 		}
281 		monitor.worked(1);
282 	}

284 	private void printAndDisplayErrorMessage(Exception e, final String title, final boolean consultConsole) {
285 		String message = e.toString();
286 		if (consultConsole) {
287 			message = message + "\n\n" //$NON-NLS-1$
288 					+ Messages.InstantiateDepPlan_ConsultConsole;
289 		}
290 
291 		printAndDisplayErrorMessage(e, title, message, consultConsole);
292 	}
293 
294 	private void printAndDisplayErrorMessage(Exception e, final String title, final String message, final boolean consultConsole) {
295 		e.printStackTrace();
296 		displayError(title, message);
297 		Activator.log.error(e);
298 	}
299 
300 	private void displayError(final String title, final String message) {
301 		Display.getDefault().syncExec(new Runnable() {
302 			@Override
303 			public void run() {
304 				Shell shell = Display.getDefault().getActiveShell();
305 				MessageDialog.openInformation(shell, title, message);
306 			}
307 		});
308 	}
