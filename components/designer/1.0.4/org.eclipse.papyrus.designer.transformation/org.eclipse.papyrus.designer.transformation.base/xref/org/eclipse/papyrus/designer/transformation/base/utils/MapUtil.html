27  	/**
28  	 * Get or create a derived type for a given element
29  	 *
30  	 * @param element
31  	 *            An element for which the derived element should be created. Used to determine the model in which
32  	 *            to create the derived element
33  	 * @param prefix
34  	 *            prefix string
35  	 * @param type
36  	 *            type name
37  	 * @param createKind
38  	 *            if non NONE, create either an interfaces or a class, if it does not exist yet
39  	 * @return
40  	 */
41  	public enum ECreateKind {
42  		CLASS, INTERFACE, NONE
43  	}
44  
45  	/**
46  	 * 
47  	 * @param element
48  	 * @param format a format string for the name of the new type. This name is obtained by call the String.format function and
49  	 * 	passing the original type name as parameter. E.g. the call can use a fixed name, if the format string does not contain
50  	 *  a %s directive, a %s at the end of the format string enables a fixed prefix and so on.
51  	 * @param type
52  	 * @param createKind
53  	 * @return
54  	 */
55  	public static Type getOrCreateDerivedType(Element element, String format, Type type, ECreateKind createKind) {
56  		String typeName = String.format(format, type.getName());
57  	
58  		// create derived element in "derivedInterface" package within the model owning
59  		// the port (which must be an FCM model, since the port carries the FCM stereotype)
60  		Package baseModelOfPort = getTop(element);
61  		// handle specific case of type within template. Place derived type into same template.
62  		Package derivedTypes = getAndCreate(baseModelOfPort, DERIVED_TYPES, createKind != ECreateKind.NONE);
63  		if (derivedTypes != null) {
64  			Package owner = getAndCreate(derivedTypes, type.allNamespaces(), createKind != ECreateKind.NONE);
65  			Interface intf = null;
66  	
67  			PackageableElement pe = owner.getPackagedElement(typeName);
68  			if (pe instanceof Type) {
69  				// type already exists
70  				return (Type) pe;
71  			} else if (createKind == ECreateKind.INTERFACE) {
72  				// System.out.println ("Derived port types: create new interface " + typeName + " in package " + owner.getQualifiedName ());
73  				intf = owner.createOwnedInterface(typeName);
74  	
75  				// System.out.println ("Derived port types: Apply derived stereotype annotation to interface: " + intf.getQualifiedName());
76  				DerivedElement de = applyDE(intf);
77  				if (de != null) {
78  					// de may be null, if FCM is not properly applied
79  					de.setSource(type);
80  				}
81  				return intf;
82  			} else if (createKind == ECreateKind.CLASS) {
83  				// System.out.println ("Derived port types: create new interface " + typeName + " in package " + owner.getQualifiedName ());
84  				Class newType = owner.createOwnedClass(typeName, false);
85  	
86  				// System.out.println ("Derived port types: Apply derived stereotype annotation to interface: " + intf.getQualifiedName());
87  				DerivedElement de = applyDE(newType);
88  				if (de != null) {
89  					// de may be null, if FCM is not properly applied
90  					de.setSource(type);
91  				}
92  				return newType;
93  			}
94  		}
95  	
96  		// instead of returning null, return a dummy interface that indicates the user that an element needs updating.
97  		PackageableElement portKinds = baseModelOfPort.getImportedMember("PortKinds"); //$NON-NLS-1$
98  		if (portKinds instanceof Package) {
99  			PackageableElement pe = ((Package) portKinds).getPackagedElement("Please update derived elements"); //$NON-NLS-1$
100 			if (pe instanceof Interface) {
101 				// dummy interface exists
102 				return (Interface) pe;
103 			}
104 		}
105 		return null;
106 	}

108 	/**
109 	 * Get the template signature of a templateable element (typically a package). The class must
110 	 * (1) either own the signature
111 	 * (2) or merge with a package which owns the signature.
112 	 * Qompass enables the "extension" of existing packages via the package merge mechanism
113 	 *
114 	 * @param template
115 	 *            The potential template
116 	 * @return the signature or null, if none can be found.
117 	 */
118 	public static TemplateSignature getSignature(TemplateableElement template) {
119 		for (Element element : template.getOwnedElements()) {
120 			if (element instanceof TemplateSignature) {
121 				return (TemplateSignature) element;
122 			}
123 		}
124 	
125 		// enable multiple package templates sharing the same signature.
126 		if (template instanceof Package) {
127 			Package pkg = (Package) template;
128 			for (PackageMerge pkgImport : pkg.getPackageMerges()) {
129 				Package importedPkg = pkgImport.getMergedPackage();
130 				return getSignature(importedPkg);
131 			}
132 		}
133 		return null;
134 	}

136 	/**
137 	 * Return the top-level owner of an element. This function returns the same value
138 	 * as getModel, if the top-level element is a model and not contained in a template
139 	 * While this is the case for models, model libraries have a top-level package (not
140 	 * a model). In this case, getTop returns the top-level package whereas getModel
141 	 * would return null.
142 	 * In case of a package owning a signature, it returns this package instead of the
143 	 * top level package. The reason is that the derived types might reference elements
144 	 * of the template signature which would be undefined outside.
145 	 *
146 	 * @param element
147 	 *            An element of which we want to query
148 	 * @return the top-level owning package
149 	 */
150 	public static Package getTop(Element element) {
151 		while (element != null) {
152 			Element owner = element.getOwner();
153 			if (owner == null) {
154 				if (element instanceof Package) {
155 					return (Package) element;
156 				}
157 			}
158 			if ((element instanceof Package) && (getSignature((Package) element) != null)) {
159 				return (Package) element;
160 			}
161 			element = owner;
162 		}
163 		return null;
164 	}

166 	private static final String DERIVED_TYPES = "derivedTypes"; //$NON-NLS-1$

168 	/**
169 	 * Apply the derived element stereotype to a model element
170 	 *
171 	 * @param element
172 	 * @return
173 	 */
174 	public static DerivedElement applyDE(Element element) {
175 		Stereotype stereotype = element.getApplicableStereotype("FCM::DerivedElement"); //$NON-NLS-1$
176 		if (stereotype != null) {
177 			EObject alreadyApplied = element.getStereotypeApplication(stereotype);
178 			if (alreadyApplied instanceof DerivedElement) {
179 				return (DerivedElement) alreadyApplied;
180 			}
181 			alreadyApplied = element.applyStereotype(stereotype);
182 			if (alreadyApplied instanceof DerivedElement) {
183 				return (DerivedElement) alreadyApplied;
184 			}
185 		}
186 		return null;
187 	}

189 	/**
190 	 * This function returns a Package reference that corresponds to a qualified name.
191 	 * Packages are created, if not existing yet - the function is thus a bit similar to
192 	 * the unix "mkdir -p" command. Note that the main model within the list of name-spaces is
193 	 * ignored, in order to avoid that a sub-package within a model starts with the name
194 	 * of the model.
195 	 *
196 	 * @param model
197 	 *            a model
198 	 * @param list
199 	 *            a list of namespace elements (as can be obtained via allNamespaces),
200 	 *            the top-level element is the last within the list.
201 	 * @param skipTop
202 	 *            if true, skip top level namespace element
203 	 * @return
204 	 */
205 	public static Package getAndCreate(Package root, EList<Namespace> list, boolean createOnDemand) {
206 		for (int i = list.size() - 1; i >= 0; i--) {
207 			Namespace ns = list.get(i);
208 	
209 			NamedElement pkg = root.getOwnedMember(ns.getName());
210 			if ((pkg == null) && createOnDemand) {
211 				// package does not exist => create it.
212 				pkg = root.createNestedPackage(ns.getName());
213 				// copy stereotype to create package
214 			}
215 			if (!(pkg instanceof Package)) {
216 				break;
217 			}
218 			root = (Package) pkg;
219 		}
220 		return root;
221 	}

223 	/**
224 	 * Get or create a package within another package (often the root of a model), i.e. return
225 	 * a package with the passed name, if it exists or create a new one of this name
226 	 *
227 	 * @param root
228 	 *            a package potentially owning a certain element
229 	 * @param name
230 	 *            the name of a package that should be be returned.
231 	 * @return a package
232 	 */
233 	public static Package getAndCreate(Package root, String name, boolean createOnDemand) {
234 		NamedElement pkg = root.getOwnedMember(name);
235 		if ((pkg == null) && createOnDemand) {
236 			pkg = root.createNestedPackage(name);
237 		}
238 		return (Package) pkg;
239 	}
