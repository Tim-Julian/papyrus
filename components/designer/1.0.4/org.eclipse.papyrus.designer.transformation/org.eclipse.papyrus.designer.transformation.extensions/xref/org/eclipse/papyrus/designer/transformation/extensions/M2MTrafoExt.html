12  package org.eclipse.papyrus.designer.transformation.extensions;
13  
14  import org.eclipse.core.runtime.CoreException;
15  import org.eclipse.core.runtime.IConfigurationElement;
16  import org.eclipse.core.runtime.IExtensionRegistry;
17  import org.eclipse.core.runtime.Platform;
18  import org.eclipse.emf.common.util.EList;
19  import org.eclipse.papyrus.designer.transformation.base.utils.TransformationRTException;
20  import org.eclipse.papyrus.designer.transformation.profile.Transformation.ApplyTransformation;
21  import org.eclipse.papyrus.designer.transformation.profile.Transformation.M2MTrafo;
22  import org.eclipse.uml2.uml.Element;
23  import org.eclipse.uml2.uml.util.UMLUtil;
24  
25  
26  /**
27   * Support for binding helper
28   */
29  public class M2MTrafoExt {
30  
31  	public static final String TRANSFORMATION_ID = "transformationID"; //$NON-NLS-1$
32  	public static final String M2M_TRAFO = Activator.PLUGIN_ID + ".m2mTrafo"; //$NON-NLS-1$
33  
34  	/**
35  	 * Obtain a list of M2M transformations that are explicitly applied to a model element 
36  	 * @param element the model element
37  	 * @return the list of M2M transformations
38  	 */
39  	public static EList<M2MTrafo> getM2MTrafos(Element element) {
40  		ApplyTransformation applyTrafo =
41  				UMLUtil.getStereotypeApplication(element, ApplyTransformation.class);
42  		if (applyTrafo != null) {
43  			// execute the set of transformations applied to the element
44  			return applyTrafo.getTrafo();
45  		}
46  		return null;
47  	}
48  	
49  	/**
50  	 * Get a reference to the M2M transformation class using the m2mTrafo element (stereotype)
51  	 * if the kind option is set, the transformation class is obtained via the kind.
52  	 *
53  	 * @param m2mTrafo a m2m transformation reference
54  	 * @return
55  	 */
56  	public static IM2MTrafo getM2MTrafo(M2MTrafo m2mTrafo) {
57  		if (m2mTrafo != null) {
58  			return getM2MTrafo(m2mTrafo.getBase_Class().getQualifiedName());
59  		}
60  		return null;
61  	}
62  
63  	/**
64  	 * The a language specific M2M transformation from a generic one, provided that the following
65  	 * convention is applied: a language specific transformation is registered by appending the language
66  	 * name and :: as separator as transformation ID
67  	 * @param m2mTrafo a generic transformation
68  	 * @param language the name of a language, e.g. C++ or Java (as defined currently in the common languages.uml library)
69  	 * @return a language specific model library
70  	 */
71  	public static IM2MTrafo getM2MTrafoForLanguage(M2MTrafo m2mTrafo, String language) {
72  		return M2MTrafoExt.getM2MTrafo(String.format("%s::%s", m2mTrafo.getBase_Class().getQualifiedName(), language)); //$NON-NLS-1$
73  	}
74  	
75  	/**
76  	 * Get a transformation by passing the transformation ID
77  	 * @param trafoID the ID of a registered transformation, typically the qualified name of the associated model element
78  	 * @return the transformation, if it exists, null otherwise
79  	 */
80  	public static IM2MTrafo getM2MTrafo(String trafoID) {
81  		IExtensionRegistry reg = Platform.getExtensionRegistry();
82  		IConfigurationElement[] configElements = reg.getConfigurationElementsFor(M2M_TRAFO);
83  		for (IConfigurationElement configElement : configElements) {
84  			try {
85  				final String transformationID = configElement.getAttribute(TRANSFORMATION_ID);
86  				if ((transformationID != null) && transformationID.equals(trafoID)) {
87  					final Object obj = configElement.createExecutableExtension("class"); //$NON-NLS-1$
88  					if (obj instanceof IM2MTrafo) {
89  						return (IM2MTrafo) obj;
90  					}
91  				}
92  			} catch (CoreException exception) {
93  				Activator.log.error(exception);
94  			}
95  		}
96  		throw new TransformationRTException(
97  				String.format("Cannot find class for transformation <%s>", trafoID));
98  	}
99  }
