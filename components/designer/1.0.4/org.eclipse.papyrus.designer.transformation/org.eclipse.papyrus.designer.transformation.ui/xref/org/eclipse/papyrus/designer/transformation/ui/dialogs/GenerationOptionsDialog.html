15  package org.eclipse.papyrus.designer.transformation.ui.dialogs;
16  
17  import java.util.Arrays;
18  
19  import org.eclipse.papyrus.designer.transformation.core.generate.GenerationOptions;
20  import org.eclipse.swt.SWT;
21  import org.eclipse.swt.events.SelectionEvent;
22  import org.eclipse.swt.events.SelectionListener;
23  import org.eclipse.swt.widgets.Button;
24  import org.eclipse.swt.widgets.Composite;
25  import org.eclipse.swt.widgets.Control;
26  import org.eclipse.swt.widgets.Shell;
27  import org.eclipse.ui.dialogs.SelectionStatusDialog;
28  
29  /**
30   * Select options for code generation.
31   *
32   * @author ansgar
33   */
34  public class GenerationOptionsDialog extends SelectionStatusDialog implements GenerationOptions {
35  
36  	protected Button fButtonModelOnly;
37  
38  	protected Button fButtonRewriteSettings;
39  
40  	protected Button fButtonOnlyChanged;
41  
42  	protected Button fButtonCACOnly;
43  
44  	public static int oldResultInt = 0;
45  
46  	public GenerationOptionsDialog(Shell parent) {
47  		super(parent);
48  	}
49  
50  	/**
51  	 * @see SelectionStatusDialog#computeResult()
52  	 */
53  	@Override
54  	protected void computeResult() {
55  		int resultInt = (isRewriteSettings() ? REWRITE_SETTINGS : 0) +
56  				(isOnlyChanged() ? ONLY_CHANGED : 0) +
57  				(isModelOnly() ? MODEL_ONLY : 0) +
58  				(isCACOnly() ? CAC_ONLY : 0);
59  		oldResultInt = resultInt;
60  		Object[] result = new Object[] {
61  				resultInt
62  		};
63  		setResult(Arrays.asList(result));
64  	}
65  
66  	@Override
67  	public Control createDialogArea(Composite parent) {
68  		Composite contents = (Composite) super.createDialogArea(parent);
69  		// (parent, "Container rules", "Avail. extensions/interceptors");
70  
71  		fButtonCACOnly = new Button(contents, SWT.CHECK);
72  		fButtonCACOnly.setText("Only perform container and connector expansion (intermediate model without code generation)");
73  		fButtonCACOnly.addSelectionListener(new SelectionListener() {
74  
75  			@Override
76  			public void widgetSelected(SelectionEvent e) {
77  				updateButtonEnableStatus();
78  			}
79  
80  			@Override
81  			public void widgetDefaultSelected(SelectionEvent e) {
82  			}
83  		});
84  		fButtonCACOnly.setSelection((oldResultInt & CAC_ONLY) != 0);
85  
86  		fButtonModelOnly = new Button(contents, SWT.CHECK);
87  		fButtonModelOnly.setText("Only generate model (not code)");
88  		fButtonModelOnly.addSelectionListener(new SelectionListener() {
89  
90  			@Override
91  			public void widgetSelected(SelectionEvent e) {
92  				// only write types that have changed does not make sense, if the whole directory is cleaned
93  				boolean modelOnly = fButtonModelOnly.getSelection();
94  				fButtonOnlyChanged.setEnabled(!modelOnly);
95  			}
96  
97  			@Override
98  			public void widgetDefaultSelected(SelectionEvent e) {
99  			}
100 		});
101 		fButtonModelOnly.setSelection((oldResultInt & MODEL_ONLY) != 0);
102 
103 		fButtonRewriteSettings = new Button(contents, SWT.CHECK);
104 		fButtonRewriteSettings.setText("Rewrite project settings, if project exists already (may overwrite user settings)");
105 		fButtonRewriteSettings.setSelection((oldResultInt & REWRITE_SETTINGS) != 0);
106 
107 		fButtonOnlyChanged = new Button(contents, SWT.CHECK);
108 		fButtonOnlyChanged.setText("Experimental: only write types that have changed (based on a difference model via EMF compare)");
109 		fButtonOnlyChanged.setSelection((oldResultInt & ONLY_CHANGED) != 0);
110 
111 		updateButtonEnableStatus();
112 		return contents;
113 	}
114 
115 	protected void updateButtonEnableStatus() {
116 		boolean onlyCAC = fButtonCACOnly.getSelection();
117 		boolean modelOnly = fButtonModelOnly.getSelection();
118 		fButtonModelOnly.setEnabled(!onlyCAC);
119 		fButtonOnlyChanged.setEnabled(!modelOnly && !onlyCAC);
120 		fButtonRewriteSettings.setEnabled(!onlyCAC);
121 	}
122 
123 	public boolean isRewriteSettings() {
124 		return fButtonRewriteSettings.getSelection();
125 	}
126 
127 	public boolean isOnlyChanged() {
128 		return fButtonOnlyChanged.getSelection();
129 	}
130 
131 	public boolean isModelOnly() {
132 		return fButtonModelOnly.getSelection();
133 	}
134 
135 	public boolean isCACOnly() {
136 		return fButtonCACOnly.getSelection();
137 	}
138 }
