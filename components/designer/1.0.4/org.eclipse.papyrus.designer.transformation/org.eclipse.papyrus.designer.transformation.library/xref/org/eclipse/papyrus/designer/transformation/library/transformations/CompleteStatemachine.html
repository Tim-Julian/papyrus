25  import org.eclipse.uml2.uml.Element;
26  import org.eclipse.uml2.uml.StateMachine;
27  
28  /**
29   * Programming language neutral state-machine transformation that delegates to a language
30   * specific variant
31   */
32  public class CompleteStatemachine implements IM2MTrafoElem {
33  
34  	@Override
35  	public void transformElement(M2MTrafo trafo, Element element) throws TransformationException {
36  		if (element instanceof Class) {
37  			Class clazz = (Class) element;
38  			if (clazz.getClassifierBehavior() instanceof StateMachine) {
39  				String language = DepUtils.getLanguageFromElement(clazz);
40  				IM2MTrafo trafoForLang = M2MTrafoExt.getM2MTrafoForLanguage(trafo, language);
41  				if (trafoForLang instanceof IM2MTrafoElem) {
42  					((IM2MTrafoElem) trafoForLang).transformElement(trafo, element);
43  				}
44  				else {
45  					throw new TransformationException(String.format(Messages.CompleteStatemachine_NoTransformationFound, language));
46  				}
47  			}
48  		}
49  	}
50  }
