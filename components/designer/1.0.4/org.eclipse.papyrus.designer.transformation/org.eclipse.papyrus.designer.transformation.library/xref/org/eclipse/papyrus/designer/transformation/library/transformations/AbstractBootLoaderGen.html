124 	/**
125 	 * Return the path from the main instance towards a sub-instance using the proper dereference
126 	 * operators (only relevant in case of C++)
127 	 *
128 	 * @param slotPath
129 	 * @param instance
130 	 * @param accessName
131 	 *            return the name to access the feature. Returns access path to instance, not
132 	 *            the name of the variable for this instance (if instantiated by bootloader)
133 	 * @return
134 	 */
135 	public String getPath(Stack<Slot> slotPath, InstanceSpecification instance, boolean accessName) {
136 		if (slotPath.size() > 0) {
137 			// start with first instance
138 			String path = slotPath.get(0).getOwningInstance().getName();
139 			boolean previousInstantiatedByBL = false;
140 			for (Slot pathElement : slotPath) {
141 				if (pathElement != null) {
142 					if (previousInstantiatedByBL && accessName) {
143 						// If an instance is instantiated by the bootloader, it is only referenced via its type in the
144 						// owning composite. Thus, configuration (and activation calls) might fail as the type might not
145 						// have these configuration properties or operations.
146 						// Therefore, configuration and initial calls use
147 						// - the path, if instantiated by the composite
148 						// - the variable name, if done by the bootloader
149 						path = ElementUtils.varName(path); // use variable name instead.
150 					}
151 					path += "." + pathElement.getDefiningFeature().getName(); //$NON-NLS-1$
152 					previousInstantiatedByBL = instantiateViaBootloader(pathElement.getDefiningFeature());
153 				}
154 			}
155 			if (previousInstantiatedByBL && !accessName) {
156 				// name of the variable for this expression instantiated by the bootloader
157 				path = ElementUtils.varName(path);
158 			}
159 			return path;
160 		}
161 		else {
162 			return instance.getName(); // instance has no path via slots, it is a top level instance
163 		}
164 	}

167 	 * Check whether the passed implementation has an unconnected start port.
168 	 * This information is required, since only unconnected start ports are automatically called by the
169 	 * bootloader, in particular we want to avoid calling a start port of an executor (which is connected)
170 	 * and its container.
171 	 *
172 	 * @param implementation
173 	 * @param containerSlot
174 	 * @return
175 	 */
176 	public static boolean hasUnconnectedStartRoutine(LazyCopier copier, Class implementation, Slot containerSlot) {
177 		Port startPort = AllocUtils.getStartPort(implementation);
178 		if (startPort != null) {
179 			return !isConnected(copier, containerSlot, startPort);
180 		}
181 		return false;
182 	}

184 	public boolean implementsIStart(Class implementation) {
185 		Port startPort = AllocUtils.getStartPort(implementation);
186 		if (startPort == null) {
187 			// OO case
188 			for (InterfaceRealization ir : implementation.getInterfaceRealizations()) {
189 				if (ir.getContract().getQualifiedName().equals(SYSINTERFACES_ISTART)) {
190 					return true;
191 				}
192 			}
193 		}
194 		return false;
195 	}

197 	/**
198 	 * Check, if the passed implementation has an unconnected life-cycle interface (activate/deactivate).
199 	 * This information is required, since only unconnected life cycle ports are automatically called by the
200 	 * bootloader, in particular we want to avoid calling a life cycle port of an executor (which is connected)
201 	 * and its container.
202 	 *
203 	 * @param implementation
204 	 * @param name
205 	 * @return
206 	 */
207 	public static boolean hasUnconnectedLifeCycle(LazyCopier copy, Class implementation, Slot containerSlot) {
208 		if (implementation != null) {
209 			Element lcPortElem = ElementUtils.getNamedElementFromList(implementation.getAllAttributes(), "lc"); //$NON-NLS-1$
210 			if (lcPortElem instanceof Port) {
211 				Port lcPort = (Port) lcPortElem;
212 				// check, if port typed with ILifeCycle interface
213 				if (lcPort.getType().getName().equals(I_LIFE_CYCLE)) {
214 					return !isConnected(copy, containerSlot, lcPort);
215 				}
216 			}
217 		}
218 		return false;
219 	}

221 	/**
222 	 * The check verifies whether the passed port is connected within
223 	 * the context of the composite represented by the passed slot
224 	 *
225 	 * @param containerSlot
226 	 *            a slot within an instance that represents a composite class
227 	 * @Param a port that is checked for being connected
228 	 * @return true, if connected
229 	 */
230 	private static boolean isConnected(LazyCopier copier, Slot containerSlot, Port port) {
231 		if (containerSlot != null) {
232 			StructuralFeature sf = containerSlot.getDefiningFeature();
233 			if (sf instanceof Property) {
234 				Property part = (Property) sf;
235 				Class composite = part.getClass_();
236 				for (Connector connector : composite.getOwnedConnectors()) {
237 					// must assure same connector end connects part & port
238 					ConnectorEnd end = ConnectorUtil.connEndForPart(connector, part);
239 					if ((end != null) && (end.getRole() == port)) {
240 						return true;
241 					}
242 				}
243 			}
244 		}
245 		return false;
246 	}

256 		Slot slot = slotPath.peek();
257 		// String varName = getPath(slotPath, instance, false);
258 		StructuralFeature sf = slot.getDefiningFeature();
259 		if (sf == null) {
260 			throw new TransformationException(String.format("A slot for instance %s has no defining feature", instance.getName())); //$NON-NLS-1$
261 		}
262 
263 		String varName = instance.getName() + "." + sf.getName(); //$NON-NLS-1$
264 		for (ValueSpecification value : slot.getValues()) {
265 
266 			// only set value, if not null
267 			if (value.stringValue() != null) {
268 				m_initCodeCConfig += varName + " = " + value.stringValue() + EOL; //$NON-NLS-1$
269 			}
270 		}

273 	/**
274 	 * add the initialize operation. Must be called after a set of addInstance invocations
275 	 * @param language programming language to use (in the opaque behavior creation)
276 	 */
277 	public void addInit(String language) {
278 		Operation initOp = m_bootLoader.getOwnedOperation(INIT_OP, null, null);
279 
280 		OpaqueBehavior initBehavior = (OpaqueBehavior)
281 				m_bootLoader.createOwnedBehavior(initOp.getName(), UMLPackage.eINSTANCE.getOpaqueBehavior());
282 		initOp.getMethods().add(initBehavior);
283 
284 		String code = m_initCode;
285 		if (m_initCodeCConfig.length() > 0) {
286 			code += m_initCodeCConfig;
287 		}
288 		if (m_initCodeCConnections.length() > 0) {
289 			code += m_initCodeCConnections;
290 		}
291 		Comparator<Class> comparator = new Comparator<Class>() {
292 
293 			@Override
294 			public int compare(Class clazz1, Class clazz2) {
295 
296 				InitPrecedence precedenceC1 = UMLUtil.getStereotypeApplication(clazz1, InitPrecedence.class);
297 				InitPrecedence precedenceC2 = UMLUtil.getStereotypeApplication(clazz2, InitPrecedence.class);
298 				if (precedenceC1 != null) {
299 					// need to use named comparison instead of precedenceC1.getInvokeAfter ().contains (clazz2)
300 					// since class referenced by stereotype attribute still points to element in source model
301 					if (ElementUtils.getNamedElementFromList(precedenceC1.getInvokeAfter(), clazz2.getName()) != null) {
302 						return 1;
303 					}
304 					else if (ElementUtils.getNamedElementFromList(precedenceC1.getInvokeBefore(), clazz2.getName()) != null) {
305 						return -1;
306 					}
307 				}
308 				else if (precedenceC2 != null) {
309 					if (ElementUtils.getNamedElementFromList(precedenceC2.getInvokeAfter(), clazz1.getName()) != null) {
310 						return -1;
311 					}
312 					else if (ElementUtils.getNamedElementFromList(precedenceC2.getInvokeBefore(), clazz1.getName()) != null) {
313 						return 1;
314 					}
315 				}
316 				// singletons have precedence over "normal" classes
317 				boolean ci1IsSingleton = DepUtils.isSingleton(clazz1);
318 				boolean ci2IsSingleton = DepUtils.isSingleton(clazz2);
319 				if (ci1IsSingleton) {
320 					if (!ci2IsSingleton) {
321 						// not both are singletons
322 						return -1;
323 					}
324 				}
325 				else if (ci2IsSingleton) {
326 					return 1;
327 				}
328 				return 0;
329 			}
330 		};
331 		Class[] activationKeys = m_activation.keySet().toArray(new Class[0]);
332 
333 		if (activationKeys.length > 0) {
334 			Arrays.sort(activationKeys, comparator);
335 			code += languageActivation(activationKeys);
336 		}
337 		
338 		if (m_initCodeRun != null) {
339 			code += m_initCodeRun;
340 		}
341 		else {
342 			code += languageDefaultExecCode();
343 		}
344 		
345 		if (activationKeys.length > 0) {
346 			code += languageDeactivation(activationKeys);
347 		}
348 
349 		initBehavior.getLanguages().add(language);
350 		initBehavior.getBodies().add(code);
351 	}

354 	 * Normally, a composite instantiates its children. However, we want to enable the possibility
355 	 * to type a part in a composite with an abstract class and choose the concrete implementation
356 	 * in the moment of the deployment. In this case, the bootloader needs to perform the instantiation. 
357 	 * @param structuralFeature a structural feature in a composition (typically an attribute)
358 	 * @return true, if the boot loader should instantiate the associated component.
359 	 */
360 	protected boolean instantiateViaBootloader(StructuralFeature structuralFeature) {
361 		if (structuralFeature.getType() instanceof Classifier) {
362 			return ((Classifier) structuralFeature.getType()).isAbstract();
363 		}
364 		// should not happen (all UML types are classifiers)
365 		return false;
366 	}

408 	public Property addInstance(Stack<Slot> slotPath, InstanceSpecification instance, Class implementation)
409 			throws TransformationException
410 	{
411 		// TODO: comments not clear. seems unnecessary complex. Problem in general is that access to
412 		// shared instances needs to be configured.
413 		// It should always be possible to configure this instance via a path w/o sharing.
414 		String accessName = getPath(slotPath, instance, true);
415 		final String varName = getPath(slotPath, instance, false);
416 
417 		Property implemPart = null;
418 
419 		// containing instance not null (=> neither main instance nor singleton)
420 		Slot containerSlot = null;
421 		if (slotPath.size() > 0) {
422 			containerSlot = slotPath.peek();
423 
424 			// initialize part/property in containing instance. The containing instance itself is accessed
425 			// via the naming of the associated instance, the part itself via the name of the defining feature.
426 			if (DepUtils.isShared(containerSlot)) {
427 				// we need to initialize the property (a reference) with the given instance
428 				Stack<Slot> referencePath = DepUtils.getAccessPath(instance);
429 				final String referenceVarName = getPath(referencePath, instance, false);
430 
431 				// add code for initialization
432 				m_initCode += languageAssignRef(accessName, referenceVarName);
433 						
434 				// is a reference which should not be called via activation & start
435 				// return now and skip code below
436 				return implemPart;
437 			}
438 			else if (instantiateViaBootloader(containerSlot.getDefiningFeature())) {
439 				// let bootloader instantiate
440 				implemPart = m_bootLoader.createOwnedAttribute(/* "i_" + */varName, implementation);
441 				// add code for initialization (TODO: specific to C++!)
442 				m_initCode += accessName + " = &" + varName + EOL; //$NON-NLS-1$
443 				implemPart.setIsComposite(true);
444 			}
445 		}
446 		else {
447 			// top level instance => bootloader instantiates, create attribute
448 			implemPart = m_bootLoader.createOwnedAttribute(/* "i_" + */varName, implementation);
449 			implemPart.setIsComposite(true);
450 			// depending on the programming language, the created attribute is a reference and additional
451 			// code is required to create the instance
452 			m_initCode += languageCreateInstance(instance, implementation);
453 		}
454 		if (outputSizeof) {
455 			// TODO - specific for C++
456 			m_initCode += "cout << \"sizeof " + implementation.getName() + ": \" << sizeof (" + varName + ") << endl;" + EOL; //$NON-NLS-1$//$NON-NLS-2$//$NON-NLS-3$
457 		}
458 
459 		// Need to check whether implementation is an executor which is encapsulated in a container. In this case, only
460 		// the method of the container and not the method of the executor (which owns the same port) maybe called.
461 		// Currently, this check is based on the use of "executor" as reserved part name (validation checks that the
462 		// user does not use this name for application components)
463 		boolean unconnectedStart = hasUnconnectedStartRoutine(m_copier, implementation, containerSlot);
464 		boolean implementsIStart = implementsIStart(implementation);
465 		if (unconnectedStart || implementsIStart) {
466 			// check if already assigned earlier
467 			if (m_initCodeRun.equals(EMPTYSTR)) {
468 				// call start's run method
469 				// TODO: Need path that uses the right dereference operator ("->" or ".")
470 				m_initCodeRun = languageRunStart(varName, implementsIStart);
471 			} else {
472 				throw new TransformationException(String.format(
473 						Messages.BootLoaderGen_AtLeastOneBlockingCall,
474 						varName, m_initCodeRun));
475 			}
476 		}
477 		if (hasUnconnectedLifeCycle(m_copier, implementation, containerSlot)) {
478 			// precedence is checked below (when code is actually produced)
479 			// multiple varNames might share the same implementation. Put a list of variable names into the table
480 			EList<String> varNameList = m_activation.get(implementation);
481 			if (varNameList == null) {
482 				varNameList = new BasicEList<String>();
483 			}
484 			varNameList.add(varName + "."); //$NON-NLS-1$
485 			m_activation.put(implementation, varNameList);
486 		}
487 
488 		// check, if implementation contains a composite with assembly connectors
489 		for (Connector connector : implementation.getOwnedConnectors()) {
490 			if (ConnectorUtil.isAssembly(connector)) {
491 				m_initCodeCConnections += languageCreateConn(varName);
492 				break;
493 			}
494 		}
495 
496 		return implemPart;
497 	}

499 	public void addInstance(InstanceSpecification is, Stack<Slot> slotPath) throws TransformationException {
500 		Classifier implementation = DepUtils.getClassifier(is);
501 		if (implementation instanceof Class) {
502 			addInstance(slotPath, is, (Class) implementation);
503 		}
504 		for (Slot slot : is.getSlots()) {
505 			InstanceSpecification subIS = DepUtils.getInstance(slot);
506 			slotPath.push(slot);
507 			if (subIS != null) {
508 				addInstance(subIS, slotPath);
509 			}
510 			else {
511 				instanceConfig(slotPath, is);
512 			}
513 			slotPath.pop();	
514 		}
