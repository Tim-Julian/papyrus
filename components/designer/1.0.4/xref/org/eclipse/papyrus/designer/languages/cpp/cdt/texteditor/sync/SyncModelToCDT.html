30  /**
31   * Main listener for model changes (registered via plugin.xml). It will delegate
32   * to the sub-listeners for specific sub-elements (type, operation, port, ...) that
33   * can be found in this package
34   *
35   */
36  public class SyncModelToCDT {
37  
38  	private static final String CPP_SUFFIX = "cpp"; //$NON-NLS-1$
39  	/**
40  	 * set to true, if a synchronization from an CDT editor to the model is active
41  	 */
42  	public static boolean syncFromEditor;
43  
44  	public static IFile syncModelToCDT(Classifier classifier, String generatorID) {
45  		if ((classifier == null) || (classifier.eResource() == null)) {
46  			return null;
47  		}
48  
49  		ILangCodegen codegen = LanguageCodegen.getGenerator(TextEditorConstants.CPP, generatorID);
50  		
51  		IProject modelProject = codegen.getTargetProject(classifier, false);
52  		if (modelProject == null) {
53  			return null;
54  		}
55  
56  		IContainer srcPkg = null;
57  		IFile cppFile = null;
58  		try {
59  			codegen.generateCode(modelProject, classifier, null); // need listener for sync in both directions!
60  
61  			String suffix = (codegen instanceof ILangCodegen2) ?
62  				suffix = ((ILangCodegen2) codegen).getSuffix(FILE_KIND.BODY) : CPP_SUFFIX;
63  			
64  			cppFile = modelProject.getFile(new Path(codegen.getFileName(modelProject, classifier) + Constants.DOT + suffix));
65  	
66  			// IStorage storage = new TextStorage(string);
67  		} finally {
68  			// Refresh the container for the newly created files. This needs to be done even
69  			// during error because of the possibility for partial results.
70  			try {
71  				if (srcPkg != null) {
72  					srcPkg.refreshLocal(IResource.DEPTH_INFINITE, null);
73  				}
74  			} catch (CoreException e) {
75  				Activator.log.error(e);
76  			}
77  		}
78  		return cppFile;
79  	}
80  }
