43  /**
44   * Currently unused synchronization for deployment plans.
45   *
46   * @see comp.cea.ec3m.gentools.core.listeners.DepPlanListener
47   *
48   * @author ansgar
49   */
50  public class DepPlanSync {
51  
52  	/**
53  	 * Synchronize a deployment plan: remove slots that refer to properties that exist no longer
54  	 * and add those that do not exist yet
55  	 *
56  	 * @param element
57  	 *            an arbitrary element of the source model (i.e. the model that will
58  	 *            store the deployment plan
59  	 * @throws TransformationException
60  	 */
61  	public static void syncAllDepPlans(Element element) {
62  		for (Package depPlan : DepPlanUtils.getAllDepPlans(element)) {
63  			syncDepPlan(depPlan);
64  		}
65  	}
66  
67  	/**
68  	 * Synchronize a deployment plan: remove slots that refer to properties that exist no longer
69  	 * and add those that do not exist yet
70  	 *
71  	 * @param depPlan
72  	 * @throws TransformationException
73  	 */
74  	public static void syncDepPlan(Package depPlan) {
75  		removeUnusedSlots(depPlan);
76  		for (InstanceSpecification is : DepUtils.getTopLevelInstances(depPlan)) {
77  			addCDP(depPlan, is, is.getName());
78  		}
79  	}
80  
81  	private static void addCDP(Package depPlan, InstanceSpecification instance, String canonicalName)
82  	{
83  		Class implementation = DepUtils.getImplementation(instance);
84  		if (!instance.getName().equals(canonicalName)) {
85  			instance.setName(canonicalName);
86  		}
87  		// check sub-instances
88  		for (Slot slot : instance.getSlots()) {
89  			InstanceSpecification subInstance = DepUtils.getInstance(slot);
90  			if (subInstance != null) {
91  				addCDP(depPlan, subInstance, canonicalName + "." + slot.getDefiningFeature().getName()); //$NON-NLS-1$
92  			}
93  		}
94  		for (Property attribute : ElementUtils.getParts(implementation)) {
95  			Type type = attribute.getType();
96  			if (type instanceof Class) {
97  				if (!hasSlot(instance, attribute)) {
98  					try {
99  						InstanceSpecification partIS =
100 								DepCreation.createDepPlan(depPlan, (Class) type, instance.getName() + "." + attribute.getName(), true); //$NON-NLS-1$
101 						DepPlanUtils.createSlot(depPlan, instance, partIS, attribute);
102 					} catch (TransformationException e) {
103 						Activator.log.error(e);
104 						throw new TransformationRTException(e.getMessage());
105 					}
106 				}
107 			}
108 		}
109 		// TODO: handle interaction components (extension point?)
110 	}
111 
112 	/**
113 	 * return a candidate for an interaction component. Since these instances have
114 	 * no associated slot, the derivedElement hints to the original connector
115 	 * (could still be problematic!)
116 	 * 
117 	 * @param cdp
118 	 * @param cl
119 	 * @return
120 	 */
121 	protected static InstanceSpecification findISforConn(Package cdp, final Connector conn) {
122 		ElementFilter filter = new ElementFilter() {
123 
124 			@Override
125 			public boolean acceptElement(Element element) {
126 				if (element instanceof InstanceSpecification) {
127 					InstanceSpecification is = (InstanceSpecification) element;
128 					DerivedElement de = UMLUtil.getStereotypeApplication(is, DerivedElement.class);
129 					if (de != null && de.getSource() == conn) {
130 						return true;
131 					}
132 				}
133 				return false;
134 			}
135 		};
136 		EList<InstanceSpecification> iList = new BasicEList<InstanceSpecification>();
137 		DepUtils.getAllInstances(cdp, iList, filter);
138 		if (iList.size() > 0) {
139 			return iList.get(0);
140 		}
141 		return null;
142 	}
143 
144 	private static boolean hasSlot(InstanceSpecification instance, Property attribute) {
145 		for (Slot slot : instance.getSlots()) {
146 			if (slot.getDefiningFeature() == attribute) {
147 				return true;
148 			}
149 		}
150 		return false;
151 	}
152 
153 	public static void removeUnusedSlots(Package depPlan) {
154 		// remove elements that are no longer in the plan
155 		for (NamedElement member : depPlan.getMembers()) {
156 			if (member instanceof InstanceSpecification) {
157 				InstanceSpecification instance = (InstanceSpecification) member;
158 				Iterator<Slot> slotIt = instance.getSlots().iterator();
159 				while (slotIt.hasNext()) {
160 					Slot slot = slotIt.next();
161 					if (slot.getDefiningFeature() == null) {
162 						// property has been removed => remove associated slot
163 						for (ValueSpecification value : slot.getValues()) {
164 							if (value instanceof InstanceValue) {
165 								InstanceSpecification is = ((InstanceValue) value).getInstance();
166 								DepPlanUtils.delDepPlan(is);
167 							}
168 						}
169 						slot.destroy();
170 						slotIt.remove();
171 					}
172 				}
173 			}
174 		}
175 	}
