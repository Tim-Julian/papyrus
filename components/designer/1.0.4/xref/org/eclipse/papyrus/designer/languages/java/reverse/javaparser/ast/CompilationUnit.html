34  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.body.EmptyTypeDeclaration;
35  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.body.EnumDeclaration;
36  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.comments.Comment;
37  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.comments.JavadocComment;
38  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.body.TypeDeclaration;
39  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.visitor.GenericVisitor;
40  import org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.visitor.VoidVisitor;
41  
42  import java.util.List;
43  
44  import static org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.internal.Utils.*;

46  /**
47   * <p>
48   * This class represents the entire compilation unit. Each java file denotes a
49   * compilation unit.
50   * </p>
51   * The CompilationUnit is constructed following the syntax:<br>
52   * <pre>
53   * {@code
54   * CompilationUnit ::=  ( }{@link PackageDeclaration}{@code )?
55   *                      ( }{@link ImportDeclaration}{@code )*
56   *                      ( }{@link TypeDeclaration}{@code )*
57   * }
58   * </pre>
59   * @author Julio Vilmar Gesser
60   */
61  public final class CompilationUnit extends Node {
62  
63      private PackageDeclaration pakage;
64  
65      private List<ImportDeclaration> imports;
66  
67      private List<TypeDeclaration> types;
68  
69      public CompilationUnit() {
70      }
71  
72      public CompilationUnit(PackageDeclaration pakage, List<ImportDeclaration> imports, List<TypeDeclaration> types) {
73          setPackage(pakage);
74          setImports(imports);
75          setTypes(types);
76      }
77  
78      public CompilationUnit(int beginLine, int beginColumn, int endLine, int endColumn, PackageDeclaration pakage, List<ImportDeclaration> imports, List<TypeDeclaration> types) {
79          super(beginLine, beginColumn, endLine, endColumn);
80          setPackage(pakage);
81          setImports(imports);
82          setTypes(types);
83      }
84  
85      @Override
86      public <R, A> R accept(GenericVisitor<R, A> v, A arg) {
87          return v.visit(this, arg);
88      }
89  
90      @Override
91      public <A> void accept(VoidVisitor<A> v, A arg) {
92          v.visit(this, arg);
93      }
94  
95      /**
96       * Return a list containing all comments declared in this compilation unit.
97       * Including javadocs, line comments and block comments of all types,
98       * inner-classes and other members.<br>
99       * If there is no comment, <code>null</code> is returned.
100      * 
101      * @return list with all comments of this compilation unit or
102      *         <code>null</code>
103      * @see JavadocComment
104      * @see org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.comments.LineComment
105      * @see org.eclipse.papyrus.designer.languages.java.reverse.javaparser.ast.comments.BlockComment
106      */
107     public List<Comment> getComments() {
108         return this.getAllContainedComments();
109     }
110 
111     /**
112      * Retrieves the list of imports declared in this compilation unit or
113      * <code>null</code> if there is no import.
114      * 
115      * @return the list of imports or <code>null</code> if there is no import
116      */
117     public List<ImportDeclaration> getImports() {
118         imports = ensureNotNull(imports);
119         return imports;
120     }
121 
122     /**
123      * Retrieves the package declaration of this compilation unit.<br>
124      * If this compilation unit has no package declaration (default package),
125      * <code>null</code> is returned.
126      * 
127      * @return the package declaration or <code>null</code>
128      */
129     public PackageDeclaration getPackage() {
130         return pakage;
131     }
132 
133     /**
134      * Return the list of types declared in this compilation unit.<br>
135      * If there is no types declared, <code>null</code> is returned.
136      * 
137      * @return the list of types or <code>null</code> null if there is no type
138      * @see AnnotationDeclaration
139      * @see ClassOrInterfaceDeclaration
140      * @see EmptyTypeDeclaration
141      * @see EnumDeclaration
142      */
143     public List<TypeDeclaration> getTypes() {
144         types = ensureNotNull(types);
145         return types;
146     }
147 
148     /**
149      * Sets the list of comments of this compilation unit.
150      * 
151      * @param comments
152      *            the list of comments
153      */
154     public void setComments(List<Comment> comments) {
155         throw new RuntimeException("Not implemented!");
156     }
157 
158     /**
159      * Sets the list of imports of this compilation unit. The list is initially
160      * <code>null</code>.
161      * 
162      * @param imports
163      *            the list of imports
164      */
165     public void setImports(List<ImportDeclaration> imports) {
166         this.imports = imports;
167 		setAsParentNodeOf(this.imports);
168     }
169 
170     /**
171      * Sets or clear the package declarations of this compilation unit.
172      * 
173      * @param pakage
174      *            the pakage declaration to set or <code>null</code> to default
175      *            package
176      */
177     public void setPackage(PackageDeclaration pakage) {
178         this.pakage = pakage;
179 		setAsParentNodeOf(this.pakage);
180     }
181 
182     /**
183      * Sets the list of types declared in this compilation unit.
184      * 
185      * @param types
186      *            the lis of types
187      */
188     public void setTypes(List<TypeDeclaration> types) {
189         this.types = types;
190 		setAsParentNodeOf(this.types);
191     }
192 }
