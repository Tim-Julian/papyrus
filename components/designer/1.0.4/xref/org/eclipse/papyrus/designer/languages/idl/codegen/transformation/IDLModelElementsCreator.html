13  package org.eclipse.papyrus.designer.languages.idl.codegen.transformation;
14  
15  import org.eclipse.core.resources.IProject;
16  import org.eclipse.core.runtime.IProgressMonitor;
17  import org.eclipse.papyrus.designer.languages.common.base.HierarchyLocationStrategy;
18  import org.eclipse.papyrus.designer.languages.common.base.ModelElementsCreator;
19  import org.eclipse.papyrus.designer.languages.idl.codegen.GenIDL2;
20  import org.eclipse.papyrus.designer.languages.idl.codegen.GenIDL3;
21  import org.eclipse.papyrus.designer.languages.idl.codegen.preferences.IDLCodeGenUtils;
22  import org.eclipse.papyrus.infra.tools.file.ProjectBasedFileAccess;
23  import org.eclipse.uml2.uml.Class;
24  import org.eclipse.uml2.uml.Classifier;
25  import org.eclipse.uml2.uml.DataType;
26  import org.eclipse.uml2.uml.Element;
27  import org.eclipse.uml2.uml.Interface;
28  import org.eclipse.uml2.uml.PackageableElement;
29  
30  public class IDLModelElementsCreator extends ModelElementsCreator {
31  
32  	private static final String IDL_LANG = "IDL"; //$NON-NLS-1$
33  
34  	public IDLModelElementsCreator(IProject project) {
35  		this(project, IDL_LANG);
36  		idlExt = IDLCodeGenUtils.getIDLSuffix();
37  	}
38  
39  	public IDLModelElementsCreator(IProject project, String language) {
40  		super(new ProjectBasedFileAccess(project), new HierarchyLocationStrategy(), language);
41  		this.project = project;
42  		idlExt = IDLCodeGenUtils.getIDLSuffix();
43  	}
44  
45  	protected String idlExt;
46  
47  	@Override
48  	protected boolean noCodeGen(Element element) {
49  		return false;
50  	}
51  
52  	@Override
53  	protected void createPackageableElementFile(PackageableElement pe, IProgressMonitor monitor) {
54  		if (pe instanceof Classifier) {
55  			Classifier classifier = (Classifier) pe;
56  			final String fileNameIDL = locStrategy.getFileName(classifier) + "." + idlExt; //$NON-NLS-1$
57  			String result = ""; //$NON-NLS-1$
58  			if ((classifier instanceof Interface) || (classifier instanceof DataType)) {
59  				result = GenIDL2.genIDL2(classifier).toString();
60  			} else if (classifier instanceof Class) {
61  				result = GenIDL3.genIDL3((Class) classifier).toString();
62  			}
63  			fileSystemAccess.generateFile(fileNameIDL, result);
64  		}
65  	}
66  }
