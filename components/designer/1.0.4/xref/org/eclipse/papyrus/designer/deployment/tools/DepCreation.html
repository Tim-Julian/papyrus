62  	 * Create a slot and add an instance value associated with a part instance
63  	 * specification.
64  	 *
65  	 * @param is
66  	 *            the instance specification for which a slot should be created
67  	 * @param partIS
68  	 *            the instance specification associated with the slot (to be
69  	 *            precise with slot's instance value)
70  	 * @param part
71  	 *            the part associated with the slot, i.e. its defining feature
72  	 * @return The created slot
73  	 */
74  	public static Slot createSlot(InstanceSpecification is, InstanceSpecification partIS, Property part) {
75  		// the instance specification of the composite has a slot for each part
76  		// and it points
77  		// to the instance specification associated with the part.
78  		Slot slot = is.createSlot();
79  		slot.setDefiningFeature(part);
80  
81  		InstanceValue iv = (InstanceValue) slot.createValue(null, null, UMLPackage.eINSTANCE.getInstanceValue());
82  		iv.setInstance(partIS);
83  		return slot;
84  
85  	}

88  	/**
89  	 * Create a Slot and a value for a configuration attribute
90  	 *
91  	 * @param attribute
92  	 */
93  	public static Slot createSlotForConfigProp(InstanceSpecification is, Property attribute) {
94  
95  		Slot slot = is.createSlot();
96  		slot.setDefiningFeature(attribute);
97  		// For primitive types, the UML type does not provide sufficient information to decide
98  		// whether it is a string or a numerical value. In case of the C++ profile, primitive
99  		// UML types denote a language specific type, but we want to avoid C++ specific code, if
100 		// possible.
101 		// => pragmatic solution: use LiteralString as default, unless a check for some known numerical
102 		// primitive types holds (in particular the UML primitive types)
103 		// In addition, string values are not automatically quoted, i.e. it is possible to enter e.g.
104 		// a float value as a string expression (it also means that a "real" string value must be quoted
105 		// by the user).
106 		Type type = attribute.getType();
107 		if (type != null) {
108 			String name = type.getName();
109 			String qname = type.getQualifiedName();
110 			if (type instanceof Enumeration) {
111 				slot.createValue(valueFor + attribute.getName(), type, UMLPackage.eINSTANCE.getInstanceValue());
112 			} else if (name.equals("Integer") || qname.equals(CORBAtypeNames.Octet) //$NON-NLS-1$
113 					|| qname.equals(CORBAtypeNames.Long) || qname.equals(CORBAtypeNames.UnsignedLong) || qname.equals(CORBAtypeNames.Short) || qname.equals(CORBAtypeNames.UnsignedShort)) {
114 				slot.createValue(valueFor + attribute.getName(), type, UMLPackage.eINSTANCE.getLiteralInteger());
115 			} else if (name.equals("Boolean")) { //$NON-NLS-1$
116 				slot.createValue(valueFor + attribute.getName(), type, UMLPackage.eINSTANCE.getLiteralBoolean());
117 			} else {
118 				slot.createValue(valueFor + attribute.getName(), type, UMLPackage.eINSTANCE.getLiteralString());
119 			}
120 		}
121 		return slot;
122 	}

162 	public static InstanceSpecification createDepPlan(Package cdp, Class typeOrImplem, String name, boolean createSlotsForConfigValues, Stack<Classifier> visitedClassifiers) throws TransformationException {
163 		// create an instance specification for the composite
164 		if (visitedClassifiers.contains(typeOrImplem)) {
165 			String path = ""; //$NON-NLS-1$
166 			for (Classifier cl : visitedClassifiers) {
167 				if (path.length() > 0) {
168 					path += ", "; //$NON-NLS-1$
169 				}
170 				path += cl.getName();
171 			}
172 			path += ", " + typeOrImplem.getName(); //$NON-NLS-1$
173 			throw new TransformationException(String.format(Messages.DepCreation_CircularReference, typeOrImplem.getQualifiedName(), path));
174 		}
175 		visitedClassifiers.push(typeOrImplem);
176 
177 		InstanceSpecification is;
178 		// treat singleton
179 		if (DepUtils.isSingleton(typeOrImplem)) {
180 
181 			// use canonical name for singleton instance - lower case for type-name
182 			String instanceName = DeployConstants.singletonPrefix + typeOrImplem.getName().toLowerCase();
183 			PackageableElement pe = cdp.getPackagedElement(instanceName);
184 
185 			if (pe == null) {
186 				// instance specification for singleton does not exist yet => create
187 				is = (InstanceSpecification) cdp.createPackagedElement(name, UMLPackage.eINSTANCE.getInstanceSpecification());
188 			} else if (pe instanceof InstanceSpecification) {
189 				// exists already, return it without recursing into its sub-specifications
190 				return (InstanceSpecification) pe;
191 			} else {
192 				// unlikely case that a packaged element with the name
193 				// <singletonISname> exists already, but is not an instance specification
194 				throw new TransformationException(String.format(Messages.DepCreation_SingletonExistsAlready, name));
195 			}
196 		} else {
197 			is = (InstanceSpecification) cdp.createPackagedElement(name, UMLPackage.eINSTANCE.getInstanceSpecification());
198 		}
199 
200 		Class implementation = null;
201 
202 		// treat HW architecture or nodes as implementations
203 		if (!typeOrImplem.isAbstract() || typeOrImplem instanceof Node) {
204 			// implementation is known => must be able to do this.
205 			if (typeOrImplem instanceof Class) {
206 				implementation = typeOrImplem;
207 			}
208 		} else {
209 			// problem? further tree expansion might depend on chosen
210 			// implementation)
211 			// TODO: don't know node yet => implementation choice is more
212 			// general than necessary
213 			implementation = DepUtils.chooseImplementation(typeOrImplem, null, null);
214 		}
215 
216 		if (!(implementation instanceof Class)) {
217 			throw new TransformationException(String.format(Messages.DepCreation_CannotFindImplementation, name, typeOrImplem.getName()));
218 		}
219 		// else implementation is instance of Class (and not null)
220 
221 		is.getClassifiers().add(implementation);
222 		// create slots for elements that are not in the current model but will be added by transformations, e.g.
223 		// container and connector transformations.
224 		// TODO: this is not very clean since the referenced elements are actually not in the model (e.g. the connector is
225 		// not yet a part which could referenced as a defining feature of a slot). This might require a different strategy
226 		// in which the transformed model is part of the user model and synchronized).
227 		if (createSlotsForConfigValues) {
228 			addConfigurationOfAddedElements(cdp, is);
229 		}
230 		/*
231 		 * EList<ContainerRule> rules = FCMUtil.getAllContainerRules(implementation);
232 		 * // create slots for configuration.
233 		 * for (ContainerRule rule : rules) {
234 		 * addConfigurationOfContainer(rule, is);
235 		 * }
236 		 * }
237 		 * 
238 		 * for (Connector connector : implementation.getOwnedConnectors()) {
239 		 * org.eclipse.papyrus.designer.components.FCM.Connector fcmConn = UMLUtil.getStereotypeApplication(connector, org.eclipse.papyrus.designer.components.FCM.Connector.class);
240 		 * if (fcmConn != null) {
241 		 * String partName = name + "." + connector.getName(); //$NON-NLS-1$
242 		 * InteractionComponent connectorComp = fcmConn.getIc();
243 		 * if (connectorComp != null) {
244 		 * Class cl = fcmConn.getIc().getBase_Class();
245 		 * if (cl == null) {
246 		 * throw new TransformationException(Messages.DepCreation_FCMconnectorWithoutBaseClass);
247 		 * }
248 		 * // create sub-instance for connector. It is not possible to
249 		 * // create a slot in the owning instance specification,
250 		 * // since the connector cannot be referenced as a defining-feature
251 		 * createDepPlan(cdp, cl, partName, createSlotsForConfigValues, visitedClassifiers);
252 		 * }
253 		 * }
254 		 * }
255 		 */
256 
257 		for (Property attribute : implementation.getAllAttributes()) {
258 			// loop over all attributes (not only parts, since we need to
259 			// capture singletons)
260 			if (attribute instanceof Port) {
261 				continue;
262 			}
263 			Type type = attribute.getType();
264 
265 			if (DepUtils.isComposition(attribute)) {
266 				// composition, attribute is a part
267 				if (type instanceof Class) {
268 					Class cl = (Class) type;
269 
270 					// TODO: ad-hoc replication support. Better solution via design patterns
271 					int upper = attribute.getUpper();
272 					String infix = ""; //$NON-NLS-1$
273 
274 					// TODO: check validation constraints
275 					for (int i = 0; i < upper; i++) {
276 						String partName = name + "." + attribute.getName(); //$NON-NLS-1$
277 						if (upper > 1) {
278 							partName += "_" + infix + i; //$NON-NLS-1$
279 						}
280 						InstanceSpecification partIS = createDepPlan(cdp, cl, partName, createSlotsForConfigValues, visitedClassifiers);
281 						// may not create slot for singleton, since automatically done
282 						if (!DepUtils.isSingleton((Class) type)) {
283 							createSlot(is, partIS, attribute);
284 						}
285 					}
286 				} else if (StereotypeUtil.isApplied(attribute, ConfigurationProperty.class) && createSlotsForConfigValues) {
287 					// is a configuration property, create slot
288 					// TODO: implicit assumption that configuration attributes
289 					// are not components
290 					createSlotForConfigProp(is, attribute);
291 				}
292 			} else if (type instanceof Class) {
293 				// no composition - only create slot, if a singleton
294 				// (otherwise, it's not clear with which instance the slot
295 				// should be associated)
296 				if (DepUtils.isSingleton((Class) type)) {
297 					// is a singleton - exactly one instance exists
298 					// recursive call - pass empty name, since name for singletons is re-calculated.
299 					InstanceSpecification singletonIS = createDepPlan(cdp, (Class) type, "", createSlotsForConfigValues, visitedClassifiers); //$NON-NLS-1$
300 					createSlot(is, singletonIS, attribute);
301 				}
302 			} else if (type == null) {
303 				throw new TransformationException(String.format(Messages.DepCreation_TypeInAttributeUndefined, attribute.getName(), implementation.getName()));
304 			}
305 		}
306 		visitedClassifiers.pop();
307 		return is;
308 	}

345 	 * create a deployment plan, i.e. a set of instances that correspond to an
346 	 * implementation which is potentially a composite. In case of the latter,
347 	 * this function is called recursively to create deployment plans for the
348 	 * parts. Since a composite implementation may optionally specify only the
349 	 * type of parts (not the concrete implementation), this function also
350 	 * supports that a type is passed. In this case, an implementation is
351 	 * automatically assigned.
352 	 *
353 	 * @param cdp
354 	 *            the deployment plan (package) in which to create instances
355 	 * @param typeOrImplem
356 	 *            the type
357 	 * @param nane
358 	 *            the name of the instance
359 	 * @param createSlotsForConfigValues
360 	 *            if true, create slots for configuration values
361 	 */
362 	public static InstanceSpecification createPlatformInstances(Package platform, Class implementation, String name) throws TransformationException {
363 		// create an instance specification for the composite
364 		InstanceSpecification is = null;
365 		is = (InstanceSpecification) platform.createPackagedElement(name, UMLPackage.eINSTANCE.getInstanceSpecification());
366 		is.getClassifiers().add(implementation);
367 
368 		// add connector and container implementations
369 
370 		for (Property attribute : implementation.getAllAttributes()) {
371 			// loop over all attributes (not only parts, since we need to
372 			// capture singletons)
373 			if (attribute instanceof Port) {
374 				continue;
375 			}
376 			Type type = attribute.getType();
377 
378 			if (DepUtils.isComposition(attribute)) {
379 				// composition, attribute is a part
380 				if (type instanceof Class) { // should be a node, but do not require it
381 					Class cl = (Class) type;
382 
383 					int upper = attribute.getUpper();
384 					String infix = ""; //$NON-NLS-1$
385 					// TODO: check validation constraints
386 					for (int i = 0; i < upper; i++) {
387 						// prefix with name, unless null
388 						String partName = (name != null) ? name + DeployConstants.SEP_CHAR : ""; //$NON-NLS-1$
389 						partName += attribute.getName();
390 						if (upper > 1) {
391 							partName += "_" + infix + i; //$NON-NLS-1$
392 						}
393 						InstanceSpecification partIS = createPlatformInstances(platform, cl, partName);
394 
395 						if (is != null) {
396 							createSlot(is, partIS, attribute);
397 						}
398 					}
399 				} else if (StereotypeUtil.isApplied(attribute, ConfigurationProperty.class)) {
400 					// is a configuration property, create slot
401 					// TODO: implicit assumption that configuration attributes
402 					// are not components
403 					createSlotForConfigProp(is, attribute);
404 				}
405 			}
406 		}
407 		return is;
408 	}

441 	/**
442 	 * Initialize the automatic values within a deployment plan.
443 	 *
444 	 * @param is
445 	 *            the main instance of the deployment plan
446 	 */
447 	public static void initAutoValuesHelper(InstanceSpecification is) {
448 		for (Slot slot : is.getSlots()) {
449 			StructuralFeature sf = slot.getDefiningFeature();
450 			if (sf == null) {
451 				throw new RuntimeException(String.format(Messages.DepCreation_DefiningFeatureNull, slot.getOwningInstance().getName()));
452 			}
453 			if (StereotypeUtil.isApplied(sf, AutoIndex.class)) {
454 				Integer value = null;
455 				Object key;
456 				if (StereotypeUtil.isApplied(sf, AutoIndexPerNode.class)) {
457 					InstanceSpecification nodeOrThread = AllocUtils.getNode(is);
458 					key = sf.getName() + nodeOrThread.getName();
459 				} else {
460 					key = sf;
461 				}
462 				if (!map.containsKey(key)) {
463 					map.put(key, new Integer(0));
464 				}
465 				value = map.get(key);
466 				value++; // no need to put incremented value back into map,
467 							// since Integer is a reference => value is updated
468 
469 				// create slot and value specification (literal-integer) for the
470 				// auto index
471 				LiteralInteger li = (LiteralInteger) slot.createValue(sf.getName() + DeployConstants.AUTO_POSTFIX, sf.getType(), UMLPackage.eINSTANCE.getLiteralInteger());
472 				li.setValue(value);
473 
474 				// recursion in case of values that are instance values
475 				for (ValueSpecification vs : slot.getValues()) {
476 					if (vs instanceof InstanceValue) {
477 						InstanceSpecification subIS = ((InstanceValue) vs).getInstance();
478 						initAutoValues(subIS);
479 					}
480 				}
481 			}
482 		}
483 	}

485 	/**
486 	 * Copy automatic values within a deployment plan.
487 	 * Attributes can apply the CopyAttributeValue stereotype. The stereotype can point
488 	 * to another attribute (source). If the other attribute has a configuration value, it gets
489 	 * copied.
490 	 * The function takes into account that the source attribute might be in a component that is
491 	 * instantiated several times with different configuration values. The attribute that gets the
492 	 * copy must use the "nearest" value which means that there must be a common composite that
493 	 * contains both attribute (indirectly).
494 	 * The main use case is that connector fragments need to have common configuration attributes
495 	 * (e-g- server fragment has automatic index value, client fragment needs to copy this value)
496 	 *
497 	 * @param is
498 	 *            the main instance of the deployment plan
499 	 */
500 	public static void copyAutoValues(Stack<InstanceSpecification> isStack, InstanceSpecification is) {
501 		isStack.push(is);
502 		for (Slot slot : is.getSlots()) {
503 			StructuralFeature sf = slot.getDefiningFeature();
504 			if (sf == null) {
505 				throw new TransformationRTException(is.getName() + " has a slot without defining feature"); //$NON-NLS-1$
506 			}
507 			if (StereotypeUtil.isApplied(sf, CopyAttributeValue.class)) {
508 				CopyAttributeValue cav = UMLUtil.getStereotypeApplication(sf, CopyAttributeValue.class);
509 				Property source = cav.getSource();
510 				ValueSpecification vs = getNearestValue(isStack, source);
511 				if (vs instanceof LiteralInteger) {
512 					LiteralInteger liCopy = (LiteralInteger) slot.createValue(sf.getName() + "_copy", sf.getType(), //$NON-NLS-1$
513 							UMLPackage.eINSTANCE.getLiteralInteger());
514 					int value = ((LiteralInteger) vs).getValue();
515 					liCopy.setValue(value);
516 				}
517 			}
518 
519 			// recursion in case of values that are instance values
520 			for (ValueSpecification vs : slot.getValues()) {
521 				if (vs instanceof InstanceValue) {
522 					InstanceSpecification subIS = ((InstanceValue) vs).getInstance();
523 					if (subIS != null) {
524 						copyAutoValues(isStack, subIS);
525 					}
526 				}
527 			}
528 		}
529 		isStack.pop();
530 	}

532 	/**
533 	 * try to find a value (ValueSpecification) for the passed source element,
534 	 * beginning a the "deepest" instance specification of the passed stack. If
535 	 * not found, continue at elements higher up in the hierarchy
536 	 *
537 	 * @param isStack
538 	 *            a stack of instance specifications corresponding to a path
539 	 *            within an instance tree
540 	 * @param source
541 	 *            a property (defining feature) for which we search an instance
542 	 *            specification
543 	 * @return The value specifications for the passed source property or null
544 	 */
545 	public static ValueSpecification getNearestValue(Stack<InstanceSpecification> isStack, Property source) {
546 		Stack<InstanceSpecification> copy = new Stack<InstanceSpecification>();
547 		copy.addAll(isStack);
548 		while (!copy.isEmpty()) {
549 			InstanceSpecification pop = copy.pop();
550 			ValueSpecification vs = getNearestValueHelper(isStack, pop, source);
551 			if (vs != null) {
552 				return vs;
553 			}
554 		}
555 		return null;
556 	}

558 	/**
559 	 * Helper for getNearestValue: search for an instance specification that has
560 	 * source as defining feature. Start at the passed instance specification,
561 	 * but do not recurse into elements that are contained in the stack.
562 	 *
563 	 * @param isStack
564 	 *            A stack of instance specifications corresponding to a path
565 	 *            within an instance tree
566 	 * @param is
567 	 *            the starting instance specification
568 	 * @param source
569 	 *            source a property (defining feature) for which we search an
570 	 *            instance specification
571 	 * @return The value specifications for the passed source property or null
572 	 */
573 	public static ValueSpecification getNearestValueHelper(Stack<InstanceSpecification> isStack, InstanceSpecification is, Property source) {
574 		for (Slot slot : is.getSlots()) {
575 			StructuralFeature sf = slot.getDefiningFeature();
576 			if (sf == source) {
577 				// found property, now return first value specification
578 				for (ValueSpecification vs : slot.getValues()) {
579 					return vs;
580 				}
581 				return null;
582 			}
583 
584 			// recursion in case of values that are instance values
585 			for (ValueSpecification vs : slot.getValues()) {
586 				if (vs instanceof InstanceValue) {
587 					InstanceSpecification subIS = ((InstanceValue) vs).getInstance();
588 					if (!isStack.contains(subIS)) {
589 						// only recurse, if not contained in stack of instance
590 						// specifications (avoid traversing
591 						// the same elements multiple times)
592 						return getNearestValueHelper(isStack, subIS, source);
593 					}
594 				}
595 			}
596 		}
597 		return null;
598 	}
599 }
