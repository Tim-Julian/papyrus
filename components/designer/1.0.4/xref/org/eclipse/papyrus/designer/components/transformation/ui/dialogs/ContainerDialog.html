102 	@Override
103 	public Control createDialogArea(Composite parent) {
104 		Composite contents = (Composite) super.createDialogArea(parent);
105 		// (parent, "Container rules", "Avail. extensions/interceptors");
106 
107 		if (m_rulePropertiesOnly) {
108 			createRuleInfoGroup(contents);
109 			selectRule(m_currentRule);
110 		} else {
111 			createRuleSelectionGroup(contents);
112 			createRuleInfoGroup(contents);
113 		}
114 		return contents;
115 	}

117 	protected void createRuleSelectionGroup(Composite parent) {
118 		Composite ruleSelections = new Composite(parent, SWT.NONE);
119 		GridLayout grid = new GridLayout(1, true);
120 		GridData groupGridData = new GridData();
121 		// data3.widthHint = 400;
122 		// data3.heightHint = 300;
123 		groupGridData.grabExcessVerticalSpace = true;
124 		groupGridData.grabExcessHorizontalSpace = true;
125 		groupGridData.horizontalAlignment = GridData.FILL;
126 		groupGridData.verticalAlignment = GridData.FILL;
127 		ruleSelections.setLayout(grid);
128 		ruleSelections.setLayoutData(groupGridData);
129 		ruleSelections.setSize(400, 300);
130 
131 		/*
132 		 * fRules = DialogUtils.createFilteredList (ruleSelGroup, new RuleLabelProvider(), 200, 200,
133 		 * SWT.BORDER | SWT.V_SCROLL | SWT.H_SCROLL);
134 		 */
135 		GridData data = new GridData(GridData.FILL_BOTH);
136 		data.heightHint = 150;
137 		data.widthHint = 200;
138 
139 		ICheckStateListener checkListener = new ICheckStateListener() {
140 
141 			@Override
142 			public void checkStateChanged(CheckStateChangedEvent event) {
143 				Object obj = event.getElement();
144 				if (obj instanceof ContainerRule) {
145 					ContainerRule rule = (ContainerRule) obj;
146 					if (event.getChecked()) {
147 						applyRule(rule);
148 					}
149 					else {
150 						unapplyRule(rule);
151 					}
152 				}
153 			}
154 		};
155 
156 		//
157 		// --------------- global rules -------------------
158 		//
159 		Group gRuleSelGroup = new Group(ruleSelections, SWT.BORDER);
160 		gRuleSelGroup.setText(" available rules ");
161 		// ruleGroup.setLayout(new RowLayout (SWT.VERTICAL));
162 		gRuleSelGroup.setLayout(new GridLayout(1, false));
163 		gRuleSelGroup.setLayoutData(groupGridData);
164 
165 		EList<ContainerRule> globalRuleList = ContainerUtil.getAllRules(m_model);
166 		// Table table = new Table(gRuleSelGroup, SWT.CHECK);
167 		fRules = CheckboxTableViewer.newCheckList(gRuleSelGroup, SWT.BORDER);
168 		// fRules = new CheckboxTableViewer(table);
169 
170 
171 		GridData data2 = new GridData(GridData.FILL_BOTH);
172 		data2.heightHint = 200;
173 		// data.widthHint = 200;
174 		fRules.getTable().setLayoutData(data);
175 		fRules.setLabelProvider(new RuleLabelProvider());
176 		fRules.setContentProvider(new ArrayContentProvider());
177 
178 		fRules.setInput(globalRuleList.toArray());
179 		for (ContainerRule rule : globalRuleList) {
180 			if (isRuleApplied(rule)) {
181 				fRules.setChecked(rule, true);
182 			}
183 		}
184 		fRules.addSelectionChangedListener(new ISelectionChangedListener() {
185 
186 			@Override
187 			public void selectionChanged(SelectionChangedEvent event) {
188 				ISelection selection = fRules.getSelection();
189 				if (selection instanceof StructuredSelection) {
190 					Object[] selected = ((StructuredSelection) selection).toArray();
191 					if ((selected.length == 1) && (selected[0] instanceof ContainerRule)) {
192 						if (m_currentRule != selected[0]) {
193 							selectRule((ContainerRule) selected[0]);
194 						}
195 					}
196 				}
197 			}
198 		});
199 		fRules.addCheckStateListener(checkListener);
200 	}

241 	/**
242 	 * Helper: add a local container rule to the current class
243 	 *
244 	 * @param name
245 	 *            the name of the local rule
246 	 * @return
247 	 */
248 	ContainerRule addRule(String name) {
249 		Class ruleCl = (Class)
250 				m_component.createNestedClassifier(name, UMLPackage.eINSTANCE.getClass_());
251 		StereotypeUtil.apply(ruleCl, ContainerRule.class);
252 		return UMLUtil.getStereotypeApplication(ruleCl, ContainerRule.class);
253 	}
254 
255 	/**
256 	 * Helper function: delete a container rule from a class
257 	 *
258 	 * @param rule
259 	 */
260 	void deleteRule(ContainerRule rule) {
261 		Classifier ruleCl = m_component.getNestedClassifier(rule.getBase_Class().getName());
262 		if (ruleCl instanceof Class) {
263 			ruleCl.destroy();
264 		}
265 	}
266 
267 	/**
268 	 * Helper: apply a container rule to current component
269 	 *
270 	 * @param rule
271 	 */
272 	void applyRule(ContainerRule rule) {
273 		if (!StereotypeUtil.isApplied(m_component, RuleApplication.class)) {
274 			StereotypeUtil.apply(m_component, RuleApplication.class);
275 		}
276 		RuleApplication containerConfig =
277 				UMLUtil.getStereotypeApplication(m_component, RuleApplication.class);
278 		containerConfig.getContainerRule().add(rule);
279 	}
280 
281 	/**
282 	 * Helper: apply a container rule to current component
283 	 *
284 	 * @param rule
285 	 */
286 	void unapplyRule(ContainerRule rule) {
287 		RuleApplication containerConfig =
288 				UMLUtil.getStereotypeApplication(m_component, RuleApplication.class);
289 		int index = containerConfig.getContainerRule().indexOf(rule);
290 		if ((index >= 0) && (index < containerConfig.getContainerRule().size())) {
291 			containerConfig.getContainerRule().remove(index);
292 		}
293 	}
294 
295 	/**
296 	 * check whether a container rule is applied
297 	 * h * @param rule
298 	 *
299 	 * @return true, if container rule is applied
300 	 */
301 	boolean isRuleApplied(ContainerRule rule) {
302 		RuleApplication containerConfig =
303 				UMLUtil.getStereotypeApplication(m_component, RuleApplication.class);
304 		if (containerConfig != null) {
305 			return containerConfig.getContainerRule().contains(rule);
306 		}
307 		return false;
308 	}
