12  package org.eclipse.papyrus.designer.transformation.extensions;
13  
14  import org.eclipse.core.runtime.CoreException;
15  import org.eclipse.core.runtime.IConfigurationElement;
16  import org.eclipse.core.runtime.IExtensionRegistry;
17  import org.eclipse.core.runtime.Platform;
18  import org.eclipse.papyrus.designer.deployment.profile.Deployment.UseInstanceConfigurator;
19  import org.eclipse.uml2.uml.InstanceSpecification;
20  import org.eclipse.uml2.uml.Property;
21  
22  
23  /**
24   * Support for the configuration of instances via the Eclipse extension mechanism
25   */
26  public class InstanceConfigurator {
27  
28  	public static final String IINSTANCE_CONFIG_ID = Activator.PLUGIN_ID + ".instanceConfig"; //$NON-NLS-1$
29  
30  	public static boolean onNodeModel = false;
31  
32  	/**
33  	 * Configure an instance with a given configurator
34  	 *
35  	 * @param useInstanceConfigurator
36  	 * @param instance
37  	 * @param componentPart
38  	 * @param containerContext
39  	 */
40  	public static void configureInstance(UseInstanceConfigurator useInstanceConfigurator, InstanceSpecification instance, Property componentPart, InstanceSpecification parentInstance) {
41  		if (useInstanceConfigurator != null) {
42  			org.eclipse.papyrus.designer.deployment.profile.Deployment.InstanceConfigurator instanceConfigurator =
43  					useInstanceConfigurator.getConfigurator();
44  			if (instanceConfigurator.isOnNodeModel() == onNodeModel) {
45  				if (instanceConfigurator != null) {
46  					String id = instanceConfigurator.getBase_Class().getName();
47  					IInstanceConfigurator iConfigurator = getInstanceConfigurator(id);
48  					if (iConfigurator != null) {
49  						iConfigurator.configureInstance(instance, componentPart, parentInstance);
50  
51  					}
52  				}
53  			}
54  		}
55  	}
56  
57  	protected static IInstanceConfigurator getInstanceConfigurator(String iConfiguratorID) {
58  		IExtensionRegistry reg = Platform.getExtensionRegistry();
59  		IConfigurationElement[] configElements = reg.getConfigurationElementsFor(IINSTANCE_CONFIG_ID);
60  		for (IConfigurationElement configElement : configElements) {
61  			try {
62  				final String iConfiguratorIDext = configElement.getAttribute("configuratorID"); //$NON-NLS-1$
63  				if (iConfiguratorIDext == null) {
64  					throw new RuntimeException(String.format(
65  							Messages.InstanceConfigurator_InvalidPluginExtension, iConfiguratorID));
66  				}
67  				if (iConfiguratorIDext.equals(iConfiguratorID)) {
68  					// TODO: cache returned instance (avoid creating a new instance each time => more efficient, no need for static attributes)
69  					final Object obj = configElement.createExecutableExtension("class"); //$NON-NLS-1$
70  					if (obj instanceof IInstanceConfigurator) {
71  						return (IInstanceConfigurator) obj;
72  					}
73  				}
74  			} catch (CoreException exception) {
75  				exception.printStackTrace();
76  			}
77  		}
78  		return null;
79  	}
80  }
