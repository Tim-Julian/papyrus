27  /**
28   * Panel displayed when a PrimitiveType is selected
29   * @deprecated This view will be replaced by an XWT view for the C++ profile
30   */
31  @Deprecated
32  public class CppPrimitiveTypePanel extends CppAbstractPanel {
33  
34  	// document used by the viewer for definition of the type
35  	@SuppressWarnings("unused")
36  	private SourceViewer typeViewer;
37  
38  	private Group typeGroup;
39  
40  	private IDocument typeDocument;
41  
42  	// Current selection
43  	private PrimitiveType selectedPType;
44  
45  	public CppPrimitiveTypePanel(Composite parent, int style) {
46  		super(parent, style);
47  	}
48  
49  	/*
50  	 * (non-Javadoc)
51  	 *
52  	 * @see com.cea.accordcpp.core.ui.panels.AccordCppAbstractPanel#getSelectedElement()
53  	 */
54  	@Override
55  	public PrimitiveType getSelectedElement() {
56  		return selectedPType;
57  	}
58  
59  	/*
60  	 * (non-Javadoc)
61  	 *
62  	 * @see com.cea.accordcpp.core.ui.panels.AccordCppAbstractPanel#setSelectedElement(java.lang.Element)
63  	 */
64  	@Override
65  	public void setSelectedElement(Element newElement) {
66  		super.setSelectedElement(newElement);
67  		if (newElement instanceof PrimitiveType) {
68  			this.selectedPType = (PrimitiveType) newElement;
69  		}
70  		else {
71  			throw new RuntimeException("bad selection: " + newElement + " should be an uml2 PrimitiveType");
72  		}
73  	}
74  
75  	/*
76  	 * (non-Javadoc)
77  	 *
78  	 * @see com.cea.accorduml.ui.views.panels.AccordUMLAbstractPanel#createContentHI()
79  	 */
80  	@Override
81  	public Control createContent() {
82  
83  		// /////////////////////////////////////////////////////////////////////
84  		// Create save reset buttons with superclass method
85  		// /////////////////////////////////////////////////////////////////////
86  		createSaveResetButtons();
87  
88  		// /////////////////////////////////////////////////////////////////////
89  		// Type definition text area
90  		// /////////////////////////////////////////////////////////////////////
91  
92  		typeDocument = createDocumentC();
93  		typeGroup = createGroup(this, "Type definition", buttonSave, null, true, 0, 0, true);
94  		// Use CDT CEditor coloration
95  		typeViewer = createViewerC(typeDocument, typeGroup);
96  
97  		// /////////////////////////////////////////////////////////////////////
98  
99  		return this;
100 	}
101 
102 	/**
103 	 * Saves the include declarations for a '<code>PrimitiveType</code>'
104 	 */
105 	@Override
106 	public void save() {
107 		if (selectedPType == null) {
108 			/* Log.debug("saveBody : selectedOperation is null"); */
109 		}
110 		else {
111 			CommandSupport.exec("C++ primitive type save", new Runnable() {
112 
113 				@Override
114 				public void run()
115 				{
116 					if (typeDocument.get().equals("")) {
117 						StereotypeUtil.unapply(selectedPType, Typedef.class);
118 					}
119 					else {
120 						Typedef cppType = StereotypeUtil.applyApp(selectedPType, Typedef.class);
121 						cppType.setDefinition(typeDocument.get());
122 					}
123 				}
124 			});
125 		}
126 	}
127 
128 
129 	/*
130 	 * (non-Javadoc)
131 	 *
132 	 * @see com.cea.accordcpp.core.ui.panels.AccordCppAbstractPanel#refreshPanel()
133 	 */
134 	@Override
135 	protected void refreshPanel() {
136 		if (selectedPType != null) {
137 			Typedef cppType = UMLUtil.getStereotypeApplication(selectedPType, Typedef.class);
138 			typeDocument.set(cppType != null ? cppType.getDefinition() : "");
139 		}
140 	}
141 
142 	/*
143 	 * (non-Javadoc)
144 	 *
145 	 * @see com.cea.accorduml.ui.views.panels.AccordUMLAbstractPanel#entryAction()
146 	 */
147 	@Override
148 	public void entryAction() {
149 		super.entryAction();
150 		reset();
151 	}
152 
153 	/*
154 	 * (non-Javadoc)
155 	 *
156 	 * @see com.cea.accordcpp.core.ui.panels.AccordCppAbstractPanel#checkModifications()
157 	 */
158 	@Override
159 	public boolean checkModifications() {
160 		Typedef cppType = UMLUtil.getStereotypeApplication(selectedPType, Typedef.class);
161 		String typeInModel = cppType != null ? cppType.getDefinition() : "";
162 		if (!typeDocument.get().equals(typeInModel)) {
163 			return true;
164 		}
165 		return false;
166 	}
167 }
