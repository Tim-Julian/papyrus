15  package org.eclipse.papyrus.designer.transformation.languages.cpp.library.statemachine;
16  
17  import org.eclipse.papyrus.designer.languages.cpp.library.CppUriConstants;
18  import org.eclipse.papyrus.designer.transformation.base.utils.StdModelLibs;
19  import org.eclipse.papyrus.designer.transformation.core.m2minterfaces.IM2MTrafoElem;
20  import org.eclipse.papyrus.designer.transformation.core.transformations.TransformationContext;
21  import org.eclipse.papyrus.designer.transformation.profile.Transformation.M2MTrafo;
22  import org.eclipse.uml2.uml.Class;
23  import org.eclipse.uml2.uml.Element;
24  import org.eclipse.uml2.uml.NamedElement;
25  import org.eclipse.uml2.uml.Package;
26  import org.eclipse.uml2.uml.StateMachine;
27  import org.eclipse.uml2.uml.Type;
28  
29  public class CompleteStatemachine implements IM2MTrafoElem {
30  
31  	public static final String THREAD_FUNCTIONS = "ThreadFunctions"; //$NON-NLS-1$
32  	public static final String SYSTEM_STATE_ENUM_T = "SystemStateEnum_t"; //$NON-NLS-1$
33      public static final String STRUCT_FOR_THREAD = "StructForThread_t"; //$NON-NLS-1$
34  
35  	@Override
36  	public void transformElement(M2MTrafo trafo, Element element) {
37  		if (element instanceof Class) {
38  			Class tmClass = (Class) element;
39  			if (tmClass.getClassifierBehavior() instanceof StateMachine) {
40  				StateMachine sm = (StateMachine) tmClass.getClassifierBehavior();
41  				SM2ClassesTransformationCore trafoCore = new SM2ClassesTransformationCore(TransformationContext.current.copier, sm, tmClass);
42  
43  				// TODO: move to C++ specific library
44  				StdModelLibs.addResource(CppUriConstants.PTHREAD_LIB, TransformationContext.current.copier.source);
45  
46  				Package smModel = (Package) trafo.getBase_Class().getPackage().getOwner();
47  				NamedElement stateEnum = smModel.getPackagedElement(SYSTEM_STATE_ENUM_T);
48  				NamedElement threadFuncEnum = smModel.getPackagedElement(THREAD_FUNCTIONS);
49  				Type structForThread = (Type) smModel.getPackagedElement(STRUCT_FOR_THREAD);
50  				trafoCore.setThreadStructType(structForThread);
51  				trafoCore.setSmPack(smModel);
52  				tmClass.createOwnedAttribute(SMCodeGeneratorConstants.SYSTEM_STATE_ATTR, (Type) stateEnum);
53  				tmClass.createDependency(threadFuncEnum);
54  				trafoCore.transform();
55  			}
56  		}
57  	}
58  }
