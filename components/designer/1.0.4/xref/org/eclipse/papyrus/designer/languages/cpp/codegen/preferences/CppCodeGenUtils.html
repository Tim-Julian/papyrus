12  package org.eclipse.papyrus.designer.languages.cpp.codegen.preferences;
13  
14  import org.eclipse.jface.preference.IPreferenceStore;
15  import org.eclipse.papyrus.designer.languages.cpp.codegen.Activator;
16  
17  /**
18   * Utility class that returns the preference values
19   *
20   * @author ansgar
21   *
22   */
23  public class CppCodeGenUtils {
24  
25  	protected static IPreferenceStore preferenceStore = null;
26  
27  	public static String getHeaderSuffix() {
28  		initPreferenceStore();
29  		return preferenceStore.getString(CppCodeGenConstants.P_HEADER_SUFFIX);
30  	}
31  
32  	public static String getBodySuffix() {
33  		initPreferenceStore();
34  		return preferenceStore.getString(CppCodeGenConstants.P_IMPLEM_SUFFIX);
35  	}
36  
37  	public static String getOutInoutOp() {
38  		initPreferenceStore();
39  		return preferenceStore.getString(CppCodeGenConstants.P_OUT_INOUT_OP);
40  	}
41  
42  	public static String getCommentHeader() {
43  		initPreferenceStore();
44  		return preferenceStore.getString(CppCodeGenConstants.P_COMMENT_HEADER);
45  	}
46  
47  	public static void initPreferenceStore() {
48  		if (preferenceStore == null) {
49  			preferenceStore = Activator.getDefault().getPreferenceStore();
50  		}
51  	}
52  
53  }
