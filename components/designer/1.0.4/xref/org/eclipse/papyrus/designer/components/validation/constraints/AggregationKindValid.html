9   import org.eclipse.core.runtime.IStatus;
10  import org.eclipse.emf.validation.AbstractModelConstraint;
11  import org.eclipse.emf.validation.IValidationContext;
12  import org.eclipse.papyrus.designer.deployment.profile.Deployment.ConfigurationProperty;

14  import org.eclipse.papyrus.uml.tools.utils.StereotypeUtil;

17  import org.eclipse.uml2.uml.Property;
18  import org.eclipse.uml2.uml.Type;

20  /**
21   * Verify that aggregation kinds are used well. General rule:
22   * attributes of a components should have aggregation kind "composite" unless
23   * (1) the are singletons (implementation property), i.e. there is exactly one instance.
24   *
25   * (2) they are *shared*. In this case an enclosing component has to define a property of same type
26   * and name (open if "same name" restriction is kept, since it implies an implicit relationship
27   * that would break after name changes. But it allows for having multiple shared properties of same type)
28   * [it does not make a difference, whether "aggregation" or "none" is used as aggregation kind]
29   * Alternative? give user a fine control in the instantiation model? Fractal: no need for a single
30   * composed property?
31   *
32   * Rule has been made specific to Qompass, since we verify that configurationProperty is applicable (implying that the
33   * FCM profile is applied)
34   *
35   * @author ansgar
36   *
37   */
38  public class AggregationKindValid extends AbstractModelConstraint {
39  
40  	@Override
41  	public IStatus validate(IValidationContext ctx)
42  	{
43  		Property attribute = (Property) ctx.getTarget();
44  		Type type = attribute.getType();
45  		if (type instanceof Class) {
46  			AggregationKind aggKind = attribute.getAggregation();
47  			if (aggKind != AggregationKind.COMPOSITE_LITERAL) {
48  				if (StereotypeUtil.isApplicable(attribute, ConfigurationProperty.class) && !DepUtils.isSingleton((Class) type)) {
49  					// no composition, but not singleton: warn user to make up a proper deployment plan
50  					// validation has been made Qompass specific by checking whether the Configuration attribute is applicable
51  
52  					return ctx.createFailureStatus("The attribute '" + attribute.getName() + "' has a non-composition aggregation kind, but the referenced type ('" +
53  							type.getName() + "') is not a singleton. The associated instance specifications have to be set-up manually");
54  				}
55  			}
56  		}
57  		return ctx.createSuccessStatus();
58  	}
59  }
