15  package org.eclipse.papyrus.designer.transformation.base.utils;
16  
17  import org.eclipse.emf.common.util.EList;
18  import org.eclipse.uml2.uml.NamedElement;
19  import org.eclipse.uml2.uml.Namespace;
20  import org.eclipse.uml2.uml.Package;
21  
22  public class CreationUtils {
23  
24  	/**
25  	 * This function returns a Package reference that corresponds to a qualified name.
26  	 * Packages are created, if not existing yet - the function is thus a bit similar to
27  	 * the unix mkdir -p command. Note that the main model within the list of name-spaces is
28  	 * ignored, in order to avoid that a sub-package within a model starts with the name
29  	 * of the model.
30  	 *
31  	 * Caveat: imported elements must be checked beforehand. In case of a package import,
32  	 * getMembers (and getImportedMembers) will not return the name of the imported model
33  	 * but the elements (directly) owned by the imported model.
34  	 * Example: getMember ("BasicTypes") will return null whereas getMember ("CORBA")
35  	 * returns a package (CORBA is a package within the model library BasicTypes)
36  	 *
37  	 * @param model
38  	 *            a model
39  	 * @param list
40  	 *            a list of namespace elements (as can be obtained via allNamespaces),
41  	 *            the top-level element is the last within the list.
42  	 * @param skipTop
43  	 *            if true, skip top level namespace element
44  	 * @return
45  	 */
46  	public static Package getAndCreate(Package root, EList<Namespace> list, boolean skipTop) {
47  		int offset = (skipTop ? 2 : 1);
48  		for (int i = list.size() - offset; i >= 0; i--) {
49  			Namespace ns = list.get(i);
50  
51  			NamedElement pkg = root.getOwnedMember(ns.getName());
52  			if (pkg == null) {
53  				// package does not exist => create it.
54  				pkg = root.createNestedPackage(ns.getName());
55  				CopyUtils.copyID(ns, pkg);
56  				// copy stereotype to create package
57  				StUtils.copyStereotypes(ns, pkg);
58  			}
59  			if (!(pkg instanceof Package)) {
60  				break;
61  			}
62  			root = (Package) pkg;
63  		}
64  		return root;
65  	}
66  
67  	public static Package getAndCreate(Package root, EList<Namespace> list) {
68  		return getAndCreate(root, list, true);
69  	}
70  
71  	public static Package getAndCreate(Package root, String name) {
72  		NamedElement pkg = root.getMember(name);
73  		if (pkg == null) {
74  			pkg = root.createNestedPackage(name);
75  		}
76  		return (Package) pkg;
77  	}
78  }
