7   package org.eclipse.papyrus.designer.transformation.core.transformations;
8   
9   import org.eclipse.emf.ecore.EObject;
10  import org.eclipse.emf.ecore.util.EcoreUtil.Copier;
11  import org.eclipse.uml2.uml.Element;
12  
13  /**
14   * This class enables the copy from a source to a given target element.
15   * It assumes that a copy has already been created, but will just not copy it...
16   * Unlike the @see Copier class, the element might be copied to a different place
17   */
18  public class CopyTo {
19  	/**
20  	 * Copy an element from the source to a target
21  	 *
22  	 * @param source
23  	 *            The element that should be copied
24  	 * @param targetContainer
25  	 *            The container into which it the source should be copied
26  	 * @return the copied object
27  	 */
28  	@SuppressWarnings("unchecked")
29  	public static <T extends Element> T copyTo(T source, Element targetContainer) {
30  		Copier copier = new Copier();
31  		EObject sourceContainer = source.eContainer();
32  		if (sourceContainer != null) {
33  			copier.put(sourceContainer, targetContainer);
34  		}
35  
36  		EObject copy = copier.copy(source);
37  		copier.copyAll(source.getStereotypeApplications());
38  		copier.copyReferences();
39  		return (T) copy;
40  	}
41  }
