30  /**
31   * Instance of this class is used to resolve uml::Type.
32   * This class allows to find a Type qualified name from its Name (simple or qualified) used in the Java file.
33   * 
34   * 
35   * @author cedric dumoulin
36   *
37   * @since 0.7.2
38   */
39  public class TypeResolver {
40  
41  	protected ClassifierCatalog classifierCatalog;
42  	protected ImportedTypeCatalog importedTypeCatalog;
43  	protected CreationPackageCatalog creationPackageCatalog;
44  	
45  	/**
46  	 * Constructor.
47  	 *
48  	 * @param classifierCatalog
49  	 * @param importedTypeCatalog
50  	 */
51  	public TypeResolver(ClassifierCatalog classifierCatalog, ImportedTypeCatalog importedTypeCatalog, CreationPackageCatalog creationPackageCatalog) {
52  		this.classifierCatalog = classifierCatalog;
53  		this.importedTypeCatalog = importedTypeCatalog;
54  		this.creationPackageCatalog = creationPackageCatalog;
55  	}

57  	/**
58  	 * Lookup for the {@link Classifier} denoted byt the provided name. The name is either a simple name (no dot) or a 
59  	 * qualified name (with dot).
60  	 * If found, the Classifier is returned. If not found, appropriate exception is thrown :
61  	 * <ul>
62  	 *   <li></li>
63  	 *   <li></li>
64  	 *   <li></li>
65  	 * </ul>
66  	 * 
67  	 * @param name Name of the classifier to search for. Can be a simple name or a qualified name.
68  	 * @param context
69  	 * @return the corresponding Classifer.
70  	 * 
71  	 * @throws ClassifierNotFoundException 
72  	 * @throws NestedClassifierNeedCreationException 
73  	 * @throws ClassifierNeedCreationException 
74  	 */
75  	public Classifier lookupClassifier( String name, LocalContext context ) throws ClassifierNotFoundException, ClassifierNeedCreationException, NestedClassifierNeedCreationException {
76  		
77  		// Search locally
78  		Classifier result = lookupInImportsAndContext( name, context );
79  		return result;
80  	}
81  
82  	/**
83  	 * Search for a Classifier by its Name. 
84  	 * Search is done 'locally' : using the 'import', and the context.
85  	 * 
86  	 * 
87  	 * @param name Name of the classifier to search for
88  	 * @return The requested Classifier.
89  	 * @throws ClassifierNotFoundException 
90  	 * @throws ClassifierNeedCreationException 
91  	 * @throws NestedClassifierNeedCreationException 
92  	 */
93  	private Classifier lookupInImportsAndContext(String name, LocalContext context) throws ClassifierNotFoundException, ClassifierNeedCreationException, NestedClassifierNeedCreationException {
94  
95  		if( isSimpleName(name) ) {
96  			return lookupClassifierFromSimpleName( name, context );
97  		}
98  		else {
99  			List<String> qualifiedName = UmlUtils.toQualifiedName(name);
100 
101 			return lookupClassifierFromTypename(qualifiedName, context );
102 		}
103 	}

105 	/**
106 	 * Lookup for the classifier from a simple name.
107 	 * <ul>
108 	 *   <li></li>
109 	 *   <li></li>
110 	 *   <li></li>
111 	 * </ul>
112 	 * 
113 	 * @param sname
114 	 * @param context
115 	 * @return The requested {@link Classifier}
116 	 * @throws ClassifierNotFoundException A valid qualified name is found (in imports), but no classifier is found in classifier catalog. The exception contains the qname.
117 	 * @throws ClassifierNeedCreationException A qualified name is found for the sname, but the corresponding Classifier is not found in the classifierCatalog.
118 	 * @throws NotFoundException No Classifier can be found for the sname.
119 	 */
120 	private Classifier lookupClassifierFromSimpleName(String sname, LocalContext context) throws ClassifierNotFoundException, ClassifierNeedCreationException {
121 		
122 		Classifier result;
123 		// First, check if the sname is declared in imports
124 		try {
125 			List<String> qualifiedName = importedTypeCatalog.getImportQualifiedNameChecked(sname);
126 			result =  classifierCatalog.getClassifier(qualifiedName);
127 			if( result != null ) {
128 				return result;
129 			}
130 			// QName found, but the classifier not exist in catalog
131 			throw new ClassifierNeedCreationException(qualifiedName);
132 		} catch (ImportNotFoundException e) {
133 			// silently fails;
134 		}
135 		
136 		// check in '*' imports
137 		// Only check if the requested classifier exist in UML model.
138 		for( List<String> parentPackageQName : importedTypeCatalog.getStarImports() ) {
139 			// Build a possible Fully qualified name
140 			List<String> fullQName = new ArrayList<String>(parentPackageQName);
141 			fullQName.add(sname);
142 			
143 			result = classifierCatalog.getClassifier(fullQName);
144 			if( result != null) {
145 				return result;
146 			}
147 		}
148 		
149 		// Still not found
150 		// Check in context
151 		// Check if a corresponding type can be found in context.
152 		result = context.lookupClassifier(sname, UmlUtils.CLASSIFIER_TYPE);
153 		if( result != null) {
154 			return result;
155 		}
156 		
157 		// Check in classifier catalog : maybe the sname is a Class in root packages ?
158 		result = classifierCatalog.getClassifier(sname);
159 		if( result != null) {
160 			return result;
161 		}
162 		
163 		// Still not found
164 		throw new ClassifierNotFoundException(sname);
165 		
166 	}

168 	/**
169 	 * Lookup for the Classifier from a typename. A typename is a qualified name where the first name denote a type.
170 	 * (ex : Typename.A);
171 	 * Check if the first Typename (first part of the qname) can be found in import, '*' and then Context.
172 	 * If a corresponding Classifier is found, try to get the requested Classifier from this found Classifier.
173 	 * 
174 	 * @param qualifiedName
175 	 * @param context
176 	 * @throws NestedClassifierNeedCreationException The first segment of the qualifiedName denote a Classifier. This classifier, or its name has been found, but the nested classifier does not exist. 
177 	 * @throws ClassifierNotFoundException No Classifier found for the requested name.
178 	 */
179 	private Classifier lookupClassifierFromTypename(List<String> qualifiedName, LocalContext context) throws NestedClassifierNeedCreationException, ClassifierNotFoundException {
180 	
181 		String typeName = qualifiedName.get(0);
182 		Classifier result;
183 		Classifier typenameClassifier;
184 		
185 		// Search the Classifier corresponding to the Typename
186 		try {
187 			typenameClassifier = lookupInImportsAndContext(typeName, context);
188 			// Found. Now, try to get the requested Classifier from this typenameClassifier
189 			List<String> searchedQualifiedName = qualifiedName.subList(1, qualifiedName.size());
190 			result = UmlUtils.lookupNestedClassifier(typenameClassifier, qualifiedName, UmlUtils.CLASSIFIER_TYPE);
191 		} catch (ClassifierNeedCreationException e) {
192 			// We have the qualified name of the parent classifier, but this later need to be created.
193 			throw new NestedClassifierNeedCreationException(e.getQualifiedName(),  qualifiedName);
194 		} catch (ClassifierNotFoundException e) {
195 			// Find not typename for the qualified name.
196 			// Try to use the qname as a fully qualified name
197 			result = classifierCatalog.getClassifier(qualifiedName);
198 			if( result != null) {
199 				return result;
200 			}
201 			// Still not found
202 			throw new ClassifierNotFoundException( qualifiedName );
203 		} 
204 		
205 		// Classifier found ?
206 		if( result != null) {
207 			return result;
208 		}
209 		// Not exist !
210 		// We found the parentClassifier, but no requested classifier inside.
211 		throw new NestedClassifierNeedCreationException(typenameClassifier,  qualifiedName);
212 
213 	}

215 	/**
216 	 * Return true if the name is a simple name (no dot), false otehrwise.
217 	 * 
218 	 * @param name The name to check
219 	 * @return true if the name is a simple name, false otherwise.
220 	 * 
221 	 */
222 	private boolean isSimpleName(String name) {
223 		return (name.indexOf('.')==-1);
224 	}

226 	/**
227 	 * Get the type corresponding to the type name. Lookup the type, and create it if not found.
228 	 * Lookup is done by using {@link #lookupClassifier(String, LocalContext)};
229 	 * Creation is done in the namespace provided by {@link CreationPackageCatalog}.
230 	 * 
231 	 * @param typename Name of the classifier to search for. Can be a simple name or a qualified name.
232 	 * @param context
233 	 * @return
234 	 */
235 	public Type getType(String typename, LocalContext context) {
236 		
237 		Type result;
238 		// First, try to found the type
239 		try {
240 			result = lookupClassifier(typename, context);
241 			return result;
242 		} catch (ClassifierNotFoundException e) {
243 			List<String> qualifiedName;
244 			// Try to create the classifier
245 			if( e.getRequestedSimpleName() != null) {
246 				// Simple name ==> create it in the current package from context
247 				qualifiedName = UmlUtils.toQualifiedName(e.getRequestedSimpleName());
248 				Package creationPackage = context.getCurrentPackage();
249 				EClass requestedType = creationPackageCatalog.guessBestType( qualifiedName );
250 				result = UmlUtils.getClassifier(creationPackage, qualifiedName, requestedType);
251 			}
252 			else {
253 				// Create using the qualified name.
254 				qualifiedName = e.getRequestedQualifiedName();
255 				Package creationPackage = creationPackageCatalog.getCreationPackage(qualifiedName, null);
256 				EClass requestedType = creationPackageCatalog.guessBestType( qualifiedName );
257 				result = UmlUtils.getClassifier(creationPackage, qualifiedName, requestedType);
258 			}
259 			
260 			return result;
261 			
262 		} catch (ClassifierNeedCreationException e) {
263 			// Try to create the classifier
264 			List<String> qualifiedName = e.getQualifiedName();
265 			Package creationPackage = creationPackageCatalog.getCreationPackage(qualifiedName, null);
266 			EClass requestedType = creationPackageCatalog.guessBestType( qualifiedName );
267 			result = UmlUtils.getClassifier(creationPackage, qualifiedName, requestedType);
268 
269 			return result;
270 			
271 		} catch (NestedClassifierNeedCreationException e) {
272 			// Create the nested classifier
273 			EClass requestedType = creationPackageCatalog.guessBestType( e.getRequestedNestedClassifierRelativename() );
274 			if( e.getParentClassifier() != null) {
275 				// Parent is known, create the nested classifier inside it.
276 			   result = UmlUtils.getNestedClassifier(e.getParentClassifier(), e.getRequestedNestedClassifierRelativename(), requestedType);
277 			}
278 			else {
279 				// Create parent first
280 				List<String> parentQualifiedName = e.getParentQualifiedname();
281 				Package creationPackage = creationPackageCatalog.getCreationPackage(parentQualifiedName, null);
282 				EClass parentRequestedType = creationPackageCatalog.guessBestType( parentQualifiedName );
283 				Classifier parent = UmlUtils.getClassifier(creationPackage, parentQualifiedName, parentRequestedType);
284 				
285 			    result = UmlUtils.getNestedClassifier(parent, e.getRequestedNestedClassifierRelativename(), requestedType);
286 			}
287 			
288 			return result;
289 		}
290 	}
