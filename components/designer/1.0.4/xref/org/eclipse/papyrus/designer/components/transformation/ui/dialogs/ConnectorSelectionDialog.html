95  	@Override
96  	public Control createDialogArea(Composite parent) {
97  		Composite contents = (Composite) super.createDialogArea(parent);
98  		createMessageArea(contents);
99  		// createLabel(contents, "Connector type");
100 		GridData gridData = DialogUtils.createFillGridData();
101 		gridData.heightHint = 200;
102 
103 		Group fUpperGroup = new Group(contents, SWT.NONE);
104 		fUpperGroup.setLayout(new GridLayout());
105 		fUpperGroup.setLayoutData(gridData);
106 		fUpperGroup.setText(" Abstract interaction components "); //$NON-NLS-1$
107 		createFilterText(fUpperGroup);
108 		fUpper = createFilteredList(fUpperGroup);
109 
110 		final Group fLowerGroup = new Group(contents, SWT.NONE);
111 		fLowerGroup.setLayout(new GridLayout());
112 		fLowerGroup.setLayoutData(gridData);
113 		fLowerGroup.setText(" Realizations/groups of interaction components "); //$NON-NLS-1$
114 		// createLabel(contents, lowerLabel);
115 		fLower = createFilteredList(fLowerGroup);
116 		fFilteredList = fUpper;
117 		org.eclipse.papyrus.designer.components.FCM.Connector fcmConnector = UMLUtil.getStereotypeApplication(m_selectedConnector, org.eclipse.papyrus.designer.components.FCM.Connector.class);
118 		if (fcmConnector != null) {
119 			// getInitialElementSelections();
120 			initialSelection = fcmConnector.getIc();
121 		} else {
122 			initialSelection = null;
123 		}
124 
125 		fTypeOnly = new Button(contents, SWT.CHECK);
126 		fTypeOnly.setText("Select type only"); //$NON-NLS-1$
127 		SelectionListener fTypeOnlySL = new SelectionListener() {
128 
129 			@Override
130 			public void widgetSelected(SelectionEvent e) {
131 				if (fTypeOnly.getSelection()) {
132 					fLower.setEnabled(false);
133 					fLowerGroup.setEnabled(false);
134 					fLower.setSelection(new int[0]);
135 					fLower.setElements(null);
136 				}
137 				else {
138 					fLower.setEnabled(true);
139 					fLowerGroup.setEnabled(true);
140 					handleSelectionChanged();
141 				}
142 			}
143 
144 			@Override
145 			public void widgetDefaultSelected(SelectionEvent e) {
146 			}
147 		};
148 		fTypeOnly.addSelectionListener(fTypeOnlySL);
149 
150 		fUpper.setElements(connectorList.toArray());
151 
152 		// create initial selections
153 		if (initialSelection instanceof InteractionComponent) {
154 			InteractionComponent componentComp = initialSelection;
155 			Class componentCompBase = componentComp.getBase_Class();
156 			if (!componentCompBase.isAbstract()) {
157 				// get first superclass that is a component type
158 				Class componentTypeBase = ElementUtils.componentType(componentCompBase);
159 				componentComp = UMLUtil.getStereotypeApplication(componentTypeBase, InteractionComponent.class);
160 			} else {
161 				// fLower.setEnabled (false);
162 				// fLowerGroup.setEnabled(false);
163 				fTypeOnly.setSelection(true);
164 			}
165 			// update status (listener does not evaluate passed data)
166 			fTypeOnlySL.widgetSelected(null);
167 
168 			// set initial selections
169 			fUpper.setSelection(new Object[] { componentComp.getBase_Class() });
170 		}
171 
172 		Label info = createMessageArea(contents);
173 		info.setText("Information about connector implementation:"); //$NON-NLS-1$
174 		// m_description = createMessageArea(contents);
175 
176 		GridData descriptionGD = new GridData();
177 		descriptionGD.horizontalAlignment = GridData.FILL;
178 		descriptionGD.grabExcessHorizontalSpace = true;
179 		descriptionGD.heightHint = 80;
180 		descriptionGD.grabExcessVerticalSpace = true;
181 		descriptionGD.verticalAlignment = GridData.FILL;
182 
183 		m_description = new Text(contents, SWT.NONE | SWT.WRAP | SWT.V_SCROLL | SWT.READ_ONLY);
184 		m_description.setLayoutData(descriptionGD);
185 
186 		fLower.addSelectionListener(new SelectionListener() {
187 
188 			@Override
189 			public void widgetSelected(SelectionEvent e) {
190 				Object[] selected = fLower.getSelection();
191 				if ((selected.length > 0) && (selected[0] instanceof NamedElement)) {
192 					NamedElement selectedNE = (NamedElement) selected[0];
193 					m_description.setText(DescriptionUtils.getDescription(selectedNE));

195 				else {
196 					m_description.setText("");
197 					updateOkState();
198 				}
199 			}
200 
201 			@Override
202 			public void widgetDefaultSelected(SelectionEvent e) {
203 			}
204 		});
205 
206 		return contents;
207 	}

275 	private void handleUpperSelectionChanged() {
276 		int index = getSelectionIndex();
277 		if (index >= 0) {
278 			fUpperElements = getFoldedElements(index);
279 
280 			if ((fUpperElements == null) || fTypeOnly.getSelection()) {
281 				// lower list remains empty, if typeOnly is true
282 				fLower.setElements(new Object[] {});
283 			} else {
284 				fLower.setElements(getLowerList(fUpperElements[0]));
285 				if (initialSelection instanceof InteractionComponent) {
286 					// select existing selection in lower section
287 					fLower.setSelection(new Object[] { initialSelection });
288 					// do not try to set it again.
289 					initialSelection = null;
290 				}
291 			}
292 		}
293 		validateCurrentSelection();
294 	}

296 	/**
297 	 * Returns the selected element from the upper or lower pane.
298 	 *
299 	 * @return Object
300 	 */
301 	protected Object getSelectedElement(boolean upper) {
302 		Object[] selection;
303 		if (upper) {
304 			selection = fFilteredList.getSelection();
305 		} else {
306 			selection = fLower.getSelection();
307 		}
308 		if (selection.length > 0) {
309 			return selection[0];
310 		}
311 		return null;
312 	}

314 	protected Object[] getLowerList(Object selectedUpperObj) {
315 		EList<Class> connectorList = new UniqueEList<Class>();
316 		if (selectedUpperObj instanceof Class) {
317 			visitedPackages = new BasicEList<Package>();
318 			getImplGroups(m_model, (Class) selectedUpperObj, connectorList);
319 		}
320 		return connectorList.toArray();
321 	}

323 	void getAllConnectors(Package pkg, Feature selectedConnector, EList<Class> connectorList) {
324 		for (Element el : pkg.getMembers()) {
325 			if (el instanceof Package) {
326 				if (!visitedPackages.contains(el)) {
327 					visitedPackages.add((Package) el);
328 					getAllConnectors((Package) el, selectedConnector, connectorList);
329 				}
330 			} else if (el instanceof Class) {
331 				if (StereotypeUtil.isApplied(el, InteractionComponent.class) && ((Class) el).isAbstract()) {
332 
333 					if (selectedConnector == null) {
334 						connectorList.add((Class) el);
335 					} else {
336 						try {
337 							Element owner = selectedConnector.getOwner();
338 							if (owner instanceof Class) {
339 								Class composite = (Class) owner;
340 								Activator.log.info("ConnectorSelectionDialog.getAllConnectors: try to bind connector " + //$NON-NLS-1$
341 										((Class) el).getQualifiedName());
342 								ConnectorBinding.obtainBinding(composite, selectedConnector, (Class) el, false);
343 								connectorList.add((Class) el);
344 							} else {
345 								connectorList.add((Class) el);
346 							}
347 						} catch (TransformationException e) {
348 							// silently ignore exception: it is normal that we cannot find a binding for some connectors
349 						}
350 					}
351 				}
352 			}
353 		}
354 	}
355 
356 	void getImplGroups(Package pkg, Class selectedConnType, EList<Class> implGroupList) {
357 		for (Element el : pkg.getMembers()) {
358 			if (el instanceof Package) {
359 				if (!visitedPackages.contains(el)) {
360 					visitedPackages.add((Package) el);
361 					getImplGroups((Package) el, selectedConnType, implGroupList);
362 				}
363 			} else if (el instanceof Class) {
364 				Class class_ = (Class) el;
365 				if (class_.getGeneralization(selectedConnType) != null) {
366 					implGroupList.add((Class) el);
367 				}
368 			}
369 		}
