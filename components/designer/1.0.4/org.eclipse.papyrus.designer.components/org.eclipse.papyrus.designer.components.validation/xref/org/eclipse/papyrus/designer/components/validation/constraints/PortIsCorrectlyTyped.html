15  package org.eclipse.papyrus.designer.components.validation.constraints;
16  
17  import org.eclipse.core.runtime.IStatus;
18  import org.eclipse.emf.validation.AbstractModelConstraint;
19  import org.eclipse.emf.validation.IValidationContext;
20  import org.eclipse.uml2.uml.Class;
21  import org.eclipse.uml2.uml.Port;
22  
23  /**
24   * Check whether the port is not correctly typed, e.g. the port kind ProvideInterface is used in
25   * conjunction with a data type.
26   * Technically, the criterion will be that not both, derived provided and derived required interfaces
27   * are empty.
28   *
29   */
30  public class PortIsCorrectlyTyped extends AbstractModelConstraint {
31  
32  	@Override
33  	public IStatus validate(IValidationContext ctx)
34  	{
35  		Port port = (Port) ctx.getTarget();
36  
37  		if ((port.getProvideds().size() == 0) && (port.getRequireds().size() == 0)) {
38  			Class class_ = port.getClass_();
39  			return ctx.createFailureStatus("The port '" + port.getName() + "' owned by class '" + class_.getQualifiedName() + //$NON-NLS-1$ //$NON-NLS-2$
40  					"' is not correctly typed, since derived provided and required interface is null"); //$NON-NLS-1$
41  		}
42  		return ctx.createSuccessStatus();
43  
44  	}
45  }
