17    /**
18     * return the operator for (de-) referencing a part. If the part is
19     * instantiate via the bootloader, it becomes a pointer. If it is
20     * instantiated by the composite itself, it is not a pointer, it will be
21     * instantiated along with the composite
22     * 
23     * @param part
24     * @return
25     */
26    public static String refOp(final Property part) {
27      String _xifexpression = null;
28      boolean _or = false;
29      AggregationKind _aggregation = part.getAggregation();
30      boolean _equals = Objects.equal(_aggregation, AggregationKind.SHARED_LITERAL);
31      if (_equals) {
32        _or = true;
33      } else {
34        boolean _isApplied = StereotypeUtil.isApplied(part, Ptr.class);
35        _or = _isApplied;
36      }
37      if (_or) {
38        _xifexpression = "->";
39      } else {
40        _xifexpression = ".";
41      }
42      return _xifexpression;
43    }
