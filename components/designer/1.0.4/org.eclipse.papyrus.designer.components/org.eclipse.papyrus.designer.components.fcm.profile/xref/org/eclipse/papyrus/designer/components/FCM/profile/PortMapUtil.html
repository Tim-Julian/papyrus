30  /**
31   * This class is responsible for creating derived types associated with ports. The derived
32   * type is the class that will type the port and contain imported and used interfaces depending
33   * on the FCM type and kind.
34   * A particular question related to the derived types is into which package we can place them
35   * within the type hierarchy:
36   * (1) Of course, the package must be writable. That means, it must not belong to an imported library.
37   * (2) The source type must be clearly identifiable.
38   * (3) If a template get's instantiated, derived types within (if we place them there) might
39   * need (re-) instantiation. Imagine AMI_ interface that contains some of I's method. The creation of an AMI_I type within the template
40   * 
41   * Therefore, a global "derived types" package is created within the model that owns the port. This
42   * package contains the full folder hierarchy of the original type and places the elements there.
43   * TODO: Align with package template instantiation => create type specific sub-folder, e.g. kind_type. Use package specific side-folders.
44   * Need specific solution for elements within template (placing into same template is finally not a good idea).
45   * Idea of package template (and the possibility to extend) was, that common elements are instantiated once. Thus, we will have interfaceBased_<intfName>/UseInterface/Use.cpp,
46   * not UseInterface_<intfName>/Use.cpp
47   * [Users choice to define template with additional package or not?]
48   */

165 	/**b
166 	 * Calculate derived required interface in function of port type and kind
167 	 *
168 	 * @param port
169 	 *            the port, for which the calculation should be done
170 	 * @return
171 	 */
172 	public static Type getDerivedType(final Port port) {
173 		if (port.getBase_Port() == null) {
174 			// should not happen, but can occur in case of corrupted XMI files
175 			return null;
176 		}
177 		PortKind portKind = port.getKind();
178 		if (portKind != null) {
179 			String ruleName = portKind.isExtendedPort() ? "ExtendedPort" : portKind.getBase_Class().getName(); //$NON-NLS-1$
180 			final IMappingRule mappingRule = getMappingRule(ruleName);
181 			if (mappingRule != null) {
182 				return mappingRule.calcDerivedType(port, false);
183 			}
184 		}
185 		return null;
186 	}

189 	/**
190 	 * Obtain the mapping rule for a port when the name of the portKind is given
191 	 *
192 	 * @param portKindName
193 	 *            the name of the port-kind
194 	 *
195 	 * @return the mapping rule or null, if no rule could be found
196 	 */
197 	public static IMappingRule getMappingRule(String portKindName) {
198 		IExtensionRegistry reg = Platform.getExtensionRegistry();
199 		IConfigurationElement[] configElements = reg.getConfigurationElementsFor(Activator.PLUGIN_ID + "." + MAPPING_RULE_ID); //$NON-NLS-1$
200 		for (IConfigurationElement configElement : configElements) {
201 			try {
202 				final String extPortKindName = configElement.getAttribute("portKindName"); //$NON-NLS-1$
203 				if (extPortKindName.equals(portKindName)) {
204 					final Object obj = configElement.createExecutableExtension("class"); //$NON-NLS-1$
205 					if (obj instanceof IMappingRule) {
206 						return (IMappingRule) obj;
207 					}
208 				}
209 			} catch (CoreException exception) {
210 				exception.printStackTrace();
211 			}
212 		}
213 		return null;
214 	}

216 	/**
217 	 * Get the mapping rule for a port
218 	 *
219 	 * @param port
220 	 *            the FCM port
221 	 * @return the mapping rule or null, if no rule could be found
222 	 */
223 	public static IMappingRule getMappingRule(final Port port) {
224 		if (port.getBase_Port() == null) {
225 			// should not happen, but can occur in case of corrupted XMI files
226 			return null;
227 		}
228 		PortKind portKind = port.getKind();
229 		if (portKind == null) {
230 			return null;
231 		}
232 		if (portKind.getBase_Class() != null) {
233 			String ruleName = portKind.isExtendedPort() ? "ExtendedPort" : portKind.getBase_Class().getName(); //$NON-NLS-1$
234 			if (port instanceof TemplatePort) {
235 				ruleName = "TemplatePort"; //$NON-NLS-1$
236 			}
237 			return getMappingRule(ruleName);
238 		}
239 		return null;
240 	}

242 	public static PortKind getBoundType(final Port port) {
243 		if (port.getBase_Port() == null) {
244 			// should not happen, but can occur in case of corrupted XMI files
245 			return null;
246 		} else {
247 			String ruleName = "TemplatePort"; //$NON-NLS-1$
248 			final IMappingRule mappingRule = getMappingRule(ruleName);
249 			if (mappingRule instanceof ITemplateMappingRule) {
250 				return ((ITemplateMappingRule) mappingRule).getBoundType(port);
251 			}
252 		}
253 		return null;
254 	}

256 	/**
257 	 * Update the derived interfaces of a port this operation needs to be
258 	 * called in the context of an update command (transaction).
259 	 *
260 	 * @param port
261 	 */
262 	public static void update(final Port port) {
263 		final IMappingRule mappingRule = getMappingRule(port);
264 
265 		if (mappingRule != null) {
266 			Type type = mappingRule.calcDerivedType(port, true);
267 			port.getBase_Port().setType(type);
268 			if (mappingRule instanceof ITemplateMappingRule) {
269 				((ITemplateMappingRule) mappingRule).updateBinding(port);
270 			}
271 		}
272 	}

274 	/**
275 	 * Check whether a port needs to be updated, since its derived interfaces
276 	 * are either not existing or out of date.
277 	 *
278 	 * @param port
279 	 * @return
280 	 */
281 	public static boolean needsUpdate(final Port port) {
282 		final IMappingRule mappingRule = getMappingRule(port);
283 		if (mappingRule != null) {
284 			return mappingRule.needsUpdate(port);
285 		}
286 		return false;
287 	}
