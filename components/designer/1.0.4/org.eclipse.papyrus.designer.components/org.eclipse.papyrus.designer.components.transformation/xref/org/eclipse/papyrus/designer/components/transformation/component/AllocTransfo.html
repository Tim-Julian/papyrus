27  /**
28   * Transform the allocations towards an execution resource (thread) into a suitable connection
29   * Current restriction: a thread needs to be defined the same composite as the component which
30   * is allocated to this thread.
31   */
32  public class AllocTransfo {
33  	/**
34  	 * Name of port provided by component being allocation target
35  	*/
36  	public static final String RUN = "run"; //$NON-NLS-1$
37  	
38  	/**
39  	 * Name of port provided by thread component (a "required" port)
40  	 */
41  	public static final String R_RUN = "rRun"; //$NON-NLS-1$
42  	
43  	/**
44  	 * name of main thread
45  	 */
46  	public static final String MAIN = "main"; //$NON-NLS-1$
47  
48  	/**
49  	 *
50  	 * @param copier
51  	 *            The copier class. Currently unused
52  	 * @param composite
53  	 *            A composite. We analyze the allocations of the properties within this
54  	 *            Composite. The composite must be an element of the target model that can be modified.
55  	 */
56  	public void transformAllocs(LazyCopier copy, Class composite) {
57  		EList<Property> attributeList = new BasicEList<Property>();
58  		attributeList.addAll(composite.getOwnedAttributes()); // use a copy, since the loop eventually destroys an element of the list.
59  		for (Property attribute : attributeList) {
60  			Type type = attribute.getType();
61  			if (!(type instanceof Class)) {
62  				continue;
63  			}
64  			Property thread = AllocUtils.getThreadAlloc(attribute);
65  			if (thread != null) {
66  				// this attribute represents a thread.
67  				if (thread.getName().equals(MAIN)) {
68  					// TODO: currently, no support for setting up the priority of the main thread.
69  					// we need to get a cleaner solution for the main thread.
70  					// destroy eventual connector and associated thread => thread as component with unconnected start thread
71  					Connector conn = ConnectorUtil.getConnector(composite, attribute, thread);
72  					// Connector targetConn = copier.getCopy(conn);
73  					// Property targetThread = copier.getCopy(thread);
74  					if (conn != null) {
75  						conn.destroy();
76  					}
77  					// destroy not only the thread, but also the slot within the composites instance specification
78  					// (that references the thread via its defining feature).
79  					for (Setting setting : UML2Util.getInverseReferences(thread)) {
80  						if (setting.getEObject() instanceof Slot) {
81  							((Element) setting.getEObject()).destroy();
82  						}
83  					}
84  					thread.destroy();
85  				}
86  				// check, if there is already a connector between the attribute and the thread
87  				else if (!ConnectorUtil.existsConnector(composite, attribute, thread)) {
88  					// no connector => create;
89  					Class component = (Class) type;
90  					Connector c = composite.createOwnedConnector("alloc to " + thread.getName()); //$NON-NLS-1$
91  					ConnectorEnd ce1 = c.createEnd();
92  					ce1.setPartWithPort(attribute);
93  					// TODO: should really use start port instead...
94  					// ce1.setRole(AllocUtils.getStartPort(component));
95  					NamedElement ne = ElementUtils.getNamedElementFromList(component.getAllAttributes(), RUN);
96  					if (ne instanceof Property) {
97  						ce1.setRole((Property) ne);
98  					}
99  					ConnectorEnd ce2 = c.createEnd();
100 					ce2.setPartWithPort(thread);
101 					Type threadT = thread.getType();
102 					if (threadT instanceof Class) {
103 						// TODO: nicer way to obtain port ...
104 						Property rLC = ((Class) threadT).getAttribute(R_RUN, null);
105 						if (rLC instanceof Port) {
106 							ce2.setRole(rLC);
107 						}
108 					}
109 				}
110 			}
111 		}
112 	}
