97  	/**
98  	 * creates the executor. Needs to be called *before* the other operations of
99  	 * this class.
100 	 *
101 	 * @param tmComponent
102 	 *            the implementation of a component
103 	 *
104 	 * @throws TransformationException
105 	 */
106 	@Override
107 	public void createContainer(Class smComponent, Class tmComponent) throws TransformationException {
108 		Package tmPkgOwner;
109 		if (tmComponent.eResource() == tmCDP.eResource()) {
110 			// tmComponent is in same model as tmContainer
111 			tmPkgOwner = tmComponent.getNearestPackage();
112 		}
113 		else {
114 			// tmComponent is not a member of the target model. This is the case, if tmComponent is
115 			// in an imported model that has not been copied during the transformation towards an intermediate
116 			// model.
117 			tmPkgOwner = MapUtil.getAndCreate(PackageUtil.getRootPackage(tmCDP), tmComponent.allNamespaces(), true);
118 		}
119 		// create a container with the suitable postfix
120 		tmContainerImpl = tmPkgOwner.createOwnedClass(tmComponent.getName() + containerPostfix, false);
121 
122 		// Apply singleton to container, if original class is a singleton
123 		boolean isSingleton = StereotypeUtil.isApplied(smComponent, Singleton.class);
124 		if (isSingleton) {
125 			StereotypeUtil.apply(tmContainerImpl, Singleton.class);
126 		}
127 		CopyUtils.copyID(tmComponent, tmContainerImpl, containerPostfix);
128 
129 		// add part and slot corresponding to component;
130 		executorPart = tmContainerImpl.createOwnedAttribute(executorPartName, tmComponent);
131 		CopyUtils.copyID(tmComponent, executorPart, "e"); //$NON-NLS-1$
132 
133 		executorPart.setIsComposite(true);
134 
135 		// copy ports from the executor (tmComponent) to the container
136 		// (tmContainerImpl)
137 		// TODO: connectors still point to the ports of the executor, implying a non-well-formed
138 		// model. Yet code generation works, as port names are identical
139 		for (Port port : PortUtils.getAllPorts(tmComponent)) {
140 			// copy port
141 			if (port.getVisibility() == VisibilityKind.PUBLIC_LITERAL) {
142 				Port newPort = EcoreUtil.copy(port); // don't use copier, since this is not a copier from source to target model
143 				tmContainerImpl.getOwnedAttributes().add(newPort);
144 				StUtils.copyStereotypes(port, newPort);
145 
146 				// create delegation for application port
147 				Connector containerDelegation = tmContainerImpl.createOwnedConnector("delegation " + port.getName()); //$NON-NLS-1$
148 				CopyUtils.copyID(tmContainerImpl, containerDelegation);
149 				ConnectorEnd end1 = containerDelegation.createEnd();
150 				end1.setRole(newPort);
151 				ConnectorEnd end2 = containerDelegation.createEnd();
152 				end2.setPartWithPort(executorPart);
153 				end2.setRole(port);
154 			}
155 		}
156 
157 		// Inheritance issues:
158 		// currently: container never copies inheritance relationships, but copies inherited ports into container
159 		// Problem: polymorphism would stop working.
160 		// Objective: use inheritance, if possible, but avoid that container inherits from an original non-abstract class (avoid duplication of attributes, ...)
161 		// Option: container inherits from fully abstract super-classes (might have to consider indirect super-classes).
162 		// All connectors and container extension remain within container (i.e. no inheritance of connectors and container extensions)
163 		// Remaining problems: Imagine that component A inherits from component B and both have state-charts.
164 		// => containers of A and B would define a state-Machine and an event pool.
165 		// => No shared pool (but anyway: semantics of this construct is not very clear).
166 		containers.put(tmComponent, this);
167 	}

179 		Package tmPkgOwner = (Package) tmComponent.getOwner();
180 		tmContainerImpl = tmPkgOwner.createOwnedClass(tmComponent.getName() + hwContainerPostfix, false);
181 		CopyUtils.copyID(tmComponent, tmContainerImpl, hwContainerPostfix);

184 		containers.put(tmComponent, this);

187 	/**
188 	 * Create an instance of the container (UML instance specification)
189 	 *
190 	 * @param tmComponent
191 	 *            the component within the target model for which we want to create an instance
192 	 *
193 	 * @param tmIS
194 	 *            the existing instance specification for the component before container expansion. This
195 	 *            instance specification becomes the new instance specification of the executor.
196 	 * @param context
197 	 *            Additional information about the container that is used by instance configurators
198 	 */
199 	@Override
200 	public InstanceSpecification createContainerInstance(Class tmComponent, InstanceSpecification tmIS) throws TransformationException {
201 		// create an instance specification for the container
202 		InstanceSpecification containerIS =
203 				(InstanceSpecification) tmCDP.createPackagedElement(tmIS.getName(), UMLPackage.eINSTANCE.getInstanceSpecification());
204 		// assign new name to original instance specification which reflects
205 		// role within containment hierarchy
206 		tmIS.setName(tmIS.getName() + "." + executorPartName); //$NON-NLS-1$
207 		containerIS.getClassifiers().add(tmContainerImpl);
208 		// existing instance specification (tmIS) is the executor instance specification
209 		DepCreation.createSlot(containerIS, tmIS, executorPart);
210 		counter = 0;
211 		// containers.put(tmComponent, this);
212 		EList<Slot> connectorSlots = new BasicEList<Slot>();
213 
214 		executorIS = tmIS;
215 		// now create instances for the contained elements
216 		for (Property extensionPart : tmContainerImpl.getAttributes()) {
217 			Type tmContainerExtImpl = extensionPart.getType();
218 			if (tmContainerExtImpl instanceof Class) {
219 				if (DepUtils.getSlot(containerIS, extensionPart) == null) {
220 					// no slot for part exists => assume that the part has been added by the container and create an instance specification for it.
221 					InstanceSpecification containerExtIS = null;
222 
223 					String isName = containerIS.getName() + "." + extensionPart.getName(); //$NON-NLS-1$
224 
225 					// create sub-instance and slot for extensions
226 					if (containerExtIS == null) {
227 						containerExtIS = DepCreation.createDepPlan(tmCDP, (Class) tmContainerExtImpl, isName, false);
228 					}
229 					Slot partSlot = DepCreation.createSlot(containerIS, containerExtIS, extensionPart);
230 					if (StereotypeUtil.isApplied(tmContainerExtImpl, InteractionComponent.class)) {
231 						connectorSlots.add(partSlot);
232 					}
233 				}
234 			}
235 		}
236 
237 		// handle propagation of node allocation for connectors. Do that in 2nd loop, since the allocation
238 		// of parts might be incomplete before.
239 		// TODO: is in library now
240 		//for (Slot connectorSlot : connectorSlots) {
241 		//	ConnectorReification.propagateNodeAllocation(tmContainerImpl, containerIS, connectorSlot);
242 		// }
243 		moveSlots(executorIS);
244 		return containerIS;
245 	}

284 	/**
285 	 * apply a container rule, i.e. add either a container extension or an
286 	 * interceptor to the container
287 	 *
288 	 * @param smContainerRule
289 	 *            An container rule
290 	 * @param smComponent
291 	 *            the application component in the source model
292 	 * @param tmComponent
293 	 *            the application component in the target model
294 	 * @param tmIS
295 	 *            the instance specification for the application component in the target model
296 	 * @throws TransformationException
297 	 */
298 	@Override
299 	public void applyRule(ContainerRule smContainerRule, Class smComponent, Class tmComponent)
300 			throws TransformationException
301 	{
302 		Map<Property, EList<Property>> interceptorPartsMap = new HashMap<Property, EList<Property>>();
303 
304 		for (Property part : smContainerRule.getBase_Class().getAllAttributes()) {
305 			Type type = part.getType();
306 			if (type == null) {
307 				String ruleName = (smContainerRule.getBase_Class() != null) ? smContainerRule.getBase_Class().getName() : "undefined"; //$NON-NLS-1$
308 				throw new TransformationException(String.format(
309 						Messages.ContainerTrafo_CannotApplyRule, ruleName));
310 			}
311 			if (part instanceof Port) {
312 				Port newPort = tmContainerImpl.createOwnedPort(part.getName(), part.getType());
313 				StUtils.copyStereotypes(part, newPort);
314 			}
315 			else if (type instanceof Class) {
316 				Class extOrInterceptor = (Class) type;
317 				// DepUtils.chooseImplementation((Class) type,
318 				// new BasicEList<InstanceSpecification>(), false);
319 
320 				if (StereotypeUtil.isApplied(part, InterceptionRule.class)) {
321 					if (StereotypeUtil.isApplied(extOrInterceptor, InteractionComponent.class)) {
322 
323 						InterceptionRule interceptionRule = UMLUtil.getStereotypeApplication(part, InterceptionRule.class);
324 						InterceptionKind interceptionKind = interceptionRule.getInterceptionKind();
325 						EList<Feature> interceptFeatures = interceptionRule.getInterceptionSet();
326 						EList<Property> interceptorParts =
327 								expandInterceptorExtension(interceptionKind, interceptFeatures, extOrInterceptor, tmComponent);
328 						interceptorPartsMap.put(part, interceptorParts);
329 					}
330 					else {
331 						throw new TransformationException(String.format(
332 								Messages.ContainerTrafo_InterceptionRuleButNoInterceptor,
333 								part.getName(), smContainerRule.getBase_Class().getName()));
334 					}
335 				}
336 				else {
337 					Property extensionPart =
338 							expandAggregationExtension(part, extOrInterceptor, tmComponent);
339 					// register relation to facilitate connector copy
340 					copier.setPackageTemplate(smContainerRule.getBase_Class(), tmContainerImpl);
341 					copier.putPair(part, extensionPart);
342 					copier.setPackageTemplate(null, null);
343 				}
344 			}
345 		}
346 		createConnectorForAssociations();
347 
348 		// tell copier that tmcontainerImpl is associated with the smContainerRule
349 		// register a package template (although it is not a template) to assure that the connectors
350 		// get copied, although they are in a different resource (only the connectors are copied, not
351 		// the types of the referenced parts).
352 		// [main issue here: properties of container rule are not copies in the sense of identical
353 		// copies]
354 		copier.setPackageTemplate(smContainerRule.getBase_Class(), tmContainerImpl);
355 
356 		for (Connector connector : smContainerRule.getBase_Class().getOwnedConnectors()) {
357 			Property ruleInterceptorPart = null;
358 			for (Property part : interceptorPartsMap.keySet()) {
359 				if (ConnectorUtil.connectsPart(connector, part)) {
360 					// this connector is a connection between an interceptor (and another part which may not
361 					// be an interceptor => TODO: validation rule yet to write
362 					// we need to copy this connector multiple times, once for each associated interceptor part
363 					ruleInterceptorPart = part;
364 					break;
365 				}
366 			}
367 			if (ruleInterceptorPart != null) {
368 				for (Property interceptorPart : interceptorPartsMap.get(ruleInterceptorPart)) {
369 					// map an interceptor part within rule successively to to an interceptor part
370 					// within the container
371 					copier.putPair(ruleInterceptorPart, interceptorPart);
372 					copier.getCopy(connector);
373 				}
374 			}
375 			else {
376 				// check whether FCM connector
377 				org.eclipse.papyrus.designer.components.FCM.Connector fcmConn =
378 						UMLUtil.getStereotypeApplication(connector, org.eclipse.papyrus.designer.components.FCM.Connector.class);
379 				if (fcmConn != null) {
380 					//
381 					// TODO: is in library now
382 					// ConnectorReification.reifyConnector(copier, tmContainerImpl,
383 					//		ElementUtils.varName(connector), connector, executorIS);
384 				}
385 				else {
386 					copier.remove(connector);
387 					copier.getCopy(connector);
388 				}
389 			}
390 		}
391 		copier.setPackageTemplate(null, null);
392 
393 		TemplateUtils.retargetConnectors(tmContainerImpl);
394 	}

396 	/**
397 	 * This container expansion does not create a new composite. Instead, it
398 	 * adds the extension as a part to the copied application component. The
399 	 * application component also inherits from the type of the container
400 	 * extension in order to avoid copying ports.
401 	 */
402 	Property expandAggregationExtension(Property smExtensionPart, Class smContainerExtImpl, Class tmComponent) throws TransformationException {
403 		Property extensionPart;
404 		String name = smExtensionPart.getName();
405 		Class tmContainerExtImpl = null;
406 
407 		TemplateSignature signature = TemplateUtils.getSignature(smContainerExtImpl);
408 		if (signature == null) {
409 			// no template signature, just copy the container extension into the target model
410 			tmContainerExtImpl = copier.getCopy(smContainerExtImpl);
411 		}
412 		else {
413 			// template signature found, instantiate container extension via template binding mechanism
414 			// use executor component as actual
415 			Classifier actual = tmComponent;
416 			// template signature and instantiate container extension via the
417 			// template binding mechanism, use executor component as actual
418 			TemplateBinding binding = TemplateUtils.fixedBinding(copier.target, smContainerExtImpl, actual);
419 			TemplateInstantiation ti = new TemplateInstantiation(copier, binding);
420 			tmContainerExtImpl = ti.bindElement(smContainerExtImpl);
421 		}
422 
423 		// --------------------------------------------------
424 		// a container extension could be subject to container rules as well, i.e. need
425 		// to handle recursive rule application.
426 		EList<ContainerRule> rules = FCMUtil.getAllContainerRules(smContainerExtImpl);
427 
428 		// get container trafo instance, if already existing
429 		AbstractContainerTrafo containerTrafo = AbstractContainerTrafo.get(tmContainerExtImpl);
430 
431 		if (containerTrafo == null) {
432 			// no container exists, check rules and create eventually
433 			for (ContainerRule rule : rules) {
434 				// at least one active rule => create container (or get previously instantiated))
435 				if (containerTrafo == null) {
436 					if (rule.getKind() == ContainerRuleKind.LIGHT_WEIGHT_OO_RULE) {
437 						throw new TransformationException(Messages.ContainerTrafo_RecursiveLWnotSupported);
438 					}
439 					else {
440 						containerTrafo = new ContainerTrafo(copier, tmCDP, null);
441 					}
442 					containerTrafo.createContainer(smContainerExtImpl, tmContainerExtImpl);
443 				}
444 				else {
445 					// configure only??
446 				}
447 				containerTrafo.applyRule(rule, smContainerExtImpl, tmContainerExtImpl);
448 			}
449 			if (containerTrafo != null) {
450 				containerTrafo.finalize();
451 				tmContainerExtImpl = ((ContainerTrafo) containerTrafo).getContainer();
452 			}
453 		}
454 		// --------------------------------------------------
455 
456 		// add part associated with the extension to the container
457 		extensionPart = tmContainerImpl.createOwnedAttribute(name, tmContainerExtImpl);
458 
459 		// Copy.copyID(tmComponent, extensionPart, "a");
460 		extensionPart.setAggregation(smExtensionPart.getAggregation());
461 		CopyUtils.copyMultElemModifiers(smExtensionPart, extensionPart);
462 		CopyUtils.copyFeatureModifiers(smExtensionPart, extensionPart);
463 
464 		return extensionPart;
465 	}

467 	/**
468 	 * Expand an interceptor definition between the container and the executor.
469 	 *
470 	 * @param extKind
471 	 *            the interception kind (intercept which ports: all, some, out, ...)
472 	 * @param featureList
473 	 *            The set of ports to intercept (In case of "some" above)
474 	 * @param smContainerConnImpl
475 	 *            The connector (interceptor) implementation in the source model
476 	 * @param tmComponent
477 	 *            the application component (executor) in the target model
478 	 * @return
479 	 * @throws TransformationException
480 	 */
481 	EList<Property> expandInterceptorExtension(InterceptionKind extKind, EList<Feature> featureList, Class smContainerConnImpl, Class tmComponent)
482 			throws TransformationException
483 	{
484 		EList<Property> connectorParts = new BasicEList<Property>();
485 		// replace delegation connectors with extension connector
486 		for (Port port : PortUtils.getAllPorts(tmComponent)) {
487 			// delegate to component via connector
488 			boolean match = true;
489 			if (extKind == InterceptionKind.INTERCEPT_ALL_IN) {
490 				// IN-PORT = provided port
491 				match = (PortUtils.getProvided(port) != null);
492 			}
493 			else if (extKind == InterceptionKind.INTERCEPT_ALL_OUT) {
494 				// IN-PORT = provided port
495 				match = (PortUtils.getRequired(port) != null);
496 			}
497 			else if (extKind == InterceptionKind.INTERCEPT_SOME) {
498 				// comparison based on name, since in different models
499 				match = (ElementUtils.getNamedElementFromList(featureList, port.getName()) != null);
500 			}
501 			else if (extKind == InterceptionKind.INTERCEPT_MATCHING) {
502 				EList<Port> interceptorPorts = PortUtils.getAllPorts(smContainerConnImpl);
503 				match = false;
504 				// get first port from interception connector that is typed with a template parameter
505 				for (Port interceptorPort : interceptorPorts) {
506 					if (interceptorPort.getType().getOwner() instanceof TemplateParameter) {
507 						match = (PortUtils.getKind(port) == PortUtils.getKind(interceptorPort));
508 						break;
509 					}
510 				}
511 			}
512 
513 			// else INTERCEPT_ALL_PORT => match remains true
514 			if (!match) {
515 				// port does not match criterion, continue with next port
516 				continue;
517 			}
518 
519 			Property connectorPart;
520 			String interceptionName = interceptorName + port.getName() + counter;
521 			Connector interceptionConnector = null;
522 			// get delegation connector
523 			for (Connector connector : tmContainerImpl.getOwnedConnectors()) {
524 				if (ConnectorUtil.connectsPort(connector, port)) {
525 					interceptionConnector = connector;
526 					break;
527 				}
528 			}
529 			// interceptionConnector = tmContainerImpl.getOwnedConnector
530 			// ("delegation " + port.getName ());
531 			if (interceptionConnector == null) {
532 				throw new TransformationException(Messages.ContainerTrafo_CannotFindDelegationConn);
533 			}
534 
535 			interceptionConnector.setName(interceptionName);
536 			org.eclipse.papyrus.designer.components.FCM.Connector fcmConn = StereotypeUtil.applyApp(interceptionConnector, org.eclipse.papyrus.designer.components.FCM.Connector.class);
537 			InteractionComponent fcmConnType = UMLUtil.getStereotypeApplication(smContainerConnImpl, InteractionComponent.class);
538 			fcmConn.setIc(fcmConnType);
539 
540 			// pass target component and port to interceptor (not clean, define
541 			// suitable template signature as for instance in methodCall_comp
542 			ContainerTrafo.instance = executorIS;
543 			ContainerTrafo.port = port;
544 			// TODO: is in library now
545 			// connectorPart = ConnectorReification.reifyConnector(copier, tmContainerImpl, ElementUtils.varName(interceptionConnector), interceptionConnector, executorIS);
546 			connectorPart = null;
547 			connectorParts.add(connectorPart);
548 			ContainerTrafo.port = null;
549 			portInfo.put(connectorPart, port);
550 			// delete intermediate connector (has been replaced by two
551 			// connections to the reified connector)
552 			interceptionConnector.destroy();
553 		}
554 		counter++;
555 		return connectorParts;
556 	}

558 	/**
559 	 * Move a slot from the executor instance specification to an instance
560 	 * specification of an extension/interceptor. This is required, since
561 	 * users cannot configure containers directly. The modification is done in
562 	 * the target model, i.e. the user model is not affected.
563 	 *
564 	 * Configuration is based on the idea that we can have multiple classifiers for an
565 	 * instance specification and that we add additional slots for attributes of the 2nd
566 	 * classifier which becomes "visible" after container transformation.
567 	 * Main use: configure priority of thread within container.
568 	 *
569 	 * TODO: Not clear, whether we should keep this operation, since configuration of elements that are not
570 	 * (directly) part of the user model is confusing. Application was: configuration
571 	 * of thread priorities within container.
572 	 */
573 	public void moveSlots(InstanceSpecification containerIS) {
574 		// Log.log(Status.INFO, Log.TRAFO_CONTAINER, String.format(Messages.ContainerTrafo_InfoMoveSlots, executorIS.getQualifiedName()));
575 		Classifier mainCl = DepUtils.getClassifier(executorIS);
576 		Iterator<Slot> slotIt = executorIS.getSlots().iterator();
577 		while (slotIt.hasNext()) {
578 			Slot slot = slotIt.next();
579 			String featureName = slot.getDefiningFeature().getName();
580 			if (!mainCl.getAllAttributes().contains(slot.getDefiningFeature())) {
581 				// defining feature does not belong to main classifier
582 				// => move it to first matching instance specification
583 				for (InstanceSpecification is : DepUtils.getContainedInstances(containerIS)) {
584 					Classifier containedCl = DepUtils.getClassifier(is);
585 					if (ElementUtils.getNamedElementFromList(containedCl.getAllAttributes(), featureName) != null) {
586 						if (executorIS != is) {
587 							// remove slot first from iterator, as addition below removes it from the list (slots are owned)
588 							slotIt.remove();
589 							is.getSlots().add(slot);
590 							break;
591 						}
592 					}
593 				}
594 			}
595 		}
596 	}

598 	/**
599 	 * Create an connector for parts that have an association. The rationale
600 	 * behind this is support connections between the state machine in the container
601 	 * and the executor.
602 	 *
603 	 */
604 	public void createConnectorForAssociations() {
605 		// TODO: keep list of added parts, only re-check those!
606 
607 		for (Property part : ElementUtils.getParts(tmContainerImpl)) {
608 			if (part.getType() == null) {
609 				continue;
610 			}
611 			for (Association association : part.getType().getAssociations()) {
612 				// Part has an association. Check if the other association end
613 				// targets one of
614 				// other parts within the container
615 				for (Property end : association.getMemberEnds()) {
616 					Type type = end.getType();
617 					if ((type != null) && (type != part.getType())) {
618 						// end is not typed with the part we are currently
619 						// examining
620 						// thus, it might belong to another part of the
621 						// composite
622 						for (Property checkPart : ElementUtils.getParts(tmContainerImpl)) {
623 							if (type == checkPart.getType()) {
624 								// found an association between two parts of the
625 								// container => create connection,
626 								// unless already existing.
627 								if (!ConnectorUtil.existsConnector(tmContainerImpl, part, checkPart)) {
628 									Connector conn = tmContainerImpl.createOwnedConnector(part.getName() + "_" + checkPart.getName()); //$NON-NLS-1$
629 									conn.setType(association);
630 									conn.createEnd().setRole(part);
631 									conn.createEnd().setRole(checkPart);
632 								}
633 							}
634 						}
635 					}
636 				}
637 			}
638 		}
639 	}

646 	/**
647 	 * Connector ports of executor with compatible ports offered by container extensions
648 	 * (runtime system)
649 	 */
650 	public void discoverServices() {
651 		Type executorType = executorPart.getType();
652 		if (!(executorType instanceof EncapsulatedClassifier)) {
653 			return;
654 		}
655 		EncapsulatedClassifier ec = (EncapsulatedClassifier) executorType;
656 		for (Port executorPort : ec.getOwnedPorts()) {
657 			for (Property svcPart : tmContainerImpl.getOwnedAttributes()) {
658 				if (svcPart == executorPart) {
659 					continue;
660 				}
661 				Type containerSvcType = svcPart.getType();
662 				if (containerSvcType instanceof EncapsulatedClassifier) {
663 					EncapsulatedClassifier containerSvc = (EncapsulatedClassifier) containerSvcType;
664 					for (Port svcPort : containerSvc.getOwnedPorts()) {
665 						if (PortUtils.matches(executorPort, svcPort, true)) {
666 							// create connector
667 							Connector c = tmContainerImpl.createOwnedConnector(
668 									String.format("auto from %s to %s", executorPart.getName(), svcPart.getName())); //$NON-NLS-1$
669 							ConnectorEnd ce1 = c.createEnd();
670 							ConnectorEnd ce2 = c.createEnd();
671 							ce1.setPartWithPort(executorPart);
672 							ce1.setRole(executorPort);
673 							ce2.setPartWithPort(svcPart);
674 							ce2.setRole(svcPort);
675 						}
676 					}
677 				}
678 			}
679 		}
680 	}

685 	 * a part with a port that is intercepted.

687 	 * @param containerInstance

692 		Classifier containerCandidate = DepUtils.getClassifier(containerInstance);

696 			if (containerTrafo instanceof ContainerTrafo) {

698 			}
699 		}
