130 	/**
131 	 * override method in order to save & restore action bars entries
132 	 */
133 	@Override
134 	public void createPartControl(Composite parent) {
135 		// save Papyrus (GMF) handlers
136 		IActionBars actionBars = getEditorSite().getActionBars();
137 		if ((actionBars != null) && (gmfUndo == null)) {
138 			gmfUndo = actionBars.getGlobalActionHandler(ITextEditorActionConstants.UNDO);
139 			gmfRedo = actionBars.getGlobalActionHandler(ITextEditorActionConstants.REDO);
140 		}
141 
142 		super.createPartControl(parent);
143 
144 		if (actionBars != null) {
145 			// save CDT (textEditor) handlers
146 			textUndo = actionBars.getGlobalActionHandler(ITextEditorActionConstants.UNDO);
147 			textRedo = actionBars.getGlobalActionHandler(ITextEditorActionConstants.REDO);
148 			// & restore Papyrus bar
149 			actionBars.setGlobalActionHandler(ITextEditorActionConstants.UNDO, gmfUndo);
150 			actionBars.setGlobalActionHandler(ITextEditorActionConstants.REDO, gmfRedo);
151 			actionBars.updateActionBars();
152 		}
153 
154 		gotoListener = new Adapter() {
155 
156 			// assure that gotoElement is called, if the element in the model gets updated
157 			@Override
158 			public void notifyChanged(Notification notification) {
159 				if (notification.getEventType() == Notification.SET) {
160 					Object newValue = notification.getNewValue();
161 					if (newValue instanceof NamedElement) {
162 						gotoElement((NamedElement) newValue);
163 					}
164 
165 				}
166 			}
167 
168 			@Override
169 			public Notifier getTarget() {
170 				return null;
171 			}
172 
173 			@Override
174 			public void setTarget(Notifier newTarget) {
175 			}
176 
177 			@Override
178 			public boolean isAdapterForType(Object type) {
179 				return false;
180 			}
181 		};
182 		papyrusTextInstance.eAdapters().add(gotoListener);
183 
184 		if (papyrusTextInstance.getSelectedObject() instanceof NamedElement) {
185 			gotoElement((NamedElement) papyrusTextInstance.getSelectedObject());
186 		}
187 	}

189 	/**
190 	 *
191 	 * @see org.eclipse.ui.texteditor.AbstractDecoratedTextEditor#createSourceViewer(org.eclipse.swt.widgets.Composite, org.eclipse.jface.text.source.IVerticalRuler, int)
192 	 *
193 	 * @param parent
194 	 * @param ruler
195 	 * @param styles
196 	 * @return
197 	 */
198 	@Override
199 	public ISourceViewer createSourceViewer(Composite parent, IVerticalRuler ruler, int styles) {
200 
201 		final ISourceViewer viewer = super.createSourceViewer(parent, ruler, styles);
202 		// ISourceViewer viewer = new DelegatingSourceViewer(origViewer, (Classifier)papyrusTextInstance.getEditedObject());
203 		focusListener = new FocusListener() {
204 
205 			@Override
206 			public void focusLost(FocusEvent e) {
207 				// potential problem for undo/redo!!
208 				if (isDirty()) {
209 					doSave(new NullProgressMonitor());
210 					syncCpp.syncCDTtoModel();
211 					Classifier classifier = (Classifier) papyrusTextInstance.getEditedObject();
212 					// regenerate code. TODO: raises update dialog
213 					// SyncModelToCDT.syncModelToCDT(classifier);
214 				}
215 				// restore handlers
216 				IActionBars actionBars = getEditorSite().getActionBars();
217 				if (actionBars != null) {
218 					if ((gmfUndo != null) && (gmfRedo != null)) {
219 						actionBars.setGlobalActionHandler(ITextEditorActionConstants.UNDO, gmfUndo);
220 						actionBars.setGlobalActionHandler(ITextEditorActionConstants.REDO, gmfRedo);
221 						actionBars.updateActionBars();
222 					}
223 				}
224 			}
225 
226 			@Override
227 			public void focusGained(FocusEvent e) {
228 				// restore handlers
229 				IActionBars actionBars = getEditorSite().getActionBars();
230 				if (actionBars != null) {
231 					if ((textUndo != null) && (textRedo != null)) {
232 						actionBars.setGlobalActionHandler(ITextEditorActionConstants.UNDO, textUndo);
233 						actionBars.setGlobalActionHandler(ITextEditorActionConstants.REDO, textRedo);
234 						actionBars.updateActionBars();
235 					}
236 				}
237 			}
238 		};
239 		// register focus listener
240 		viewer.getTextWidget().addFocusListener(focusListener);
241 		SelectionListener selectionListener = new SelectionListener() {
242 
243 			@Override
244 			public void widgetSelected(SelectionEvent e) {
245 				IEditorPart ep = getEditorSite().getPage().getActiveEditor();
246 				ISelection selection = getSelectionProvider().getSelection();
247 				if ((ep instanceof IGotoMarker) && (selection instanceof ITextSelection)) {
248 					if (((ITextSelection) selection).getLength() > 0) {
249 						// don't try to navigate during selection of a chunk of text.
250 						return;
251 					}
252 					IMarker marker;
253 					/*
254 					 * create a temporary validation marker on the
255 					 * srcFile file, call the gotoMarker operation of the editor
256 					 * and remove the marker afterwards.
257 					 * TODO: operation activates the model-explorer
258 					 */
259 					try {
260 						Element element = reveal.obtainSelectedElement((ITextSelection) selection);
261 						String modelURI = EcoreUtil.getURI(element).toString();
262 						marker = srcFile.createMarker(EValidator.MARKER);
263 						marker.setAttribute(IMarker.SEVERITY, IMarker.SEVERITY_INFO); // set severity before URI to avoid marker update without associated icon
264 						marker.setAttribute(EValidator.URI_ATTRIBUTE, modelURI);
265 						IWorkbenchPage wbpage = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getActivePage();
266 						IWorkbenchPart activePart = wbpage.getActivePart();
267 						((IGotoMarker) ep).gotoMarker(marker);
268 						// reactivate this editor
269 						wbpage.activate(activePart);
270 						marker.delete();
271 					} catch (CoreException e1) {
272 						// TODO Auto-generated catch block
273 						e1.printStackTrace();
274 					}
275 				}
276 			}
277 
278 			@Override
279 			public void widgetDefaultSelected(SelectionEvent e) {
280 			}
281 		};
282 
283 		// TODO: don't add selection listener for the moment. It can cause deadlocks (setting a marker attribute while file is locked, reproducible with cut (^X) on selection)
284 		// viewer.getTextWidget().addSelectionListener(selectionListener);
285 
286 		return viewer;
287 	}

289 	/**
290 	 * Goto a specific element within the text editor. Currently, only methods are supported.
291 	 *
292 	 * @param element
293 	 * @throws CoreException
294 	 */
295 	public void gotoElement(NamedElement element) {
296 		ICElement ice = CDTUITools.getEditorInputCElement(m_input);
297 
298 		if (ice instanceof ITranslationUnit) {
299 			ITranslationUnit itu = (ITranslationUnit) ice;
300 			ICElement icElement = ObtainICElement.getICElement(syncCpp.getCodeGen(), itu, element);
301 			if (icElement instanceof ISourceReference) {
302 				try {
303 					ISourceRange range = ((ISourceReference) icElement).getSourceRange();
304 
305 					ISourceViewer viewer = getSourceViewer();
306 					viewer.revealRange(range.getStartPos(), 1);
307 					viewer.setSelectedRange(range.getStartPos(), range.getLength());
308 					return;
309 				} catch (CoreException e) {
310 					Activator.log.error(e);
311 				}
312 			}
313 		}
314 	}

317 	public boolean isDirty() {
318 		boolean isDirty = super.isDirty();
319 		if (isDirty != oldDirty) {
320 			oldDirty = isDirty;
321 			// hack. Normally, the saveAndDirtyService detects this itself (should find a solution how it can do that)
322 			// ((SaveAndDirtyService) saveAndDirtyService).fireIsDirtyChanged();
323 		}
324 		return isDirty;

327 	/**
328 	 * We override this method because setInput can't be overriden for TextEditor.
329 	 * We replace the default Papyrus input with the CPP file
330 	 *
331 	 * @see org.eclipse.ui.editors.text.TextEditor#doSetInput(org.eclipse.ui.IEditorInput)
332 	 *
333 	 * @param input
334 	 * @throws CoreException
335 	 */
336 	@Override
337 	protected void doSetInput(IEditorInput input) throws CoreException {
338 
339 		URI uri = papyrusTextInstance.eResource().getURI();
340 		Classifier classifier = (Classifier) papyrusTextInstance.getEditedObject();
341 		String generatorID = papyrusTextInstance.getGeneratorID();
342 		srcFile = SyncModelToCDT.syncModelToCDT(classifier, generatorID);
343 		if (srcFile == null || !srcFile.exists()) {
344 			throw new PartInitException("Code generation before editing failed. Please check error log");
345 		}
346 		IEditorInput newInput = new FileEditorInput(srcFile);
347 
348 		// IStorage storage = new TextStorage(string);
349 		super.doSetInput(newInput);
350 
351 		syncCpp = new SyncCDTtoModel(newInput, classifier, uri.segment(1), generatorID);
352 		m_input = newInput;
353 		reveal = new RevealCurrentOperation(newInput, classifier, uri.segment(1));
354 
355 		// add the reconciler to get syntax check and completion. (still no full checks)
356 		addReconcileListener(new CodanCReconciler());
357 	}

359 	/**
360 	 *
361 	 * @see org.eclipse.ui.editors.text.TextEditor#dispose()
362 	 *
363 	 */
364 	@Override
365 	public void dispose() {
366 
367 		saveAndDirtyService.removeIsaveablePart(this);
368 		if (gotoListener != null) {
369 			papyrusTextInstance.eAdapters().remove(gotoListener);
370 		}
371 
372 		// we remove the listener
373 		StyledText st = getSourceViewer().getTextWidget();
374 		st.removeFocusListener(focusListener);
375 		super.dispose();
376 		// assure that embedded text widget is disposed now. Otherwise, it will be closed
377 		// later, when the editorSite is already set to null by Papyrus (mainEditorSite attribute in MultiPageEditorSite4x)
378 		// resulting in a null pointer exception and resulting inconsistencies in the Papyrus editor.
379 		st.dispose();
380 	}
