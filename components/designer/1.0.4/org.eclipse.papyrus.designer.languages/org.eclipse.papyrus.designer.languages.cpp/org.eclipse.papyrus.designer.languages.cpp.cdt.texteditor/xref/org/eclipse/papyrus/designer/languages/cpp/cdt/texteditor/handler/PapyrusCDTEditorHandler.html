72  	/**
73  	 *
74  	 * @see org.eclipse.infra.core.commands.AbstractHandler#isEnabled()
75  	 *
76  	 * @return
77  	 */
78  	@Override
79  	public boolean isEnabled() {
80  		updateSelectedEObject();
81  		// Filter Classes (including Behaviors, since Behavior inherits from Class), Operation and Transition
82  		if (selectedEObject instanceof Class ||
83  				selectedEObject instanceof Operation ||
84  				selectedEObject instanceof Transition) {
85  			URI uri = selectedEObject.eResource().getURI();
86  
87  			// URIConverter uriConverter = resource.getResourceSet().getURIConverter();
88  			IWorkspaceRoot root = ResourcesPlugin.getWorkspace().getRoot();
89  			if (uri.segmentCount() < 2) {
90  				return false;
91  			}
92  			IProject modelProject = root.getProject(uri.segment(1));
93  			return modelProject.exists();
94  		}
95  		return false;
96  	}

99  	 * @see org.eclipse.infra.core.commands.IHandler#execute(org.eclipse.infra.core.commands.ExecutionEvent)
100 	 *
101 	 * @param event
102 	 * @return
103 	 * @throws ExecutionException
104 	 */
105 	@Override
106 	public Object execute(final ExecutionEvent event) throws ExecutionException {
107 		try {
108 			final ServicesRegistry serviceRegistry = ServiceUtilsForHandlers.getInstance().getServiceRegistry(event);
109 			TransactionalEditingDomain domain = ServiceUtils.getInstance().getTransactionalEditingDomain(serviceRegistry);
110 
111 			// Create the transactional command
112 			AbstractEMFOperation command = new AbstractEMFOperation(domain, "Create CDT editor") {
113 
114 				@Override
115 				protected IStatus doExecute(final IProgressMonitor monitor, final IAdaptable info) throws ExecutionException {
116 					try {
117 						PapyrusCDTEditorHandler.this.doExecute(serviceRegistry);
118 					} catch (ServiceException e) {
119 						Activator.log.error(e);
120 						return Status.CANCEL_STATUS;
121 					} catch (NotFoundException e) {
122 						Activator.log.error(e);
123 						return Status.CANCEL_STATUS;
124 					}
125 					return Status.OK_STATUS;
126 				}
127 			};
128 
129 			// Execute the command
130 			CheckedOperationHistory.getInstance().execute(command, new NullProgressMonitor(), null);
131 		} catch (ExecutionException e) {
132 			Activator.log.error("Can't create a CDT editor", e);
133 		} catch (ServiceException e) {
134 			Activator.log.error("Service exception during creation of CDT editor", e);
135 		}
136 		return null;
137 	}

140 	/**
141 	 * Do the execution of the command.
142 	 *
143 	 * @param serviceRegistry
144 	 * @throws ServiceException
145 	 * @throws NotFoundException
146 	 */
147 	public void doExecute(final ServicesRegistry serviceRegistry) throws ServiceException, NotFoundException {
148 		// Get the page manager allowing to add/open an editor.
149 		final IPageManager pageMngr = ServiceUtils.getInstance().getService(IPageManager.class, serviceRegistry);
150 
151 		Classifier classifierToEdit = getClassifierToEdit();
152 		TextEditorModel editorModel = getEditorModel(serviceRegistry, classifierToEdit);
153 		if (editorModel == null) {
154 			// no editor exist for the given file => create
155 			editorModel = createEditorModel(serviceRegistry, classifierToEdit);
156 			if (editorModel == null) {
157 				return;
158 			}
159 		}
160 		ILangCodegen codegen = LanguageCodegen.getGenerator(TextEditorConstants.CPP, editorModel.getGeneratorID());
161 
162 		if (codegen.getTargetProject(classifierToEdit, true) == null) {
163 			return;
164 		}
165 
166 		if (selectedEObject instanceof Transition) {
167 			Transition transition = (Transition) selectedEObject;
168 			if (transition.getEffect() == null) {
169 				Behavior effect = transition.createEffect("effectOf" + transition.getName(), UMLPackage.eINSTANCE.getOpaqueBehavior()); //$NON-NLS-1$
170 			}
171 		}
172 		// add the new editor model to the sash.
173 		editorModel.setSelectedObject(selectedEObject);
174 
175 		final TextEditorModel editorModelFinal = editorModel;
176 		// open asynchronously to prevent handler cycles, see bug 434484
177 		Display.getDefault().asyncExec(new Runnable() {
178 
179 			@Override
180 			public void run() {
181 				if (pageMngr.isOpen(editorModelFinal)) {
182 					// select existing editor
183 					pageMngr.selectPage(editorModelFinal);
184 				} else {
185 					pageMngr.openPage(editorModelFinal);
186 				}
187 				try {
188 					// move page to the RIGHT
189 					DiSashModelManager modelMngr = ServiceUtils.getInstance().getService(DiSashModelManager.class, serviceRegistry);
190 					ISashWindowsContentProvider sashContentProvider = modelMngr.getISashWindowsContentProvider();
191 					Object rootModel = sashContentProvider.getRootModel();
192 
193 					if (rootModel instanceof TabFolder) {
194 						// root = tabFolder, i.e. there is a single folder
195 						ISashWindowsContainer sashContainer = ServiceUtils.getInstance().getService(ISashWindowsContainer.class, serviceRegistry);
196 						int index = lookupIndex((TabFolder) rootModel, editorModelFinal);
197 						if (index != -1) {
198 							sashContentProvider.createFolder(sashContainer.getSelectedTabFolderModel(), index, sashContainer.getSelectedTabFolderModel(), SWT.RIGHT);
199 						}
200 					} else if (rootModel instanceof SashPanel) {
201 						// multiple tab-folders exist. Find existing one and move editorModel to other
202 						// TODO
203 						// ISashWindowsContainer sashContainer = ServiceUtils.getInstance().getISashWindowsContainer(serviceRegistry);
204 						// sashContentProvider.movePage(sashContainer.getSelectedTabFolderModel(), lookupIndex(sourceTab, editorModel), targetTabModel, -1);
205 					}
206 				} catch (ServiceException e) {
207 
208 				}
209 			}
210 
211 		});
212 
213 	}

215 	/**
216 	 * Create a model identifying the editor. This model will be saved with the sash
217 	 *
218 	 * @return
219 	 * @throws ServiceException
220 	 * @throws NotFoundException
221 	 *             The model where to save the TableInstance is not found.
222 	 */
223 	protected TextEditorModel createEditorModel(final ServicesRegistry serviceRegistry, Classifier classifierToEdit) throws ServiceException, NotFoundException {
224 		TextEditorModel editorModel = TextEditorModelFactory.eINSTANCE.createTextEditorModel();
225 
226 		editorModel.setEditedObject(classifierToEdit);
227 		editorModel.setType(PapyrusCDTEditor.EDITOR_TYPE);
228 		editorModel.setName("CDT " + classifierToEdit.getName()); //$NON-NLS-1$
229 		ILangCodegen codegen = LanguageCodegen.chooseGenerator(TextEditorConstants.CPP, classifierToEdit);
230 		if (codegen == null) {
231 			return null;
232 		}
233 		editorModel.setGeneratorID(LanguageCodegen.getID(codegen));
234 		TextEditorModelSharedResource model = (TextEditorModelSharedResource) ServiceUtils.getInstance().getModelSet(serviceRegistry).getModelChecked(TextEditorModelSharedResource.MODEL_ID);
235 		model.addTextEditorModel(editorModel);
236 
237 		return editorModel;
238 	}

240 	/**
241 	 * The classifier to edit - corresponding to the selected object.
242 	 *
243 	 * @return
244 	 */
245 	protected Classifier getClassifierToEdit() {
246 		if (selectedEObject instanceof Operation) {
247 			return ((Operation) selectedEObject).getFeaturingClassifiers().get(0);
248 		} else if (selectedEObject instanceof Transition) {
249 			return ((Transition) selectedEObject).getContainer().getStateMachine().getContext();
250 		} else if (selectedEObject instanceof Behavior) {
251 			Element owner = (Behavior) selectedEObject;
252 			while (owner != null) {
253 				owner = owner.getOwner();
254 				if ((owner instanceof Classifier) && !(owner instanceof Behavior)) {
255 					return (Classifier) owner;
256 				}
257 			}
258 			return null;
259 		} else if (selectedEObject instanceof Classifier) {
260 			// must be class or datatype
261 			return (Classifier) selectedEObject;
262 		}
263 		return null;
264 	}

266 	/**
267 	 * return the editor model corresponding to an EObject
268 	 *
269 	 * @param serviceRegistry
270 	 *            the service registry
271 	 * @param classifierToEdit
272 	 *            The classifier for which a CDT editor should be opened
273 	 * @return
274 	 * @throws ServiceException
275 	 * @throws NotFoundException
276 	 */
277 	protected TextEditorModel getEditorModel(final ServicesRegistry serviceRegistry, Classifier classifierToEdit) throws ServiceException, NotFoundException {
278 		TextEditorModelSharedResource model = (TextEditorModelSharedResource) ServiceUtils.getInstance().getModelSet(serviceRegistry).getModelChecked(TextEditorModelSharedResource.MODEL_ID);
279 		return model.getTextEditorModel(classifierToEdit);
280 	}

282 	/**
283 	 * Recursively search in sash models for a FolderModel.
284 	 * Return the first encountered folder.
285 	 *
286 	 * @param panelModel
287 	 * @return
288 	 */
289 	public static int lookupIndex(TabFolder folder, Object model) {
290 
291 		int index = 0;
292 		for (PageRef pr : folder.getChildren()) {
293 			if (pr.getPageIdentifier() == model) {
294 				return index;
295 			}
296 			index++;
297 		}
298 		return -1;
299 	}
