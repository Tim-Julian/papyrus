13  package org.eclipse.papyrus.designer.languages.cpp.codegen.ui.preferences;
14  
15  import org.eclipse.jface.preference.FieldEditorPreferencePage;
16  import org.eclipse.jface.preference.StringFieldEditor;
17  import org.eclipse.jface.text.Document;
18  import org.eclipse.jface.text.source.SourceViewer;
19  import org.eclipse.papyrus.designer.languages.cpp.codegen.Activator;
20  import org.eclipse.papyrus.designer.languages.cpp.codegen.preferences.CppCodeGenConstants;
21  import org.eclipse.swt.SWT;
22  import org.eclipse.swt.layout.FillLayout;
23  import org.eclipse.swt.layout.GridData;
24  import org.eclipse.swt.widgets.Group;
25  import org.eclipse.ui.IWorkbench;
26  import org.eclipse.ui.IWorkbenchPreferencePage;
27  
28  
29  /**
30   * This class represents a preference page that is contributed to the
31   * Preferences dialog. By subclassing <samp>FieldEditorPreferencePage </samp>,
32   * we can use the field support built into JFace that allows us to create a page
33   * that is small and knows how to save, restore and apply itself.
34   * <p>
35   * This page is used to modify preferences only. They are stored in the preference store that belongs to the main plug-in class. That way, preferences can be accessed directly via the preference store.
36   */
37  
38  public class CodegenPreferencePage extends FieldEditorPreferencePage implements
39  		IWorkbenchPreferencePage {
40  
41  	private Document pCommentDoc = new Document();
42  
43  	public CodegenPreferencePage() {
44  		super(GRID);
45  		setPreferenceStore(Activator.getDefault().getPreferenceStore());
46  		setDescription("This preferences page allows to customize Papyrus C++ code generation");
47  	}
48  
49  	public void addTextField(String name, String label, Document currentDoc) {
50  		// ///////////////////////////////////////////////////////////////////////
51  		// Create a Group for the text
52  		// ///////////////////////////////////////////////////////////////////////
53  		Group txtGroup = new Group(getFieldEditorParent(), SWT.RESIZE);
54  		txtGroup.setLayout(new FillLayout());
55  		txtGroup.setText(label);
56  		GridData gd = new GridData(GridData.FILL_BOTH /* FILL_HORIZONTAL */);
57  		// gd.heightHint = 250;
58  		gd.horizontalSpan = 2;
59  		txtGroup.setLayoutData(gd);
60  
61  		// Text area
62  		SourceViewer txtViewer = new SourceViewer(txtGroup, null, SWT.MULTI
63  				| SWT.H_SCROLL | SWT.V_SCROLL | SWT.BORDER);
64  
65  		txtViewer.setDocument(currentDoc);
66  
67  		// Retrieving existing preference
68  		String content = getPreferenceStore().getString(name);
69  
70  		// Loading preference in txt zone
71  		currentDoc.set(content);
72  
73  	}
74  
75  	/**
76  	 * Creates the field editors. Field editors are abstractions of the common
77  	 * GUI blocks needed to manipulate various types of preferences. Each field
78  	 * editor knows how to save and restore itself.
79  	 */
80  
81  	@Override
82  	public void createFieldEditors() {
83  
84  		addField(new StringFieldEditor(CppCodeGenConstants.P_HEADER_SUFFIX,
85  				"Suffix for C++ header files ", getFieldEditorParent()));
86  		addField(new StringFieldEditor(CppCodeGenConstants.P_IMPLEM_SUFFIX,
87  				"Suffix for C++ implementation files ", getFieldEditorParent()));
88  		addField(new StringFieldEditor(CppCodeGenConstants.P_PROJECT_PREFIX,
89  				"Prefix for generated C++ (CDT) projects ", getFieldEditorParent()));
90  
91  		// Add P_COMMENT_HEADER
92  		addTextField(CppCodeGenConstants.P_COMMENT_HEADER, "Header for generated files",
93  				pCommentDoc);
94  	}
95  
96  	@Override
97  	public void init(IWorkbench workbench) {
98  	}
99  
100 	@Override
101 	public void performDefaults() {
102 
103 		// initialize comment field from default
104 		pCommentDoc.set(getPreferenceStore().getDefaultString(CppCodeGenConstants.P_COMMENT_HEADER));
105 		super.performDefaults();
106 	}
107 
108 	@Override
109 	public boolean performOk() {
110 
111 		// Store txt for P_COMMENT_HEADER
112 		String currentComment = pCommentDoc.get();
113 		getPreferenceStore().setValue(CppCodeGenConstants.P_COMMENT_HEADER, currentComment);
114 
115 		return super.performOk();
116 	}
117 }
