30  package org.eclipse.papyrus.designer.languages.java.reverse.ast.type;
31  
32  import org.eclipse.papyrus.designer.languages.java.reverse.ast.visitor.GenericVisitor;
33  import org.eclipse.papyrus.designer.languages.java.reverse.ast.visitor.VoidVisitor;
34  
35  /**
36   * @author Julio Vilmar Gesser
37   */
38  public final class ReferenceType extends Type {
39  
40  	private Type type;
41  
42  	private int arrayCount;
43  
44  	public ReferenceType() {
45  	}
46  
47  	public ReferenceType(int beginLine, int beginColumn, int endLine, int endColumn, Type type, int arrayCount) {
48  		super(beginLine, beginColumn, endLine, endColumn);
49  		this.type = type;
50  		this.arrayCount = arrayCount;
51  	}
52  
53  	@Override
54  	public <R, A> R accept(GenericVisitor<R, A> v, A arg) {
55  		return v.visit(this, arg);
56  	}
57  
58  	@Override
59  	public <A> void accept(VoidVisitor<A> v, A arg) {
60  		v.visit(this, arg);
61  	}
62  
63  	public int getArrayCount() {
64  		return arrayCount;
65  	}
66  
67  	public Type getType() {
68  		return type;
69  	}
70  
71  	public void setArrayCount(int arrayCount) {
72  		this.arrayCount = arrayCount;
73  	}
74  
75  	public void setType(Type type) {
76  		this.type = type;
77  	}
78  
79  }
