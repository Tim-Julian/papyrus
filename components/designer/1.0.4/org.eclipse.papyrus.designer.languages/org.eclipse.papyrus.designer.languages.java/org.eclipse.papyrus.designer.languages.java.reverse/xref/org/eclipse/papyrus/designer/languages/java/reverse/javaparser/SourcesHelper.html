30  package org.eclipse.papyrus.designer.languages.java.reverse.javaparser;
31  
32  import java.io.*;
33  
34  public class SourcesHelper {
35  
36      static String streamToString(InputStream in, String encoding){
37          if (encoding == null) {
38              return streamToString(in);
39          } else {
40              java.util.Scanner s = new java.util.Scanner(in, encoding).useDelimiter("\\A");
41              return s.hasNext() ? s.next() : "";
42          }
43      }
44  
45      static String streamToString(InputStream in){
46          java.util.Scanner s = new java.util.Scanner(in).useDelimiter("\\A");
47          return s.hasNext() ? s.next() : "";
48      }
49  
50      static InputStream stringToStream(String s, String encoding) throws UnsupportedEncodingException {
51          byte[] rawData = encoding != null ? s.getBytes(encoding) : s.getBytes();
52          return new ByteArrayInputStream(rawData);
53      }
54  
55      static String readerToString(Reader reader) throws IOException {
56          char[] arr = new char[8*1024]; // 8K at a time
57          StringBuilder buf = new StringBuilder();
58          int numChars;
59  
60          while ((numChars = reader.read(arr, 0, arr.length)) > 0) {
61              buf.append(arr, 0, numChars);
62          }
63  
64          return buf.toString();
65      }
66  
67      static Reader stringToReader(String s){
68          return new StringReader(s);
69      }
70  
71  }
