129 	private class SourcePrinter {
130 
131 		private int level = 0;
132 
133 		private boolean indented = false;
134 
135 		private final StringBuilder buf = new StringBuilder();
136 
137 		public void indent() {
138 			level++;
139 		}
140 
141 		public void unindent() {
142 			level--;
143 		}
144 
145 		private void makeIndent() {
146 			for (int i = 0; i < level; i++) {
147 				buf.append("    ");
148 			}
149 		}
150 
151 		public void print(String arg) {
152 			if (!indented) {
153 				makeIndent();
154 				indented = true;
155 			}
156 			buf.append(arg);
157 		}
158 
159 		public void printLn(String arg) {
160 			print(arg);
161 			printLn();
162 		}
163 
164 		public void printLn() {
165 			buf.append("\n");
166 			indented = false;
167 		}
168 
169 		public String getSource() {
170 			return buf.toString();
171 		}
172 
173 		@Override
174 		public String toString() {
175 			return getSource();
176 		}
177 	}

179 	private final SourcePrinter printer = new SourcePrinter();

181 	public String getSource() {
182 		return printer.getSource();
183 	}

185 	private void printModifiers(int modifiers) {
186 		if (ModifierSet.isPrivate(modifiers)) {
187 			printer.print("private ");
188 		}
189 		if (ModifierSet.isProtected(modifiers)) {
190 			printer.print("protected ");
191 		}
192 		if (ModifierSet.isPublic(modifiers)) {
193 			printer.print("public ");
194 		}
195 		if (ModifierSet.isAbstract(modifiers)) {
196 			printer.print("abstract ");
197 		}
198 		if (ModifierSet.isStatic(modifiers)) {
199 			printer.print("static ");
200 		}
201 		if (ModifierSet.isFinal(modifiers)) {
202 			printer.print("final ");
203 		}
204 		if (ModifierSet.isNative(modifiers)) {
205 			printer.print("native ");
206 		}
207 		if (ModifierSet.isStrictfp(modifiers)) {
208 			printer.print("strictfp ");
209 		}
210 		if (ModifierSet.isSynchronized(modifiers)) {
211 			printer.print("synchronized ");
212 		}
213 		if (ModifierSet.isTransient(modifiers)) {
214 			printer.print("transient ");
215 		}
216 		if (ModifierSet.isVolatile(modifiers)) {
217 			printer.print("volatile ");
218 		}
219 	}

222 		for (BodyDeclaration member : members) {
223 			printer.printLn();

226 		}

234 			}

242 				printer.print(" ");
243 			}
244 		}
245 	}

247 	private void printTypeArgs(List<Type> args, Object arg) {
248 		if (args != null) {
249 			printer.print("<");
250 			for (Iterator<Type> i = args.iterator(); i.hasNext();) {
251 				Type t = i.next();
252 				t.accept(this, arg);
253 				if (i.hasNext()) {
254 					printer.print(", ");
255 				}
256 			}
257 			printer.print(">");
258 		}
259 	}

262 		if (args != null) {
263 			printer.print("<");
264 			for (Iterator<TypeParameter> i = args.iterator(); i.hasNext();) {
265 				TypeParameter t = i.next();
266 				t.accept(this, arg);
267 				if (i.hasNext()) {
268 					printer.print(", ");
269 				}
270 			}
271 			printer.print(">");
272 		}
273 	}

275 	private void printArguments(List<Expression> args, Object arg) {
276 		printer.print("(");
277 		if (args != null) {
278 			for (Iterator<Expression> i = args.iterator(); i.hasNext();) {
279 				Expression e = i.next();
280 				e.accept(this, arg);
281 				if (i.hasNext()) {
282 					printer.print(", ");
283 				}
284 			}
285 		}
286 		printer.print(")");
287 	}

295 	@Override
296 	public void visit(CompilationUnit n, Object arg) {
297 		if (n.getPakage() != null) {
298 			n.getPakage().accept(this, arg);
299 		}
300 		if (n.getImports() != null) {
301 			for (ImportDeclaration i : n.getImports()) {
302 				i.accept(this, arg);
303 			}
304 			printer.printLn();
305 		}
306 		if (n.getTypes() != null) {
307 			for (Iterator<TypeDeclaration> i = n.getTypes().iterator(); i.hasNext();) {
308 				i.next().accept(this, arg);
309 				printer.printLn();
310 				if (i.hasNext()) {
311 					printer.printLn();
312 				}
313 			}
314 		}
315 	}

351 	@Override
352 	public void visit(ClassOrInterfaceDeclaration n, Object arg) {
353 		printJavadoc(n.getJavaDoc(), arg);
354 		printMemberAnnotations(n.getAnnotations(), arg);
355 		printModifiers(n.getModifiers());
356 
357 		if (n.isInterface()) {
358 			printer.print("interface ");
359 		} else {
360 			printer.print("class ");
361 		}
362 
363 		printer.print(n.getName());
364 
365 		printTypeParameters(n.getTypeParameters(), arg);
366 
367 		if (n.getExtends() != null) {
368 			printer.print(" extends ");
369 			for (Iterator<ClassOrInterfaceType> i = n.getExtends().iterator(); i.hasNext();) {
370 				ClassOrInterfaceType c = i.next();
371 				c.accept(this, arg);
372 				if (i.hasNext()) {
373 					printer.print(", ");
374 				}
375 			}
376 		}
377 
378 		if (n.getImplements() != null) {
379 			printer.print(" implements ");
380 			for (Iterator<ClassOrInterfaceType> i = n.getImplements().iterator(); i.hasNext();) {
381 				ClassOrInterfaceType c = i.next();
382 				c.accept(this, arg);
383 				if (i.hasNext()) {
384 					printer.print(", ");
385 				}
386 			}
387 		}
388 
389 		printer.printLn(" {");
390 		printer.indent();
391 		if (n.getMembers() != null) {
392 			printMembers(n.getMembers(), arg);
393 		}
394 		printer.unindent();
395 		printer.print("}");
396 	}

577 	public void visit(AssignExpr n, Object arg) {
578 		n.getTarget().accept(this, arg);
579 		printer.print(" ");
