30  /**
31   * A set of utilities to analyze JDT AST nodes.
32   * 
33   * @author cedric dumoulin
34   *
35   * @since 0.7.2
36   */
37  public class JdtAstUtils {
38  
39  	/**
40  	 * Get the short name of the underlying type
41  	 * @param requestedType
42  	 * @return
43  	 */
44  	static public String getTypeShortname(Type requestedType) {
45  		
46  		switch( requestedType.getNodeType()) {
47  		case ASTNode.SIMPLE_TYPE :
48  			return ((SimpleType)requestedType).getName().getFullyQualifiedName();
49  			
50  		case ASTNode.PARAMETERIZED_TYPE : 
51  			return getTypeShortname( ((ParameterizedType)requestedType).getType() );
52  			
53  		case ASTNode.ARRAY_TYPE : 
54  			return getTypeShortname( ((ArrayType)requestedType).getElementType() );
55  
56  		case ASTNode.PRIMITIVE_TYPE : 
57  			return ((PrimitiveType)requestedType).getPrimitiveTypeCode().toString();
58  			
59  		}
60  		return null;
61  	}
62  
63  	/**
64  	 * Get a field by its name.
65  	 * 
66  	 * @param typeDeclaration
67  	 * @param string
68  	 * @return
69  	 * @throws NotFoundException 
70  	 */
71  	static public FieldDeclaration getField(TypeDeclaration typeDeclaration, String fieldName) throws NotFoundException {
72  		
73  		FieldDeclaration[] fields = typeDeclaration.getFields();
74  		// Walk fields
75  		for( FieldDeclaration field : fields ) {
76  			// Walk variables declarations
77  			List<VariableDeclaration> varDecls = field.fragments();
78  			for( VariableDeclaration decl : varDecls) {
79  				if( fieldName.equals( decl.getName().getIdentifier() )) {
80  					return field;
81  				}
82  			}
83  			
84  		}
85  		throw new NotFoundException("Name " + fieldName + "' is not found in Field");
86  	}
87  
88  
89  }
